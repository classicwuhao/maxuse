63  public void export(Path exportFile, MSystem system, Set<MClass> classes, Set<EnumType> enums, Set<MAssociation> associations) throws IOException {
64  ModelFactory f = new ModelFactory();
65  MModel sourceModel = system.model();
66  MModel targetModel = f.createModel(sourceModel.name());
67 
68  copyAnnotations(sourceModel, targetModel);
69 
70  // Create "skeletons" for class
71  for (MClass sourceClass : classes) {
72  try {
73  MClass targetClass;
74 
75  if (sourceClass instanceof MAssociationClass)
76  targetClass = f.createAssociationClass(sourceClass.name(), sourceClass.isAbstract());
77  else
78  targetClass = f.createClass(sourceClass.name(), sourceClass.isAbstract());
79 
80  targetModel.addClass(targetClass);
81  copyAnnotations(sourceClass, targetClass);
82  } catch (MInvalidModelException e) { /* Cannot happen */ }
83  }
84 
85  // Create enumerations
86  for (EnumType sourceEnum : enums) {
87  try {
88  EnumType targetEnum = TypeFactory.mkEnum(sourceEnum.name(), sourceEnum.getLiterals());
89  targetModel.addEnumType(targetEnum);
90  copyAnnotations(sourceEnum, targetEnum);
91  } catch (MInvalidModelException e) { /* Cannot happen */ }
92  }
93 
94  // Create inheritance and attributes
95  for (MClass targetClass : targetModel.classes()) {
96  MClass sourceClass = sourceModel.getClass(targetClass.name());
97 
98  // Inheritance
99  for (MClassifier sourceParentClassifier : sourceClass.parents()) {
100  MClass sourceParentClass = (MClass)sourceParentClassifier;
101  MClass targetParentClass = findMostSpecificExportedType(sourceParentClass, targetModel);
102 
103  // Could be hidden!
104  if (targetParentClass != null) {
105  try {
106  targetModel.addGeneralization(f.createGeneralization(targetClass, targetParentClass));
107  } catch (MInvalidModelException e) { /* Cannot happen */ }
108  }
109  }
110 
111  // Attributes
112  for (MAttribute sourceAttribute : sourceClass.attributes()) {
113  Type attType = OCLCompiler.compileType(
114  targetModel, sourceAttribute.type().toString(),
115  "Export", NullPrintWriter.getInstance());
116 
117  // if type is not exported, don't export the attribute
118  if (attType != null) {
119  MAttribute targetAttribute = f.createAttribute(sourceAttribute.name(), attType);
120  try {
121  targetClass.addAttribute(targetAttribute);
122  } catch (MInvalidModelException e) { /* Cannot happen */ }
123  copyAnnotations(sourceAttribute, targetAttribute);
124  }
125  }
126 
127  // Operations
128  for (MOperation sourceOperation : sourceClass.operations()) {
129  boolean hasErrors = false;
130  Type resultType = null;
131 
132  if (sourceOperation.hasResultType()) {
133  resultType = OCLCompiler.compileType(
134  targetModel, sourceOperation.resultType().toString(),
135  "Export", NullPrintWriter.getInstance());
136 
137  // Result Type is not exported
138  if (resultType == null)
139  continue;
140  }
141 
142  VarDeclList targetArgs = new VarDeclList(false);
143 
144  // Build arguments
145  for (VarDecl arg : sourceOperation.paramList()) {
146  VarDecl v = cloneVarDecl(targetModel, arg);
147  if (v == null) {
148  hasErrors = true;
149  break;
150  }
151 
152  targetArgs.add(v);
153  }
154 
155  // If arg type is not present, continue to next operation
156  if (hasErrors)
157  continue;
158 
159  MOperation targetOperation = f.createOperation(sourceOperation.name(), targetArgs, resultType);
160 
161  try {
162  targetClass.addOperation(targetOperation);
163  } catch (MInvalidModelException e) { /* Cannot happen */ }
164 
165  copyAnnotations(sourceOperation, targetOperation);
166  }
167  }
168 
169  for (MAssociation sourceAssoc : associations) {
170  MAssociation targetAssoc;
171  if (sourceAssoc instanceof MAssociationClass) {
172  targetAssoc = (MAssociation)targetModel.getClass(sourceAssoc.name());
173  } else {
174  targetAssoc = f.createAssociation(sourceAssoc.name());
175  copyAnnotations(sourceAssoc, targetAssoc);
176  }
177 
178  for (MAssociationEnd sourceEnd : sourceAssoc.associationEnds()) {
179  List<VarDecl> targetQualifiers = new ArrayList<VarDecl>();
180  boolean hasErrors = false;
181 
182  for (VarDecl sourceQualifier : sourceEnd.getQualifiers()) {
183  VarDecl targetQualifier = cloneVarDecl(targetModel, sourceQualifier);
184  if (targetQualifier == null) {
185  hasErrors = true;
186  break;
187  }
188 
189  targetQualifiers.add(targetQualifier);
190  }
191 
192  if (hasErrors)
193  continue;
194 
195  MAssociationEnd targetEnd =
196  f.createAssociationEnd(
197  targetModel.getClass(sourceEnd.cls().name()),
198  sourceEnd.name(),
199  sourceEnd.multiplicity(),
200  sourceEnd.aggregationKind(),
201  sourceEnd.isOrdered(),
202  targetQualifiers);
203 
204  copyAnnotations(sourceEnd, targetEnd);
205  try {
206  targetAssoc.addAssociationEnd(targetEnd);
207  } catch (MInvalidModelException e) {
208 
209  }
210  }
211 
212  try {
213  targetModel.addAssociation(targetAssoc);
214  } catch (MInvalidModelException e) {
215 
216  }
217  }
218 
219  // Write result
220  try (OutputStream out = Files.newOutputStream(exportFile)) {
221  MMVisitor v = new MMPrintVisitor(new PrintWriter(out, true));
222  targetModel.processWithVisitor(v);
223  }
224  }

231  private MClass findMostSpecificExportedType(MClass sourceParentClass, MModel targetModel) {
232  MClass parent = targetModel.getClass(sourceParentClass.name());
233 
234  if (parent != null)
235  return parent;
236 
237  for (MClassifier otherParent : sourceParentClass.parents()) {
238  parent = findMostSpecificExportedType((MClass)otherParent, targetModel);
239  if (parent != null)
240  return parent;
241  }
242 
243  return null;
244  }

246  private VarDecl cloneVarDecl(MModel targetModel, VarDecl v) {
247  Type argType = OCLCompiler.compileType(
248  targetModel, v.type().toString(),
249  "Export", NullPrintWriter.getInstance());
250 
251  if (argType == null) {
252  return null;
253  }
254 
255  return new VarDecl(v.name(), argType);
256  }
