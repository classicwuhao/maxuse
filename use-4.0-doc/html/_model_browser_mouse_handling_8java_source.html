359  private void tryToFireStateChangeEvent( MouseEvent e ) {
360  if ( e.getModifiers() == InputEvent.BUTTON2_MASK ) {
361  if ( fRectangle != null && fRectangle.contains( e.getPoint() ) ) {
362  boolean highlight = false;
363  if ( fHighlightElements.containsKey( fElem ) ) {
364  highlight =
365  ((Boolean) fHighlightElements.get( fElem )).booleanValue();
366  if ( highlight ) {
367  highlight = false;
368  } else {
369  highlight = true;
370  }
371  fHighlightElements.put( fElem, Boolean.valueOf( highlight ) );
372  } else {
373  highlight = true;
374  fHighlightElements.put( fElem, Boolean.valueOf( highlight ) );
375  }
376  fMB.fireStateChanged( fElem, highlight );
377  }
378 
379  }
