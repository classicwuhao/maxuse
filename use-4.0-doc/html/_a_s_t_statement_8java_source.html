292  public MStatement generateStatement(
293  Context context,
294  MOperation operation) throws CompilationFailedException {
295 
296  // build symbol table from ...
297  SymbolTable symbolTable = new SymbolTable();
298  symbolTable.storeState(Options.explicitVariableDeclarations);
299  // ... parameters and ...
300  for (VarDecl p : operation.paramList()) {
301  symbolTable.setType(p.name(), p.type());
302  }
303  if (operation.resultType() != null && Options.explicitVariableDeclarations) {
304  symbolTable.setType("result", operation.resultType());
305  }
306  // ... self
307  symbolTable.setType("self", operation.cls());
308 
309  if (operation.hasResultType()) {
310  mustBindResultAs(operation.resultType());
311  }
312 
313  MStatement result = generateStatement(context, symbolTable);
314 
315  return result;

382  private Expression generateExpression(
383  ASTExpression expression,
384  Context context,
385  SymbolTable symbolTable) throws CompilationFailedException {
386 
387  Set<String> freeVariables = expression.getFreeVariables();
388 
389  Symtable newSymtable = new Symtable();
390  for (String name : freeVariables) {
391  if (!symbolTable.contains(name)) {
392  throw new CompilationFailedException(this, "Variable "
393  + StringUtil.inQuotes(name) + " in expression "
394  + StringUtil.inQuotes(expression.getStringRep())
395  + " is undefined.");
396  }
397 
398  if (symbolTable.isDirty(name)) {
399  ASTStatement cause = symbolTable.getCause(name);
400  throw new CompilationFailedException(
401  this,
402  "The type of variable " +
403  StringUtil.inQuotes(name) +
404  " in expression " +
405  StringUtil.inQuotes(expression.getStringRep()) +
406  " is uncertain due to possible assignment in statement " +
407  StringUtil.inQuotes(cause) +
408  " at line " +
409  cause.getSourcePosition().line() +
410  ", column " +
411  cause.getSourcePosition().column() +
412  "."
413  );
414  }
415 
416  try {
417  newSymtable.add(name, symbolTable.getType(name), null);
418  } catch (SemanticException e) {
419  // reason for the caught exception is adding a variable with a name
420  // that's already present. since we're adding the keys of a hashmap
421  // to an empty Symtable this can't happen.
422  // should this behavior change in the future, here's a hint to those
423  // who broke it...
424  throw new RuntimeException("unexpected exception", e);
425  }
426  }
427 
428  // if this statement gets parsed as part of an operation definition, we could
429  // possibly break it by fiddling around with the symtable. so let's just make sure
430  // the caller gets it back the way it was.
431  Symtable backup = context.varTable();
432  context.setVarTable(newSymtable);
433  try {
434  return expression.gen(context);
435  } catch (SemanticException e) {
436  throw new CompilationFailedException(
437  this,
438  "generation of expression " +
439  StringUtil.inQuotes(expression.getStringRep()) +
440  " failed, with following error:\n\n" +
441  e.getMessage());
442  } finally {
443  context.setVarTable(backup);
444  }
445  }

614  int numAssociationEnds = associationEnds.size();
615 
616  if (numAssociationEnds != numParticipants) {
617  throw new CompilationFailedException(this,
618  "A link for association"

622  + numAssociationEnds + " objects, found "
623  + numParticipants + ".");
624  }
625

627  // if their respective types are compatible
628  List<MRValue> result =
629  new ArrayList<MRValue>(numParticipants);
630 
631  for (int i = 0; i < numParticipants; ++i)
632  {
633  MAssociationEnd associationEnd =
634  associationEnds.get(i);
635 
636  MRValue participant = participants.get(i).generate(this);
637 
638  Type expectedType = associationEnd.cls();
639  Type foundType = participant.getType();
640 
641  if (!foundType.conformsTo(expectedType)) {
642 
643  throw new CompilationFailedException(
644  this,
645  "Participant " + (i + 1) + " of association " +
646  ((association instanceof MAssociationClass) ? "class " : "") +
647  StringUtil.inQuotes(association.name()) +
648  " must be of type " + StringUtil.inQuotes(expectedType) +
649  ", but " + StringUtil.inQuotes(participants.get(i)) +
650  " is of type " + StringUtil.inQuotes(foundType) + ".");
651  }
