298  public static void processArgs(String args[]) {
299  // parse command options
300  for (int i = 0; i < args.length; i++) {
301  if (args[i].startsWith("-") ) {
302  String arg = args[i].substring(1);
303 
304  if (arg.equals("c")) {
305  Options.compileOnly = true;
306  } else if (arg.equals("cp") ) {
307  Options.compileOnly = true;
308  Options.compileAndPrint = true;
309  } else if (arg.equals("disableCollectShorthand")) {
310  Options.disableCollectShorthand = true;
311  } else if (arg.equals("nogui")) {
312  Options.doGUI = false;
313  } else if (arg.equals("noplugins")) {
314  Options.doPLUGIN = false;
315  } else if (arg.equals("readlineTest")) {
316  Options.readlineTest = true;
317  } else if (arg.startsWith("H=")) {
318  try {
319  homeDir = Paths.get(arg.substring(2));
320  } catch (InvalidPathException e) {
321  System.err.println("Invalid path " + StringUtil.inQuotes(arg.substring(2)) + " for home directory specified.");
322  System.exit(1);
323  }
324  } else if (arg.equals("nr")) {
325  suppressWarningsAboutMissingReadlineLibrary = true;
326  } else if (arg.equals("q")) {
327  Options.quiet = true;
328  Options.doGUI = false;
329  } else if (arg.equals("qv")) {
330  Options.quiet = true;
331  Options.quietAndVerboseConstraintCheck = true;
332  Options.doGUI = false;
333  } else if (arg.equals("t")) {
334  Options.testMode = true;
335  } else if (arg.equals("v")) {
336  Log.setVerbose(true);
337  } else if (arg.equals("vt")) {
338  Log.setVerbose(true);
339  Log.setPrintTime(true);
340  } else if (arg.equals("V")) {
341  System.out.println("release " + RELEASE_VERSION);
342  System.exit(0);
343  } else if (arg.equals("implicitTypes")) {
344  Options.explicitVariableDeclarations = false;
345  } else if (arg.equals("debug")) {
346  setDebug(true);
347  } else if (arg.equals("p")) {
348  Log.setPrintStackTrace(true);
349  } else if (arg.equals("h")) {
350  printHelp();
351  } else if ( arg.equals("dimension") ) {
352  if ( args.length > i+2 ) {
353  setDimension( args[i+1], args[i+2] );
354  i = i+2;
355  }
356  } else if (arg.startsWith("oclAnyCollectionsChecks:")) {
357  String value = arg.substring("oclAnyCollectionsChecks:".length());
358  WarningType wt = WarningType.getType(value);
359  if(wt != null){
360  checkWarningsOclAnyInCollections = wt;
361  }
362  } else if (arg.startsWith("extendedTypeSystemChecks:")) {
363  String value = arg.substring("extendedTypeSystemChecks:".length());
364  WarningType wt = WarningType.getType(value);
365  if(wt != null){
366  checkWarningsUnrelatedTypes = wt;
367  }
368  } else {
369  System.out.println("invalid argument `" + arg
370  + "\', try `use -h' for help.");
371  System.exit(1);
372  }
373  } else if (specFilename == null ) {
374  specFilename = args[i];
375  } else if (cmdFilename == null ) {
376  cmdFilename = args[i];
377  } else {
378  System.err.println("extra argument `" + args[i] + "'");
379  System.exit(1);
380  }
381  }
382 
383  if (homeDir == null) {
384  // Try to get the home from Java
385  URL path = Options.class.getProtectionDomain().getCodeSource().getLocation();
386  try {
387  URI pathUri = path.toURI();
388  Path home = Paths.get(pathUri);
389  if (Files.isRegularFile(home)) {
390  // resolve jar file path
391  home = home.getParent();
392  }
393  // resolve lib/ folder path, assuming default folder structure
394  homeDir = home.getParent();
395  } catch (URISyntaxException e) { }
396  }
397 
398  if (homeDir == null ) {
399  System.err.println("Missing path to USE installation, try `use -h' for help.");
400  System.exit(1);
401  }
402 
403  setLastDirectory(homeDir);
404  iconDir = homeDir.resolve("images");
405  pluginDir = homeDir.resolve("lib").resolve("plugins");
406 
407  if (quiet && (specFilename == null || cmdFilename == null) ) {
408  System.err
409  .println("Need specification file and command file with option -q,"
410  + LINE_SEPARATOR + "try `use -h' for help.");
411  System.exit(1);
412  }
413 
414  // load property files
415  initProperties(homeDir);
416 
417  // args ok, print welcome message if in interactive mode
418  if (!compileOnly && !quiet ) {
419  Log.println("USE version " + Options.RELEASE_VERSION + ", " + Options.COPYRIGHT);
420  }
421  }

444  private static void initProperties(Path useHome) {
445  props = new TypedProperties(System.getProperties());
446 
447  // load the system properties
448  Path propFile;
449  propFile = useHome.resolve("etc").resolve(USE_PROP_FILE);
450  if (!Files.isReadable(propFile)) {
451  System.err.println("property file `" + propFile.toString()
452  + "' not found. Use -H to set the "
453  + "home of the use installation");
454  System.exit(1);
455  }
456  loadProperties(propFile);
457 
458  // load user properties if found
459  propFile = Paths.get(props.getProperty("user.dir", null), USER_PROP_FILE);
460  if ( Files.isReadable(propFile)) {
461  loadProperties(propFile);
462  } else {
463  propFile = Paths.get(props.getProperty("user.home", null), USER_PROP_FILE);
464  if ( Files.isReadable(propFile) )
465  loadProperties(propFile);
466  }
467 
468  MONITOR_ASPECT_TEMPLATE = useHome.resolve(props.getProperty(MONITOR_ASPECT_TEMPLATE_P, "etc/USEMonitor.java.template"));
469 
470  EVAL_NUMTHREADS = props.getRangeIntProperty(EVAL_NUMTHREADS_P,
471  EVAL_NUMTHREADS, 1, Integer.MAX_VALUE);
472 
473  PRINT_PAGEFORMAT_WIDTH = props.getRangeDoubleProperty(
474  PRINT_PAGEFORMAT_WIDTH_P, PRINT_PAGEFORMAT_WIDTH, 72.0,
475  72.0 * 1000);
476 
477  PRINT_PAGEFORMAT_HEIGHT = props.getRangeDoubleProperty(
478  PRINT_PAGEFORMAT_HEIGHT_P, PRINT_PAGEFORMAT_HEIGHT, 72.0,
479  72.0 * 1000);
480 
481  PRINT_PAGEFORMAT_ORIENTATION = props.getStringEnumProperty(
482  PRINT_PAGEFORMAT_ORIENTATION_P, "portrait", new String[] {
483  "landscape", "portrait", "seascape" });
484 
485  USE_HISTORY_PATH = props.getProperty("user.home", ".")
486  + props.getProperty("file.separator") + USE_HISTORY_PATH;
487 
488  // add our properties to the system properties so that
489  // Font.getFont() works with our application specific fonts.
490  System.setProperties(props);
491  }
