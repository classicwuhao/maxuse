60  public GInstruction gen(Context ctx) throws SemanticException {
61  GLoop loop = new GLoop();
62  ctx.varTable().enterScope();
63 
64  try {
65  VarDecl vardecl = fDecl.gen(ctx);
66  if (ctx.varTable().lookup(fDecl.name().getText()) != null) {
67  String err = "Variable `" + fDecl.name().getText() + "' was " +
68  "previously declared.";
69  throw new SemanticException(fDecl.name(), err);
70  }
71 
72  ctx.varTable().add( fDecl.name(), vardecl.type() );
73  // does NOT throw an exception when variable is redefined
74  // because the context scope has changed!!
75  ctx.loopVarNames().add( fDecl.name().getText() );
76  loop.setDecl( vardecl );
77 
78  GInstruction sequence = fSequence.gen(ctx);
79  if (! (sequence instanceof GValueInstruction) ||
80  ! ((GValueInstruction) sequence).type().isTypeOfSequence() ) {
81  String err = "`" + sequence + "' must be of type Sequence.";
82  // Fixme: The position of the error must be the
83  // position of fSequence.
84  throw new SemanticException( fName, err );
85  }
86 
87  if (! ((SequenceType)((GValueInstruction) sequence).type())
88  .elemType().conformsTo(vardecl.type()) ) {
89  String err = "The elements of `"+sequence+"' must be of " +
90  "type " + vardecl.type() + " (or a subtype)";
91  throw new SemanticException( fDecl.name(), err );
92  }
93 
94  loop.setSequenceInstr( (GValueInstruction) sequence );
95 
96  for (ASTGInstruction ins : fInstructions) {
97  loop.addInstruction( ins.gen(ctx) );
98  }
99  } catch (SemanticException ex ) {
100  ctx.reportError(ex);
101  }
102  ctx.loopVarNames().remove( fDecl.name().getText() );
103  ctx.varTable().exitScope();
104  return loop;
105  }
106 }
