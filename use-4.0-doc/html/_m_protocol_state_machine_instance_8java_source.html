126  public boolean validOperationCall(EvalContext ctx, MOperationCall operationCall, Map<MRegion, Set<MTransition>> possibleTransitions) {
127 
128  boolean foundValid = false;
129 
130  for (Map.Entry<MRegion, MState> entry : this.currentRegionsState.entrySet()) {
131  if (entry.getValue() instanceof MFinalState) {
132  return false;
133  }
134 
135  Set<MTransition> outgoingTransitions = entry.getValue().getOutgoing();
136  possibleTransitions.put(entry.getKey(), new LinkedHashSet<MTransition>());
137 
138  for (MTransition t : outgoingTransitions) {
139  MProtocolTransition pt = (MProtocolTransition)t;
140  // some transitions don't have an operation, e. g., create
141  if (pt.getReferred() == null) continue;
142 
143  boolean isValid = false;
144 
145  if ( operationCall.getOperation().equals(pt.getReferred())
146  || operationCall.getOperation().isValidOverrideOf(pt.getReferred())) {
147  // Check the guard
148  if (pt.hasGuard()) {
149  Evaluator oclEvaluator = new Evaluator();
150 
151  Value evalResult =
152  oclEvaluator.eval(t.getGuard(),
153  ctx.postState(), ctx.varBindings());
154 
155  if (evalResult.isBoolean() && ((BooleanValue)evalResult).isTrue()) {
156  isValid = true;
157  }
158 
159  } else {
160  isValid = true;
161  }
162  }
163 
164  if (isValid) {
165  possibleTransitions.get(entry.getKey()).add(t);
166  foundValid = foundValid || isValid;
167  }
168  }
169  }
170 
171  return foundValid;
172  }

251  public TransitionResult evaluateTransition(MTransition t, EvalContext ctx, MOperationCall operationCall) {
252 
253  TransitionResult result = new TransitionResult(t);
254 
255  if (!t.getSource().equals(this.currentRegionsState.get(t.getContainer()))) {
256  throw new IllegalArgumentException(
257  "Cannot execute transition "
258  + StringUtil.inQuotes(t.toString())
259  + ", because the protocol state machine is not in the source state of the transition");
260  }
261 
262  MProtocolTransition pt = (MProtocolTransition)t;
263  MState targetState = (MState)t.getTarget();
264 
265  if (pt.hasPostCondition()) {
266  Evaluator oclEvaluator = new Evaluator();
267 
268  Value evalResult =
269  oclEvaluator.eval(pt.getPostCondition(), ctx.preState(), ctx.postState(), ctx.varBindings());
270 
271  result.setPostConditionResult(evalResult);
272  }
273 
274  if (targetState.getStateInvariant() != null) {
275  Evaluator oclEvaluator = new Evaluator();
276  VarBindings b = new VarBindings();
277  b.push("self", new ObjectValue(stateMachine.getContext(), contextObject));
278  Value evalResult =
279  oclEvaluator.eval(targetState.getStateInvariant(), ctx.postState(), ctx.varBindings());
280 
281  result.setStateInvariantResult(evalResult);
282  }
283 
284  return result;
285  }

320  public boolean checkStateInvariant(MSystemState systemState, @NonNull PrintWriter errOut) {
321  boolean valid = true;
322  Evaluator evaluator = new Evaluator();
323  VarBindings bindings = getVarBindings(systemState);
324  MState s;
325 
326  for (Map.Entry<MRegion,MState> entry : currentRegionsState.entrySet()) {
327  s = entry.getValue();
328 
329  if (s == null || s.getStateInvariant() == null) continue;
330 
331  Value result = evaluator.eval(s.getStateInvariant(), systemState, bindings);
332 
333  if (!result.isBoolean() || ((BooleanValue)result).isFalse()) {
334  valid = false;
335  errOut.print("State invariant violation in state ");
336  errOut.print(StringUtil.inQuotes(s.name()));
337  errOut.print(" of psm ");
338  errOut.print(StringUtil.inQuotes(this.stateMachine.toString()));
339  errOut.print(" for object ");
340  errOut.println(this.contextObject.toString());
341  }
342  }
343 
344  return valid;
345  }

353  public void determineState(MSystemState systemState, PrintWriter out) {
354  Set<MState> possibleStates = new HashSet<MState>();
355 
356  Evaluator evaluator = new Evaluator();
357  VarBindings bindings = getVarBindings(systemState);
358 
359  for (MRegion r : stateMachine.getRegions()) {
360  possibleStates.clear();
361 
362  for (MVertex v : r.getSubvertices()) {
363  if (v instanceof MState) {
364  MState s = (MState)v;
365  if (s.getStateInvariant() == null) continue;
366 
367  Value result = evaluator.eval(s.getStateInvariant(), systemState, bindings);
368 
369  if (result.isBoolean() && ((BooleanValue)result).isTrue())
370  possibleStates.add(s);
371  }
372  }
373 
374  if (possibleStates.isEmpty()) {
375  out.println("Could not find a valid state for psm "
376  + stateMachine.toString() + " for "
377  + contextObject.name());
378  currentRegionsState.put(r, null);
379  } else if (possibleStates.size() > 1) {
380  out.println("Found multiple valid states for psm "
381  + stateMachine.toString() + " for "
382  + contextObject.name());
383  out.println("Valid states:");
384  for (MState s : possibleStates) {
385  out.print(" ");
386  out.println(s.name());
387  }
388  currentRegionsState.put(r, null);
389  } else {
390  currentRegionsState.put(r, possibleStates.iterator().next());
391  }
392  }
393  }
