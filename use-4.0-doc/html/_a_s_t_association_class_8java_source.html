132  public MAssociationClass genAssociation( Context ctx )
133  throws SemanticException {
134 
135  int kind = MAggregationKind.NONE;
136  if ( fKind != null ) {
137  String kindname = fKind.getText();
138 
139  if ( kindname.equals( "aggregation" ) )
140  kind = MAggregationKind.AGGREGATION;
141  else if ( kindname.equals( "composition" ) )
142  kind = MAggregationKind.COMPOSITION;
143  }
144 
145  try {
146  for (ASTAssociationEnd ae : fAssociationEnds) {
147  // kind of association determines kind of first
148  // association end
149  MAssociationEnd aend = ae.gen( ctx, kind );
150  fAssocClass.addAssociationEnd( aend );
151 
152  // further ends are plain ends
153  kind = MAggregationKind.NONE;
154  }
155  } catch ( MInvalidModelException ex ) {
156  throw new SemanticException( fName,
157  "In "
158  + MAggregationKind.name( fAssocClass.aggregationKind() )
159  + " `" +
160  fAssocClass.name() + "': " +
161  ex.getMessage() );
162  }
163 
164  return fAssocClass;
165  }
