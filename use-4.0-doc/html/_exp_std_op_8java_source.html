169  private static void checkOclAnyCollectionsWarning(OpGeneric op, Type[] params, Type resultType, WarningType warningType) throws ExpInvalidException {
170  Type sourceType = params[0];
171 
172  if (sourceType.isKindOfCollection(VoidHandling.EXCLUDE_VOID)) {
173  CollectionType sourceCollectionType = (CollectionType)sourceType;
174  Type sourceElementType = sourceCollectionType.elemType();
175 
176  while (sourceElementType.isKindOfCollection(VoidHandling.EXCLUDE_VOID)) {
177  sourceElementType = ((CollectionType)sourceElementType).elemType();
178  }
179 
180  if (sourceElementType.isTypeOfOclAny()) return;
181 
182  Type resultElementType = resultType;
183  while (resultElementType.isKindOfCollection(VoidHandling.EXCLUDE_VOID)) {
184  resultElementType = ((CollectionType)resultElementType).elemType();
185  }
186 
187  if (resultElementType.isTypeOfOclAny()) {
188  StringBuilder paramTypes = new StringBuilder();
189  for (int index = 1; index < params.length; ++index) {
190  if (index > 1) {
191  paramTypes.append(",");
192  }
193  paramTypes.append(params[index].toString());
194  }
195 
196  String message = "Operation call "
197  + StringUtil.inQuotes(sourceType.toString() + "->"
198  + op.name() + "(" + paramTypes.toString() + ")")
199  + " results in type "
200  + StringUtil.inQuotes(resultType.toString()) + "." + StringUtil.NEWLINE
201  + "This may lead to unexpected behavior." + StringUtil.NEWLINE
202  + "You can change this check using the -oclAnyCollectionsChecks switch.";
203 
204  if (warningType.equals(WarningType.ERROR)) {
205  throw new ExpInvalidException(message);
206  } else {
207  Log.warn("Warning: " + message);
208  }
209  }
210  }

278  public Value eval(EvalContext ctx) {
279  ctx.enter(this);
280  Value res = null;
281 
282  // Boolean operations need special treatment of undefined
283  // arguments. Also, short-circuit evaluation may be used to
284  // speed up the evaluation process.
285  if (getOperation().isBooleanOperation()) {
286  res = ((BooleanOperation) getOperation()).evalWithArgs(ctx, fArgs);
287  } else {
288  final Value argValues[] = new Value[fArgs.length];
289  final int opKind = getOperation().kind();
290 
291  Value v;
292 
293  for (int i = 0; i < fArgs.length && res == null; i++) {
294  argValues[i] = v = fArgs[i].eval(ctx);
295  // if any of the arguments is undefined, the result
296  // depends on the kind of operation we are about to
297  // call.
298  if (v.isUndefined()) {
299  switch (opKind) {
300  case OpGeneric.OPERATION:
301  // strict evaluation, result is undefined, no
302  // need to call the operation's eval() method.
303  res = UndefinedValue.instance;
304  break;
305  case OpGeneric.SPECIAL:
306  // these operations handle undefined arguments
307  // themselves
308  break;
309  default:
310  throw new RuntimeException(
311  "Unexpected operation kind: " + opKind);
312  }
313  }
314  }
315  if (res == null) {
316  try {
317  res = getOperation().eval(ctx, argValues, type());
318  } catch (ArithmeticException ex) {
319  // catch e.g. division by zero
320  res = UndefinedValue.instance;
321  }
322  }
323  }
324  ctx.exit(this, res);
325  return res;
326  }
