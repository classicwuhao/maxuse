169  public void update() {
170  fClassComboBox.removeActionListener(fClassComboBoxActionListener);
171 
172  MSystemState state = fSystem.state();
173  List<String> namesOfClassesWithObjects = new ArrayList<String>();
174  String previouslySelectedItem = (String)fClassComboBox.getSelectedItem();
175 
176  for (MClass cls : fSystem.model().classes()) {
177  if (!state.objectsOfClassAndSubClasses(cls).isEmpty())
178  namesOfClassesWithObjects.add(cls.name());
179  }
180 
181  if (namesOfClassesWithObjects.isEmpty()) {
182  namesOfClassesWithObjects.add(NO_OBJECTS_AVAILABLE);
183  fClassComboBox.setEnabled(false);
184  } else {
185  fClassComboBox.setEnabled(true);
186  }
187 
188  String[] clsNames = namesOfClassesWithObjects.toArray(new String[0]);
189  Arrays.sort(clsNames);
190 
191  // create combo box with available objects
192  fClassComboBox.setModel(new DefaultComboBoxModel<String>(clsNames));
193 
194  fClassComboBox.addActionListener(fClassComboBoxActionListener);
195 
196  // try to keep selection
197  if (previouslySelectedItem != null) {
198  fClassComboBox.setSelectedItem(previouslySelectedItem);
199  ((SelectionObjectTableModel) fTableModel).setSelected(fSystem.model().getClass(previouslySelectedItem));
200  } else {
201  fClassComboBox.setSelectedIndex(0);
202  }
203  }
