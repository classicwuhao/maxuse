178  @Override
179  public Point2D doCalculatePosition(PlaceableNode owner) {
180 
181  Point2D.Double rel = new Point2D.Double();
182  Rectangle2D bounds = relativeNode.getBounds();
183 
184  if (corner == Direction.WEST || corner == Direction.EAST) {
185  rel.y = bounds.getCenterY();
186  } else {
187  rel.y = (corner.isLocatedNorth() ? bounds.getMinY() : bounds.getMaxY());
188  }
189 
190  if (corner == Direction.NORTH || corner == Direction.SOUTH) {
191  rel.x = bounds.getCenterX();
192  } else {
193  rel.x = (corner.isLocatedWest() ? bounds.getMinX() : bounds.getMaxX());
194  }
195 
196  double deltaXSign = (corner.isLocatedWest() ? -1 : 1);
197  double deltaYSign = (corner.isLocatedNorth() ? -1 : 1);
198  double absDeltaX = 0;
199  double absDeltaY = 0;
200 
201  if (basisDeltaX == DeltaBasis.ABSOLUTE) {
202  absDeltaX = deltaX + ((corner.isLocatedEast() || corner.isLocatedWest()) ? owner.getWidth() / 2 : 0);
203  } else if (basisDeltaX == DeltaBasis.RELATIVE) {
204  double width = (widthCalculation == null ? bounds.getWidth() : widthCalculation.get());
205  absDeltaX = width * deltaX;
206  }
207  rel.x += Math.min(absDeltaX, maxDeltaX) * deltaXSign;
208 
209  if (basisDeltaY == DeltaBasis.ABSOLUTE) {
210  absDeltaY = deltaY + ((corner.isLocatedNorth() || corner.isLocatedSouth()) ? owner.getHeight() / 2 : 0);
211  } else if (basisDeltaY == DeltaBasis.RELATIVE) {
212  double height = (heightCalculation == null ? bounds.getHeight() : heightCalculation.get());
213  absDeltaY = height * deltaY;
214  }
215  rel.y += Math.min(absDeltaY, maxDeltaY) * deltaYSign;
216 
217  return rel;
218  }
