89  private void createSubSequence(GConfiguration conf,IGCollector collector)
90  throws GEvaluationException {
91  int origSize = fSequence.size();
92  if (wantedSize < 0)
93  collector.invalid( "Can't execute `" + fInstr + "', because `"
94  + fInstr.integerInstr() + "' has been "
95  + "evaluated to a negative integer.");
96  else if (wantedSize > origSize)
97  collector.invalid( "Can't execute `" + fInstr + "', because `"
98  +fInstr.integerInstr() + "' is greater than "
99  +"the size of `"+ fInstr.sequenceInstr() +"'.");
100  else {
101  List<Integer> mappings = new ArrayList<Integer>(origSize);
102  Collection<Value> values = new ArrayList<Value>(wantedSize);
103 
104  for (int k=0; k < origSize; k++)
105  mappings.add( Integer.valueOf(k) );
106 
107  for (int k=0; k < wantedSize; k++) {
108  int r = conf.random().nextInt(origSize-k);
109  values.add( fSequence.get(mappings.get(r).intValue() ) );
110  mappings.remove(r);
111  }
112 
113  Value subVal = new SequenceValue(fSequence.elemType(), values);
114 
115  if (collector.doDetailPrinting())
116  collector.detailPrintWriter().println(
117  "`" + fInstr + "' == " + subVal);
118 
119  fCaller.feedback( conf, subVal, collector );
120  }
