211  @Override
212  protected void onDraw( Graphics2D g ) {
213  Rectangle2D.Double currentBounds = getRoundedBounds();
214 
215  if (g.getClipBounds() != null &&
216  !Util.enlargeRectangle(currentBounds, 10).intersects(g.getClipBounds())) {
217  return;
218  }
219 
220  int y;
221 
222  FontMetrics fm;
223 
224  Font oldFont = g.getFont();
225  if ( cls().isAbstract() ) {
226  g.setFont( oldFont.deriveFont( Font.ITALIC ) );
227  }
228 
229  fm = g.getFontMetrics();
230 
231  int labelWidth = fm.stringWidth( fLabel );
232 
233  if ( isSelected() ) {
234  g.setColor( fOpt.getNODE_SELECTED_COLOR() );
235  } else {
236  if (getColor() != null)
237  g.setColor( getColor() );
238  else
239  g.setColor( fOpt.getNODE_COLOR() );
240  }
241 
242  g.fill( currentBounds );
243 
244  double x = getCenter().getX();
245  x -= labelWidth / 2;
246  y = (int)currentBounds.getY() + fm.getAscent() + VERTICAL_INDENT;
247  g.setColor( fOpt.getNODE_LABEL_COLOR() );
248  // We know that the name fits, because we require this size
249  g.drawString( fLabel, Math.round(x), y );
250 
251  y += VERTICAL_INDENT + fm.getDescent();
252 
253  // resetting font and fontMetrics if the class was abstract
254  g.setFont( oldFont );
255  fm = g.getFontMetrics();
256 
257  Line2D.Double lineAttrDivider = new Line2D.Double(currentBounds.getX(), y, currentBounds.getMaxX(), y);
258  Line2D.Double lineOpDivider = new Line2D.Double(currentBounds.getX(), y, currentBounds.getMaxX(), y);
259 
260  if ( fOpt.isShowAttributes() ) {
261  // compartment divider
262  lineAttrDivider.y1 = y;
263  lineAttrDivider.y2 = y;
264 
265  if (fAttributes.isEmpty()) {
266  y += 2 * VERTICAL_INDENT;
267  } else {
268  Rectangle2D.Double attributeBounds = new Rectangle2D.Double(currentBounds.x, y, currentBounds.width, currentBounds.height);
269 
270  if (fOpt.isShowOperations())
271  attributeBounds.height = currentBounds.getMaxY() - y - VERTICAL_INDENT - Util.getLineHeight(fm);
272 
273  y = drawCompartment(g, y, fAttrValues, fAttrColors, attributeBounds);
274  }
275  }
276 
277  if ( fOpt.isShowOperations() ) {
278  // compartment divider
279  lineOpDivider.y1 = y;
280  lineOpDivider.y2 = y;
281 
282  y = drawCompartment(g, y, fOprSignatures, fOperationColors, currentBounds);
283  }
284 
285  g.setColor( fOpt.getNODE_FRAME_COLOR() );
286  g.draw( currentBounds );
287 
288  if ( fOpt.isShowAttributes() ) {
289  g.draw(lineAttrDivider);
290  }
291  if ( fOpt.isShowOperations() ) {
292  g.draw(lineOpDivider);
293  }
294  }
