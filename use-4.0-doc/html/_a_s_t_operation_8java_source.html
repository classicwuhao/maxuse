89  public MOperation genSignature(Context ctx)
90  throws SemanticException
91  {
92  // map params to VarDeclList
93  VarDeclList varDeclList = new VarDeclList(false);
94 
95  for (ASTVariableDeclaration astDecl : fParamList) {
96  VarDecl decl = astDecl.gen(ctx);
97  try {
98  varDeclList.add(decl);
99  } catch (IllegalArgumentException ex) {
100  throw new SemanticException(astDecl.name(), "Redefinition of `" +
101  decl.name() + "'.");
102  }
103  }
104  Type resultType = null;
105  if (fType == null) {
106  if (this.fExpr != null) {
107  throw new SemanticException(fName, "Missing return type for OCL query operation " + StringUtil.inQuotes(fName.getText()) + ".");
108  }
109  } else {
110  resultType = fType.gen(ctx);
111  }
112 
113  fOperation = ctx.modelFactory().createOperation(fName.getText(), varDeclList, resultType);
114 
115  // sets the line position of the USE-Model in this attribute
116  fOperation.setPositionInModel( fName.getLine() );
117 
118  // HACK: set a temporary expression here, otherwise generating
119  // the body expression will fail if the body contains a
120  // recursive call to this operation, or a forward reference is
121  // made to another operation
122  if (fExpr != null ) {
123  fOperation.setTempExpression();
124  } else if (fStatement != null) {
125  fOperation.setStatement(MEmptyStatement.getInstance());
126  }
127 
128  this.genAnnotations(fOperation);
129 
130  return fOperation;
131  }

133  public void genFinal(Context ctx) throws SemanticException
134  {
135  // fOperation is null if genSignature exited with an Exception
136  if (fOperation == null )
137  return;
138 
139  // enter parameters into scope of expression
140  Symtable vars = ctx.varTable();
141  vars.enterScope();
142 
143  for (ASTVariableDeclaration astDecl : fParamList) {
144  VarDecl decl = astDecl.gen(ctx);
145  vars.add(astDecl.name(), decl.type());
146  }
147 
148 
150  // BEGIN SOIL EXTENSION
151  if (fStatement != null) {
152  fOperation.setStatement(genStatement(ctx));
153  }
154  // END SOIL EXTENSION
156 
157  try {
158  if (fExpr != null ) {
159  Expression expr = fExpr.gen(ctx);
160  fOperation.setExpression(expr);
161  }
162 
163 
164  for (ASTPrePostClause ppc : fPrePostClauses) {
165  ppc.gen(ctx, ctx.currentClass(), fOperation);
166  }
167 
168 
169  } catch (MInvalidModelException ex) {
170  throw new SemanticException(fName, ex);
171  } finally {
172  vars.exitScope();
173  }
174  }
