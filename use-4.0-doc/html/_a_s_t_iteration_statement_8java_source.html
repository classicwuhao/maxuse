99  protected MStatement generateStatement() throws CompilationFailedException {
100 
101  Expression range = generateExpression(fRange);
102  if (!range.type().isKindOfCollection(VoidHandling.INCLUDE_VOID)) {
103  throw new CompilationFailedException(this, "Expression "
104  + StringUtil.inQuotes(fRange.getStringRep())
105  + " is expected to be of type "
106  + StringUtil.inQuotes("Collection") + ", found "
107  + StringUtil.inQuotes(range.type()) + ".");
108  }
109 
110  Type iterVarType = ((CollectionType)range.type()).elemType();
111 
112  fSymtable.storeState();
113  fSymtable.setType(fIterVarName, iterVarType);
114  MStatement body = generateStatement(fBody);
115  fSymtable.restoreState(this);
116 
117  if (fBody.assigns(fIterVarName)) {
118  throw new CompilationFailedException(this,
119  "There is possible assignment to the iteration variable " +
120  StringUtil.inQuotes(this.getIterVarName()) +
121  ". This is prohibited.");
122  }
123 
124  // assigned(iteration) = assigned(body)
125  if (!fSymtable.isExplicit()) {
126  fAssignedSet.add(fBody.fAssignedSet);
127  fAssignedSet.add(fIterVarName, iterVarType);
128  }
129 
130  // range is empty!
131  if (iterVarType.isTypeOfVoidType()) {
132  return MEmptyStatement.getInstance();
133  } else {
134  return new MIterationStatement(fIterVarName, range, body);
135  }
136  }
