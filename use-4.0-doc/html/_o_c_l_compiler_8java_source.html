196  parser.init(errHandler);

198 
199 
200  try {
201  // Parse the input expression
202  ASTExpression astExpr = parser.expressionOnly();
203

205

207  Context ctx = new Context(inName, err, globalBindings, null);
208  ctx.setModel(model);
