125  private final Value eval0(EvalContext ctx,
126  int nesting,
127  CollectionValue rangeVal,
128  Value accuVal)
129  {
130  // loop over range elements
131  for (Value elemVal : rangeVal) {
132 
133  // bind element variable to range element, if variable was
134  // declared
135  if (! fElemVarDecls.isEmpty() )
136  ctx.pushVarBinding(fElemVarDecls.varDecl(nesting).name(), elemVal);
137 
138  if (! fElemVarDecls.isEmpty() && nesting < fElemVarDecls.size() - 1) {
139  // call recursively to iterate over range while
140  // assigning each value to each element variable
141  // eventually
142  accuVal = eval0(ctx, nesting + 1, rangeVal, accuVal);
143  } else {
144  // bind accumulator variable
145  ctx.pushVarBinding(fAccuInitializer.name(), accuVal);
146 
147  // evaluate iterate expression and assign new accumulator value
148  accuVal = fQueryExp.eval(ctx);
149 
150  ctx.popVarBinding();
151  }
152 
153  if (! fElemVarDecls.isEmpty() )
154  ctx.popVarBinding();
155  }
156  return accuVal;
157  }
