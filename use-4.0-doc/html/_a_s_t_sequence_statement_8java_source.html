129  private void checkS2Validity(ASTStatement s2) throws CompilationFailedException {
130 
131  if (fBoundSet.isEmpty() || s2.fAssignedSet.isEmpty()) {
132  return;
133  }
134 
135  VariableSet boundS1 = fBoundSet;
136  VariableSet assignedUnboundS2 = new VariableSet(s2.fAssignedSet);
137  assignedUnboundS2.remove(s2.fBoundSet);
138 
139  for (String name : boundS1.getCommonNames(assignedUnboundS2)) {
140  Set<Type> assignedTypes = assignedUnboundS2.getTypes(name);
141  Set<Type> boundTypes = boundS1.getTypes(name);
142 
143  for (Type assignedType : assignedTypes) {
144  for (Type boundType : boundTypes) {
145  if (!assignedType.conformsTo(boundType)) {
146  ASTStatement cause = fSymtable.getCause(name);
147  throw new CompilationFailedException(
148  this,
149  "Statement "
150  + StringUtil.inQuotes(cause)
151  + " at line "
152  + cause.getSourcePosition().line()
153  + ", column "
154  + cause.getSourcePosition().column()
155  + " possibly sets variable "
156  + StringUtil.inQuotes(name)
157  + " to type "
158  + StringUtil.inQuotes(boundType)
159  + ". This is prohibited, since this is not a subtype "
160  + "of its current type "
161  + StringUtil.inQuotes(assignedType)
162  + ".");
163  }
164  }
165  }
