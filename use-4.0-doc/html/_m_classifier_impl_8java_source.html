129  @Override
130  public Type getLeastCommonSupertype(Type other) {
131  if (other.isTypeOfVoidType()) return this;
132 
133  Type cType = TypeFactory.mkOclAny();
134 
135  // Object type and build in type have OclAny
136  if (!other.isKindOfClassifier(VoidHandling.EXCLUDE_VOID)) {
137  // Collections are no subtypes of OclAny
138  if (other.isKindOfCollection(VoidHandling.EXCLUDE_VOID)) {
139  return null;
140  } else {
141  return TypeFactory.mkOclAny();
142  }
143  }
144 
145  MClassifier oTypeThis = this;
146  MClassifier oTypeOther = (MClassifier)other;
147 
148  Set<MClassifier> superClassesThis = new HashSet<MClassifier>();
149  superClassesThis.add(oTypeThis);
150 
151  Set<? extends MClassifier> allP = oTypeOther.allParents();
152  Set<MClassifier> allSuperClassesOther = new HashSet<MClassifier>(allP);
153  allSuperClassesOther.add(oTypeOther);
154 
155  Set<MClassifier> commonClasses;
156 
157  while (!superClassesThis.isEmpty()) {
158  commonClasses = new HashSet<MClassifier>(superClassesThis);
159  commonClasses.retainAll(allSuperClassesOther);
160 
161  if (commonClasses.isEmpty()) {
162  Set<MClassifier> nextIteration = new HashSet<MClassifier>();
163  for (MClassifier cls : superClassesThis) {
164  for (MClassifier cf : cls.parents()) {
165  nextIteration.add(cf);
166  }
167  }
168 
169  superClassesThis = nextIteration;
170  } else {
171  // NOTE: We use the first class common to both!
172  cType = commonClasses.iterator().next();
173  break;
174  }
175  }
176 
177  return cType;
178  }
