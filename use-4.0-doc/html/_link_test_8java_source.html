261  // for counting the number of Instances between names
262  int cnt = 0;
263 
264  Iterator<MLink> linkSet = system.state().linksOfAssociation( assocClass ).links().iterator();
265  while ( linkSet.hasNext() ) {
266  MLink link = ( MLink ) linkSet.next();
267  Iterator<MObject> it = link.linkedObjects().iterator();
268  MObject first = it.next();
269  MObject second = it.next();
270 
271  if ( ( first.name().equals( "p1" ) && second.name().equals( "c1" ) )
272  || ( first.name().equals( "c1" ) && second.name().equals( "p1" ) ) ) {
273  cnt++;
274  }
275  }
276
