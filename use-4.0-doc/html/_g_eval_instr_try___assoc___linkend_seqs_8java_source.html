101  Value value,
102  IGCollector collector ) throws GEvaluationException {

108

143  GConfiguration conf,
144  IGCollector collector) throws GEvaluationException {
145 
146  // 62 Links are 2^62 combinations, which can be represented by a long
147  // (bit i == 1) -> link i exists, (bit i == 0) -> link i does not exist
148  // By the way, the search of a valid state within 2^62 combinations
149  // would not terminate in years.
150  final int MAX_LINKS = 62;
151 
152  // TODO: Shuffle
153 
154  // Just get combinations of objects and check its size.
155  List<List<MObject>> combinations =
156  CollectionUtil.<MObject>combinations(fObjectLists);
157 
158  int numLinks = combinations.size();
159 
160  if (numLinks > MAX_LINKS) {
161  collector.invalid(
162  "Can't execute `" +
163  fInstr +
164  "', because there are more than 2^" +
165  MAX_LINKS +
166  " combinations.");
167 
168  return;
169  }
170 
171  this.initProgress(numLinks);
172 
173  MSystemState state = conf.systemState();
174  MSystem system = state.system();
175  PrintWriter basicOutput = collector.basicPrintWriter();
176 
177  // for each link we need a statement to insert and delete
178  List<MStatement> insertStatements =
179  new ArrayList<MStatement>(combinations.size());
180  List<MStatement> deleteStatements =
181  new ArrayList<MStatement>(combinations.size());
182 
183  MAssociation association = fInstr.association();
184 
185  // construct the insert and delete statements. during the construction
186  // the current configuration (i.e. which links already exist in the
187  // current state) is determined and returned
188  long initialConfiguration =
189  constructInsertAndDeleteStatements(
190  combinations,
191  numLinks,
192  association,
193  state,
194  insertStatements,
195  deleteStatements);
196 
197  // in the original version of this algorithm, all links get removed so
198  // we do that as well to ensure previous test cases behave the same
199  try {
200  system.execute(
201  constructLinkChangeStatement(
202  initialConfiguration,
203  0,
204  insertStatements,
205  deleteStatements),
206  true,
207  false,
208  false);
209  system.getUniqueNameGenerator().popState();
210  } catch (MSystemException e) {
211  throw new GEvaluationException(e);
212  }
213 
214  // this is the "all links exist" configuration + 1
215  long tooLarge = (1 << insertStatements.size());
216 
217  // we start with the 0000... (all links off) combination
218  long oldConfiguration = 0;
219  // configuration after state change
220  // in the first iteration old- and newConfiguration are equal, so we
221  // can check the initial state
222  long newConfiguration = oldConfiguration;
223 
224  long numEvaluated = 0;
225  long numCut = 0;
226 
227  boolean continueEvaluation = true;
228

230  final boolean checkStructure = conf.getArguments().checkStructure();
231 
232  do {
233  // construct the statement that transforms the state from old to
234  // new configuration
235  MStatement statement = constructLinkChangeStatement(
236  oldConfiguration,
237  newConfiguration,
238  insertStatements,
239  deleteStatements);
240 
241  if (collector.doBasicPrinting())
242  basicOutput.println(statement.getShellCommand());
243 
244  try {
245  system.execute(statement, true, false, false);
246  } catch (MSystemException e) {
247  throw new GEvaluationException(e);
248  }
249 
250  if (useTryCuts && checkStructure) {
251  continueEvaluation = system.state().checkStructure(
252  fInstr.association(), NullPrintWriter.getInstance(),
253  false);
254  if (!continueEvaluation) {
255  ++numCut;
256  collector.addCut();
257  }
258  }
259 
260  if (continueEvaluation) {
261  ++numEvaluated;
262  fCaller.feedback(conf, null, collector);
263  if (collector.expectSubsequentReporting()) {
264  MSequenceStatement changeStatement =
265  constructLinkChangeStatement(
266  // 0, <- in the original the no-links configuration
267  // was used, but i'm pretty sure that's wrong
268  initialConfiguration, // transform initial state
269  newConfiguration, // to valid state
270  insertStatements,
271  deleteStatements);
272 
273  for (MStatement s : changeStatement.getStatements()) {
274  if (!s.isEmptyStatement()) {
275  collector.subsequentlyPrependStatement(s);
276  }
277  }
278  }
279  }
280 
281  // configurations for next iteration
282  oldConfiguration = newConfiguration;
283  ++newConfiguration;
284  this.outPutProgress(newConfiguration);
285  // Remove unique name state, because no undo statements are executed
286  system.getUniqueNameGenerator().popState();

288  // stop once all configurations have been built or stopping is allowed
289  } while ((newConfiguration < tooLarge) && !collector.canStop());
290 
291  this.endProgress();
292 
293  if (collector.doBasicPrinting()) {
294  basicOutput.print("Evaluated ");
295  basicOutput.print(numEvaluated);
296  basicOutput.print(" sub tress (cut ");
297  basicOutput.print(numCut);
298  basicOutput.println(")");
299  }
300 
301  // transform state to the initial state if we aren't in it anyways
302  if (oldConfiguration != initialConfiguration) {
303  MStatement statement =
304  constructLinkChangeStatement(
305  oldConfiguration,
306  initialConfiguration,
307  insertStatements,
308  deleteStatements);
309 
310  if (collector.doBasicPrinting())
311  basicOutput.println(statement.getShellCommand());
312 
313  try {
314  system.execute(statement, true, false, false);
315  system.getUniqueNameGenerator().popState();
316  } catch (MSystemException e) {
317  throw new GEvaluationException(e);
318  }
319  }

322  private long constructInsertAndDeleteStatements(
323  List<List<MObject>> combinations,
324  int numLinks,
325  MAssociation association,
326  MSystemState state,
327  List<MStatement> insertStatements,
328  List<MStatement> deleteStatements) throws GEvaluationException {
329 
330  // initial configuration
331  long initConfiguration = 0;
332 
333  // cache for MObject -> Expression transformations
334  Map<MObject, MRValue> participantCache =
335  new HashMap<MObject, MRValue>();
336 
337  // for each possible link, construct insert and delete statements
338  for (int i = 0; i < numLinks; ++i) {
339  List<MObject> objects = combinations.get(i);
340 
341  // construct participants
342  List<MRValue> participants =
343  new ArrayList<MRValue>(objects.size());
344 
345  for (MObject object : objects) {
346  MRValue participant = participantCache.get(object);
347  if (participant == null) {
348  participant = new MRValueExpression(object);
349  participantCache.put(object, participant);
350  }
351 
352  participants.add(participant);
353  }
354 
355  //FIXME: Support qualified associations in generator!
356  insertStatements.add(
357  new MLinkInsertionStatement(association, participants, Collections.<List<MRValue>>emptyList()));
358  deleteStatements.add(
359  new MLinkDeletionStatement(association, participants, Collections.<List<MRValue>>emptyList()));
360 
361  try {
362  // FIXME: Support qualifiers in generator
363  if (state.hasLink(association, objects, emptyQualifiers)) {
364  // "turn on" bit i
365  initConfiguration |= (1 << i);
366  }
367  } catch (MSystemException e) {
368  throw new GEvaluationException(e);
369  }
370  }
371 
372  return initConfiguration;
373  }

375  protected void tryLinksBinaryOne(
376  GConfiguration conf,
377  IGCollector collector) throws GEvaluationException {
378 
379  // Just get combinations of objects and check its size.
380  List<Pair<MObject>> links;
381  MinCombinationsIterator<MObject> linkSetIter;
382 
383  UniqueList unique;
384  if (!fInstr.association().associationEnds().get(0).isCollection()) {
385  unique = UniqueList.FIRST_IS_UNIQUE;
386  } else {
387  unique = UniqueList.SECOND_IS_UNIQUE;
388  }
389 
390  if (conf.getArguments().useRandomTry()) {
391  Collections.shuffle(fObjectLists.get(0), conf.random());
392  Collections.shuffle(fObjectLists.get(1), conf.random());
393  }
394 
395  linkSetIter = new MinCombinationsIterator<MObject>(fObjectLists.get(0), fObjectLists.get(1), unique);
396  MSystemState state = conf.systemState();
397  MSystem system = state.system();
398 
399  //TODO: in the original version of this algorithm, all links get removed so
400  // we do that as well to ensure previous test cases behave the same
401 
402  MSequenceStatement statements;
403  StatementEvaluationResult res;
404 
405  double ignoredStates;
406  double statesToDoOld = Math.pow(2, (fObjectLists.get(0).size() * fObjectLists.get(1).size()));
407  double statesToDo;
408 
409  if (unique == UniqueList.FIRST_IS_UNIQUE) {
410  statesToDo = (Math.pow(fObjectLists.get(0).size() + 1, fObjectLists.get(1).size()));
411  } else {
412  statesToDo = (Math.pow(fObjectLists.get(1).size() + 1, fObjectLists.get(0).size()));
413  }
414  ignoredStates = statesToDoOld - statesToDo;
415  this.initProgress((long)statesToDo);
416 
417  collector.addIgnoredStates((long)ignoredStates);
418  PrintWriter basicOutput = collector.basicPrintWriter();
419 
420  long tryNum = 0;
421 
422  do {
423  ++tryNum;
424  statements = new MSequenceStatement();
425  links = linkSetIter.next();
426 
427  for (Pair<MObject> elem : links) {
428  statements.appendStatement(new MLinkInsertionStatement(fInstr.association(), new MObject[]{elem.first, elem.second}, Collections.<List<Value>>emptyList()));
429  }
430 
431  if (collector.doBasicPrinting() && !statements.isEmpty())
432  basicOutput.println(statements.getShellCommand());
433 
434  try {
435  res = system.execute(statements, true, false, false);
436  } catch (MSystemException e) {
437  throw new GEvaluationException(e);
438  }
439 
440  fCaller.feedback(conf, null, collector);
441 
442  if (collector.expectSubsequentReporting()) {
443  for (MStatement s : statements.getStatements()) {
444  if (!s.isEmptyStatement()) {
445  collector.subsequentlyPrependStatement(s);
446  }
447  }
448  }
449 
450  this.outPutProgress(tryNum);
451 
452  try {
453  system.execute(res.getInverseStatement(), false, false, false);
454  } catch (MSystemException e) {
455  throw new GEvaluationException(e);
456  }
457 
458  system.getUniqueNameGenerator().popState();
459  system.getUniqueNameGenerator().popState();
460  // stop once all configurations have been built or stopping is allowed
461  } while (linkSetIter.hasNext() && !collector.canStop());
462  }

464  private MSequenceStatement constructLinkChangeStatement(
465  long oldCombination,
466  long newCombination,
467  List<MStatement> insertStatements,
468  List<MStatement> deleteStatements) {
469 
470 
471 
472  // becomes a sequence of insert/delete statements which transforms
473  // the state from the old configuration to the new configuration
474  MSequenceStatement result = new MSequenceStatement();
475 
476  if (oldCombination == newCombination) {
477  return result;
478  }
479 
480  // bit i is set if link i must be changed,
481  // i.e. inserted or deleted depending on whether it currently exists
482  long difference = (oldCombination ^ newCombination);
483  // start at lowest bit (the first link)
484  long mask = 1;
485  // last relevant bit
486  int maxBit = insertStatements.size();
487 
488  for (int i = 0; i < maxBit; ++i) {
489  // if we need to change this link
490  if ((difference & mask) > 0) {
491  // and it was "on" in the old state
492  if ((oldCombination & mask) > 0) {
493  // it needs to be deleted
494  result.prependStatement(deleteStatements.get(i));
495  } else {
496  // if it was "off" it needs to be inserted
497  result.prependStatement(insertStatements.get(i));
498  }
499  }
500  mask <<= 1;
501  }
502 
503  return result;
504  }
