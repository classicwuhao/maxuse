128  public static void drawArrow(final Graphics2D g, final int x1,
129  final int y1, final int x2, final int y2) {
130  final int ARR_SIZE = 5;
131 
132  // Get the current transform
133  AffineTransform saveAT = g.getTransform();
134 
135  double dx = x2 - x1, dy = y2 - y1;
136  double angle = Math.atan2(dy, dx);
137  int len = (int) Math.sqrt(dx * dx + dy * dy);
138  AffineTransform at = AffineTransform.getTranslateInstance(x1, y1);
139  at.concatenate(AffineTransform.getRotateInstance(angle));
140  // Perform transformation
141  g.transform(at);
