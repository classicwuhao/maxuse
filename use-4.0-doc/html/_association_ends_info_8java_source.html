215  public void initEntries() {
216  entries = new ArrayList<RowEntry>();
217 
218  // run through the subsetted associations
219  MAssociation assoc = getSelectedAssociation();
220 
221  if (assoc == null) return;
222 
223  // Buildup closures for the left and
224  Set<MAssociationEnd> leftEndSubsetsClosure = assoc.associationEnds().get(0).getSubsettingEndsClosure();
225  leftEndSubsetsClosure.add(assoc.associationEnds().get(0));
226  Set<MAssociationEnd> leftEndRedefinesClosure = assoc.associationEnds().get(0).getRedefiningEndsClosure();
227  leftEndRedefinesClosure.add(assoc.associationEnds().get(0));
228 
229  // Build up closures for the right end
230  Set<MAssociationEnd> rightEndSubsetsClosure = assoc.associationEnds().get(1).getSubsettingEndsClosure();
231  rightEndSubsetsClosure.add(assoc.associationEnds().get(1));
232  Set<MAssociationEnd> rightEndRedefinesClosure = assoc.associationEnds().get(1).getRedefiningEndsClosure();
233  rightEndRedefinesClosure.add(assoc.associationEnds().get(1));
234 
235  // Add info about the selected association
236  RowEntry selected = new RowEntry(assoc);
237  selected.left = assoc.associationEnds().get(0);
238  selected.leftRolename = "<html><font color='green'>" + selected.left.name() + "</font></html>";
239  selected.right = assoc.associationEnds().get(1);
240  selected.rightRolename = "<html><font color='green'>" + selected.right.name() + "</font></html>";
241  entries.add(selected);
242 
243  // We want to keep the order of the elements to get a subsetting / redefining hierarchy
244  LinkedHashSet<MAssociation> todo = new LinkedHashSet<MAssociation>(assoc.getSubsettedBy());
245  todo.addAll(assoc.getRedefinedBy());
246 
247  while (!todo.isEmpty()) {
248  MAssociation childAsso = todo.iterator().next();
249  todo.remove(childAsso);
250  RowEntry e = new RowEntry(childAsso);
251 
252  Set<MAssociationEnd> leftSubsetsSet = new HashSet<MAssociationEnd>();
253  Set<MAssociationEnd> leftRedefinedSet = new HashSet<MAssociationEnd>();
254  Set<MAssociationEnd> rightSubsetsSet = new HashSet<MAssociationEnd>();
255  Set<MAssociationEnd> rightRedefinedSet = new HashSet<MAssociationEnd>();
256 
257  // Its not guaranteed that both subset relations are given (one is guaranteed),
258  // therefore we check for both sides
259  if (leftEndSubsetsClosure.contains(childAsso.associationEnds().get(0)) ||
260  rightEndSubsetsClosure.contains(childAsso.associationEnds().get(1)) ||
261  leftEndRedefinesClosure.contains(childAsso.associationEnds().get(0)) ||
262  rightEndRedefinesClosure.contains(childAsso.associationEnds().get(1))) {
263  e.left = childAsso.associationEnds().get(0);
264  e.right = childAsso.associationEnds().get(1);
265  } else {
266  e.left = childAsso.associationEnds().get(1);
267  e.right = childAsso.associationEnds().get(0);
268  }
269 
270  if (e.left.getSubsettingEnds().size() > 0 || e.left.getRedefiningEnds().size() > 0) {
271  e.leftRolename = "<html><font color='blue'>" + e.left.name() + "</font></html>";
272  } else {
273  e.leftRolename = e.left.name();
274  }
275 
276  if (e.right.getSubsettingEnds().size() > 0 || e.right.getRedefiningEnds().size() > 0) {
277  e.rightRolename = "<html><font color='blue'>" + e.right.name() + "</font></html>";
278  } else {
279  e.rightRolename = e.right.name();
280  }
281 
282  for (MAssociationEnd end : e.left.getSubsettedEnds()) {
283  if (leftEndSubsetsClosure.contains(end)) {
284  if (e.leftSubsets.length() > 0) {
285  e.leftSubsets += ", ";
286  }
287 
288  e.leftSubsets += end.name();
289  leftSubsetsSet.add(end);
290  }
291  }
292 
293  for (MAssociationEnd end : e.right.getSubsettedEnds()) {
294  if (rightEndSubsetsClosure.contains(end)) {
295  if (e.rightSubsets.length() > 0) {
296  e.rightSubsets += ", ";
297  }
298 
299  e.rightSubsets += end.name();
300  rightSubsetsSet.add(end);
301  }
302  }
303 
304  for (MAssociationEnd end : e.left.getRedefinedEnds()) {
305  if (leftEndRedefinesClosure.contains(end)) {
306  if (e.leftRedefines.length() > 0) {
307  e.leftRedefines += ", ";
308  }
309 
310  e.leftRedefines += end.name();
311  leftRedefinedSet.add(end);
312  }
313  }
314 
315  for (MAssociationEnd end : e.right.getRedefinedEnds()) {
316  if (rightEndRedefinesClosure.contains(end)) {
317  if (e.rightRedefines.length() > 0) {
318  e.rightRedefines += ", ";
319  }
320 
321  e.rightRedefines += end.name();
322  rightRedefinedSet.add(end);
323  }
324  }
325 
326  if (leftSubsetsSet.size() != rightSubsetsSet.size()) {
327  for (MAssociationEnd end : leftSubsetsSet) {
328  MAssociationEnd otherEnd = end.getAllOtherAssociationEnds().get(0);
329  if (!rightSubsetsSet.contains(otherEnd)) {
330  if (!e.rightSubsets.isEmpty())
331  e.rightSubsets += ", ";
332 
333  e.rightSubsets += "<font color='red'>" + otherEnd.name() + "</font>";
334  }
335  }
336 
337  for (MAssociationEnd end : rightSubsetsSet) {
338  MAssociationEnd otherEnd = end.getAllOtherAssociationEnds().get(0);
339  if (!leftSubsetsSet.contains(otherEnd)) {
340  if (!e.leftSubsets.isEmpty())
341  e.leftSubsets += ", ";
342 
343  e.leftSubsets += "<font color='red'>" + otherEnd.name() + "</font>";
344  }
345  }
346 
347  e.leftSubsets = "<html>" + e.leftSubsets + "</html>";
348  e.rightSubsets = "<html>" + e.rightSubsets + "</html>";
349  }
350 
351  if (leftRedefinedSet.size() != rightRedefinedSet.size()) {
352  for (MAssociationEnd end : leftRedefinedSet) {
353  MAssociationEnd otherEnd = end.getAllOtherAssociationEnds().get(0);
354  if (!rightRedefinedSet.contains(otherEnd)) {
355  if (!e.rightRedefines.isEmpty())
356  e.rightRedefines += ", ";
357 
358  e.rightRedefines += "<font color='red'>" + otherEnd.name() + "</font>";
359  }
360  }
361 
362  for (MAssociationEnd end : rightRedefinedSet) {
363  MAssociationEnd otherEnd = end.getAllOtherAssociationEnds().get(0);
364  if (!leftRedefinedSet.contains(otherEnd)) {
365  if (!e.leftRedefines.isEmpty())
366  e.leftRedefines += ", ";
367 
368  e.leftRedefines += "<font color='red'>" + otherEnd.name() + "</font>";
369  }
370  }
371 
372  e.leftRedefines = "<html>" + e.leftRedefines + "</html>";
373  e.rightRedefines = "<html>" + e.rightRedefines + "</html>";
374  }
375 
376  entries.add(e);
377  todo.addAll(childAsso.getSubsettedBy());
378  }
