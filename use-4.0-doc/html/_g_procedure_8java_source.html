119  public void calculateBarriers(IGCollector collector, MModel model) {
120 
121  // Calculate coverage of all invariants
122  Map<MClassInvariant, BasicCoverageData> invCoverage = new HashMap<MClassInvariant, BasicCoverageData>();
123  BasicExpressionCoverageCalulator invCalc = new BasicExpressionCoverageCalulator(true);
124 
125  for (MClassInvariant inv : model.classInvariants()) {
126  if (inv.isActive()) {
127  invCoverage.put(inv, invCalc.calcualteCoverage(inv.flaggedExpression()));
128  }
129  }
130 
131  BasicInstructionCoverageCalulator instrCalc = new BasicInstructionCoverageCalulator();
132  List<GInstruction> instrList = fInstructionList.instructions();
133  Set<MClassInvariant> blockedInvs = new HashSet<>();
134 
135  for (int index = 0; index < instrList.size(); ++index) {
136  // Add user defined barrier to statistics
137  if (instrList.get(index).getClass().equals(GInstrBarrier.class) ) {
138  collector.addBarrier((GInstrBarrier)instrList.get(index));
139  }
140 
141  BasicCoverageData instrCoverage = instrCalc
142  .calcualteCoverage(instrList.subList(index,
143  instrList.size()));
144  blockedInvs.clear();
145 
146  for (Map.Entry<MClassInvariant, BasicCoverageData> invData : invCoverage.entrySet()) {
147  MClassInvariant inv = invData.getKey();
148  if (instrCoverage.disjoint(invData.getValue())) {
149  GInstrCalculatedBarrier bInstr = new GInstrCalculatedBarrier(inv);
150  blockedInvs.add(inv);
151  collector.addBarrier(bInstr);
152  instrList.add(index, bInstr);
153  inv.setCheckedByBarrier(true);
154  ++index;
155 
156  if (collector.doBasicPrinting()) {
157  collector.basicPrintWriter().println(
158  "Added Barrier for invariant "
159  + StringUtil.inQuotes(inv)
160  + " before statement "
161  + StringUtil.inQuotes(fInstructionList.instructions().get(index)));
162  }
163  }
164  }
165 
166  for (MClassInvariant inv : blockedInvs) {
167  invCoverage.remove(inv);
168  }
169  }
170  }
171 }
