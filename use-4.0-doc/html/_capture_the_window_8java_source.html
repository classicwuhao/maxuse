59  chooser.addChoosableFileFilter(new JpgFilter());

62  chooser.setFileFilter(filters[0]);
63  // no gif filter supported by java because of license reasons
64  int returnVal = chooser.showSaveDialog(fComponent);
65  if (returnVal == JFileChooser.APPROVE_OPTION) {
66  File selectedFile = chooser.getSelectedFile();
67  if (!selectedFile.exists())
68  selectedFile.createNewFile();
69  FileFilter ff = chooser.getFileFilter();
70  // if the choosen file has an valid extention
71  String format = getExtention(selectedFile);
72  if (format.equals("jpg") || format.equals("jpeg")
73  || format.equals("png") || format.equals("bmp")) {
74  if (selectedFile.canWrite())
75  ImageIO.write(img, format, selectedFile
76  .getAbsoluteFile());
77  else
78  new ErrorFrame("IO Error on File "
79  + selectedFile.getAbsoluteFile() + " occured");
80  } else if (getExtention(selectedFile).equals("eps")) {
81  if (selectedFile.canWrite())
82  ImageIO.write(img, "eps", selectedFile
83  .getAbsoluteFile());
84  else
85  new ErrorFrame("IO Error on File "
86  + selectedFile.getAbsoluteFile() + " occured");
87 
88  }// if the file has no valid extention
89  else if (ff.getDescription().equals("All Files")) {
90  // if no image format is choosen save img to the default PNG
91  // format
92  if (selectedFile.canWrite())
93  ImageIO.write(img, "png", new File(selectedFile
94  .getAbsolutePath()
95  + ".png"));
96  else
97  new ErrorFrame("IO Error on File "
98  + selectedFile.getAbsoluteFile() + " occured");
99 
100  } else {
101  if (selectedFile.canWrite()) {
102  format = ff.getDescription().substring(2);
103  ImageIO.write(img, format, new File(selectedFile
104  .getAbsoluteFile()
105  + "." + format));
106  } else
107  new ErrorFrame("IO Error on File "
108  + selectedFile.getAbsoluteFile() + " occurred");
109  }
110 
111  }

212  public void adjustTopWidth(JLabel label, String text, String htmlText) {
213  // get the Display Size
214  Toolkit tk = Toolkit.getDefaultToolkit();
215  Dimension displaySize = tk.getScreenSize();
216  int maxWidth = (int) displaySize.getWidth();
217 
218  // Dimension windowSize = getSize();
219  int windowSize = getWidth();
220 
221  if (windowSize < maxWidth)
222  maxWidth = windowSize;
223  // calculate the Dimension for the title
224  FontMetrics fm = label.getFontMetrics(label.getFont());
225  int topWidth = 0;
226  int topHeight = fm.getHeight();
227  Pattern p = Pattern.compile("\n");
228  String s[] = p.split(text);
229  for (int i = 0; i < s.length; i++) {
230  // if the current string is wider
231  if (topWidth < fm.stringWidth(s[i]))
232  if (fm.stringWidth(s[i]) < maxWidth)
233  // string is between width so far and maxWidth
234  topWidth = fm.stringWidth(s[i]);
235  else
236  topWidth = maxWidth;
237  // increment the height with 1 line or more if needed
238  topHeight += (fm.getHeight() * (1 + (fm.stringWidth(s[i]) / maxWidth)));
239  }
240  // set the Size of the South Frame
241  label.setPreferredSize(new Dimension(topWidth, topHeight));
242  // make changes visible
243  label.setText(htmlText);
244  label.setVisible(false);
245  label.setVisible(true);
246  }
247  }
