80  @Override
81  public Type getLeastCommonSupertype(Type type) {
82  if (type == null )
83  return this;
84 
85  // easy case: equal type
86  if (this.equals(type))
87  return type;
88 
89  // one of the types (this or type) is oclVoid
90  if (this.isTypeOfVoidType())
91  return type;
92 
93  if (type.isTypeOfVoidType())
94  return this;
95 
96  // Collection types have their own implementation of
97  // getLeastCommonSupertype and have no common type to
98  // simple types
99  if (type.isKindOfCollection(VoidHandling.EXCLUDE_VOID)) {
100  return null;
101  }
102 
103  // determine common supertypes = intersection of all
104  // supertypes of all elements
105  Set<Type> cs = new HashSet<Type>();
106  cs.addAll(this.allSupertypes());
107  cs.retainAll(type.allSupertypes());
108 
109  // return immediately if intersection is empty
110  if (cs.isEmpty() )
111  return null;
112 
113  // determine the least common supertype
114  // if there is only one common supertype return it
115  if (cs.size() == 1 )
116  return cs.iterator().next();
117 
118  // For performance reasons: if oclAny and another type then the other type
119  // is the least common supertype
120  if (cs.size() == 2 && cs.contains(TypeFactory.mkOclAny())) {
121  cs.remove(TypeFactory.mkOclAny());
122  // Returns the only element
123  return cs.iterator().next();
124  }
125 
126  // search for a type that is less than or equal to all other types
127  Type cType = null;
128 
129  Iterator<Type> it1 = cs.iterator();
130  outerLoop:
131  while (it1.hasNext() ) {
132  Type t1 = it1.next();
133  Iterator<Type> it2 = cs.iterator();
134 
135  while (it2.hasNext() ) {
136  Type t2 = it2.next();
137  if (! t1.conformsTo(t2) )
138  continue outerLoop;
139  }
140 
141  cType = t1;
142  break;
143  }
144 
145  return cType;
146  }
