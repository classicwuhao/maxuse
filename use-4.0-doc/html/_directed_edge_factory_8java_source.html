764  public static void drawArrow(final Graphics2D g, final int x1, final int y1, final int x2, final int y2, ArrowStyle style) {
765  int ARR_LENGTH = 8;
766  int ARR_WIDTH = 5;
767 
768  // Get the current transform
769  AffineTransform saveAT = g.getTransform();
770 
771  double dx = x2 - x1, dy = y2 - y1;
772  double angle = Math.atan2(dy, dx);
773  int len = (int)Math.round(Math.sqrt(dx*dx + dy*dy));
774  AffineTransform at = AffineTransform.getTranslateInstance(x1, y1);
775  at.concatenate(AffineTransform.getRotateInstance(angle));
776  // Perform transformation
777  g.transform(at);
778 
779  // Draw horizontal line starting in (0, 0) to start of arrowhead
780  g.drawLine(0, 0, len, 0);
781  if (style == ArrowStyle.FILLED) {
782  g.fillPolygon(new int[] {len, len-ARR_LENGTH, len-ARR_LENGTH, len},
783  new int[] {0, -ARR_WIDTH, ARR_WIDTH, 0 }, 4);
784  } else {
785  g.drawLine(len, 0, len - ARR_LENGTH, ARR_WIDTH);
786  g.drawLine(len, 0, len - ARR_LENGTH, -ARR_WIDTH);
787  }
788 
789  // Restore original transform
790  g.setTransform(saveAT);
791  }
