200  private void genRedefinesConstraints(ASTAssociationEnd aEnd)
201  throws SemanticException {
202  if (aEnd.getRedefinesRolenames().size() > 0) {
203  // Get the MAssociationEnd from the name
204  MClass cls = model.getClass(aEnd.getClassName());
205  MAssociationEnd redefiningEnd = association.getAssociationEnd(cls, aEnd.getRolename(ctx));
206 
207  // Find parent association end to redefine
208  for (Token redefinesRolenameToken : aEnd.getRedefinesRolenames()) {
209  boolean success = false;
210  StringBuilder error = new StringBuilder();
211  StringBuilder validationError;
212 
213  String redefinesRolename = redefinesRolenameToken.getText();
214  List<MAssociationEnd> possibleRedefinedEnds = cls.getAssociationEnd(redefinesRolename);
215 
216  if (possibleRedefinedEnds.size() == 0) {
217  throw new SemanticException(redefinesRolenameToken, "No association end with name '" + redefinesRolename + "' to redefine.");
218  }
219 
220  //Find a compatible association
221  //Duplicate role names are possible:
222  for (MAssociationEnd possibleRedefinedEnd : possibleRedefinedEnds) {
223  validationError = new StringBuilder();
224  if (possibleRedefinedEnd.equals(redefiningEnd)) continue;
225 
226  MAssociation parentAssociation = possibleRedefinedEnd.association();
227 
228  // Possible generation error might lead to null
229  if (parentAssociation == null) return;
230 
231  if (!validateInheritance(association, parentAssociation, ValidationContext.REDEFINES, validationError)) {
232  if (error.length() > 0) error.append(StringUtil.NEWLINE);
233  //TODO each iteration of the for loop yields the same information in the output
234  // (leading possibly multiple to identical lines of error messages)
235  error.append("Constraint {redefines ").append(redefinesRolename)
236  .append("} on association end '").append(redefiningEnd.nameAsRolename())
237  .append(":").append(association.name()).append("' is invalid.")
238  .append(StringUtil.NEWLINE)
239  .append(validationError);
240  continue;
241  }
242 
243  success = true;
244 
245  // Set redefined information on this AssociationEnd
246  // More readable
247  MAssociationEnd redefinedEnd = possibleRedefinedEnd;
248  redefiningEnd.addRedefinedEnd(redefinedEnd);
249  redefinedEnd.addRedefiningEnd(redefiningEnd);
250 
251  parentAssociation.addRedefinedBy(association);
252  association.addRedefines(parentAssociation);
253  }
254 
255  if (!success) {
256  throw new SemanticException(redefinesRolenameToken, error.toString());
257  }
258  }
259  }

328  private boolean validateInheritance(MAssociation association, MAssociation parentAssociation, ValidationContext validationContext, StringBuilder errorBuffer) {
329 
330  if (association.reachableEnds().size() != parentAssociation.reachableEnds().size()) {
331  errorBuffer.append("Child association must have the same number of association ends as its parent.");
332  return false;
333  }
334 
335  // UML 2.4.1: (self.isComposite implies prop.isComposite)
336  // self is the redefined property. prop is the redefinee.
337  if (validationContext == ValidationContext.REDEFINES) {
338  if (parentAssociation.aggregationKind() == MAggregationKind.COMPOSITION &&
339  association.aggregationKind() != MAggregationKind.COMPOSITION) {
340  errorBuffer.append("Redefinig association must be a composition, if the redefined association is one.");
341  return false;
342  }
343  }
344 
345  // We check the subsetting (and redefinition) context by index.
346  // This means, each child association must be defined in the same order
347  // as the parent.
348  MNavigableElement parentEnd;
349  MNavigableElement ourEnd;
350 
351  for (int index = 0; index < parentAssociation.associationEnds().size(); ++index) {
352  ourEnd = association.reachableEnds().get(index);
353  parentEnd = parentAssociation.reachableEnds().get(index);
354 
355  if (!ourEnd.cls().isSubClassOf(parentEnd.cls())) {
356  errorBuffer.append("The end type of the association end ");
357  errorBuffer.append(StringUtil.inQuotes(ourEnd.toString()));
358  errorBuffer.append(" has no valid inheritance relation to the association end ");
359  errorBuffer.append(StringUtil.inQuotes(parentEnd.toString()));
360  errorBuffer.append(StringUtil.NEWLINE);
361  errorBuffer.append("Note, that the ends of a child association must be defined in the same order as the ends from the parent.");
362  return false;
363  }
364 
365  if (validationContext == ValidationContext.SUBSETS) {
366  // UML 2.4.1, p. 127:
367  // A Property cannot be subset by a Property with the same name
368  // if (self.subsettedProperty->notEmpty()) then
369  // self.subsettedProperty->forAll(sp | sp.name <> self.name)
370  if (ourEnd.nameAsRolename().equals(parentEnd.nameAsRolename())) {
371  errorBuffer.append("An association end cannot be subset by an association end with the same name.");
372  return false;
373  }
374  }
375 
376  // Check multiplicity
377 
378  // UML 2.4.1, p. 127:
379  // A subsetting property may strengthen the type of the subsetted property, and its upper bound may be less.
380  // subsettedProperty->forAll(sp | type.conformsTo(sp.type) and
381  // ((upperBound()->notEmpty() and sp.upperBound()->notEmpty()) implies
382  // upperBound()<=sp.upperBound() ))
383  // The constraint to validate redefinition also checks the lower bound.
384  //
385  if (ourEnd instanceof MAssociationEnd) {
386  // The multiplicity of an association class end is defined internally
387  MAssociationEnd ourAssocEnd = (MAssociationEnd)ourEnd;
388  MAssociationEnd parentAssocEnd = (MAssociationEnd)parentEnd;
389 
390  if (!parentAssocEnd.multiplicity().includesMultiplicity(ourAssocEnd.multiplicity(), validationContext == ValidationContext.REDEFINES)) {
391  errorBuffer.append("A " + (validationContext == ValidationContext.REDEFINES ? " redefining " : "subsetting ") +
392  "association end can only reduce the multiplicity of the parent association end.");
393  return false;
394  }
395  }
396  }
397 
398  return true;
399  }
