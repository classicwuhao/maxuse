265  private List<TextLayout> layoutText(Graphics2D g, Point2D.Double pen, double width, double availableHeight, String toLayout) {
266  List<TextLayout> stateInvariantLayouts = new LinkedList<TextLayout>();
267 
268  if (toLayout == null || toLayout.isEmpty())
269  return stateInvariantLayouts;
270 
271  FontRenderContext frc = g.getFontRenderContext();
272 
273  AttributedString styledText = new AttributedString(toLayout);
274  styledText.addAttribute(TextAttribute.FONT, getInvFont());
275  styledText.addAttribute(TextAttribute.FOREGROUND, Color.BLACK);
276 
277  AttributedCharacterIterator characterIterator = styledText.getIterator();
278 
279  LineBreakMeasurer measurer = new LineBreakMeasurer(characterIterator, frc);
280 
281  float wrappingWidth = (float)width;
282  int nextOffset = 0;
283  double lineHeight = Util.getLineHeight(g.getFontMetrics(getInvFont()));
284  TextLayout layout;
285 
286  while (measurer.getPosition() < toLayout.length()) {
287  nextOffset = measurer.nextOffset(wrappingWidth);
288 
289 
290  // Enough space for more text?
291  if (nextOffset < toLayout.length() && pen.y + lineHeight > availableHeight) {
292  int position = measurer.getPosition();
293 
294  AttributedString styledSuffix = new AttributedString(toLayout.substring(0, nextOffset - 5) + " ...]");
295  styledSuffix.addAttribute(TextAttribute.FONT, getInvFont());
296  styledSuffix.addAttribute(TextAttribute.FOREGROUND, Color.BLACK);
297 
298  // Remove last six chars and add ...]
299  for (int i = 6; i > 0; --i) {
300  measurer.insertChar(styledSuffix.getIterator(), nextOffset - i);
301  }
302 
303 
304  measurer.setPosition(position);
305  layout = measurer.nextLayout(wrappingWidth);
306  stateInvariantLayouts.add(layout);
307  break;
308  } else {
309  pen.y += lineHeight;
310  layout = measurer.nextLayout(wrappingWidth);
311  stateInvariantLayouts.add(layout);
312  }
313  }
314 
315  return stateInvariantLayouts;
316  }

329  public Point2D getIntersectionCoordinate(Point2D source, Point2D target) {
330  // The original code was published here:
331  // http://gravisto.fim.uni-passau.de/doc/guide/plugins/kinds/shape.html
332  Rectangle2D rect = getBounds();
333 
334  double dist = Math.max(rect.getWidth(), rect.getHeight()) * 2;
335  if (FloatUtil.equals(dist, 0.0f))
336  return source;
337 
338  Line2D line = new Line2D.Double(source, target);
339  Util.enlargeLine(line, dist);
340 
341  double x = rect.getX();
342  double y = rect.getY();
343  double w = rect.getWidth();
344  double h = rect.getHeight();
345  double radius = RECTARG / 2d;
346 
347  Point2D lowerLeft = new Point2D.Double(x, (y + h) - radius);
348  Point2D upperLeft = new Point2D.Double(x, y + radius);
349  Point2D leftUpper = new Point2D.Double(x + radius, y);
350  Point2D rightUpper = new Point2D.Double((x + w) - radius, y);
351  Point2D upperRight = new Point2D.Double(x + w, y + radius);
352  Point2D lowerRight = new Point2D.Double(x + w, (y + h) - radius);
353  Point2D rightLower = new Point2D.Double((x + w) - radius, y + h);
354  Point2D leftLower = new Point2D.Double(x + radius, y + h);
355 
356  // turn the round rectangle into 4 lines
357  Line2D left = new Line2D.Double(lowerLeft, upperLeft);
358  Line2D upper = new Line2D.Double(leftUpper, rightUpper);
359  Line2D right = new Line2D.Double(upperRight, lowerRight);
360  Line2D lower = new Line2D.Double(rightLower, leftLower);
361 
362  // testing with which line intersects with line
363  // and then computing the intersection point
364  if (left.intersectsLine(line))
365  {
366  return Util.intersectionPoint(left, line);
367  }
368  else if (upper.intersectsLine(line))
369  {
370  return Util.intersectionPoint(upper, line);
371  }
372  else if (right.intersectsLine(line))
373  {
374  return Util.intersectionPoint(right, line);
375  }
376  else if (lower.intersectsLine(line))
377  {
378  return Util.intersectionPoint(lower, line);
379  }
380 
381  // intersection with upper left circle
382  Ellipse2D upperLeftCircle2D = new Ellipse2D.Double(x, y, RECTARG, RECTARG);
383  Point2D intUpperLeftCircle2D = Util.intersectWithCircle(upperLeftCircle2D, line, Direction.NORTH_WEST);
384 
385  if (intUpperLeftCircle2D != null)
386  {
387  return intUpperLeftCircle2D;
388  }
389 
390  // intersection with upper right circle
391  Ellipse2D upperRightCircle2D = new Ellipse2D.Double((x + w) -
392  RECTARG, y, RECTARG, RECTARG);
393  Point2D intUpperRightCircle2D = Util.intersectWithCircle(upperRightCircle2D,
394  line, Direction.NORTH_EAST);
395 
396  if (intUpperRightCircle2D != null)
397  {
398  return intUpperRightCircle2D;
399  }
400 
401  // intersection with lower left circle
402  Ellipse2D lowerLeftCircle2D = new Ellipse2D.Double(x,
403  (y + h) - RECTARG, RECTARG, RECTARG);
404  Point2D intLowerLeftCircle2D = Util.intersectWithCircle(lowerLeftCircle2D,
405  line, Direction.SOUTH_WEST);
406 
407  if (intLowerLeftCircle2D != null)
408  {
409  return intLowerLeftCircle2D;
410  }
411 
412  // intersection with lower right circle
413  Ellipse2D lowerRightCircle2D = new Ellipse2D.Double((x + w) -
414  RECTARG, (y + h) - RECTARG, RECTARG,
415  RECTARG);
416  Point2D intLowerRightCircle2D = Util.intersectWithCircle(lowerRightCircle2D,
417  line, Direction.SOUTH_EAST);
