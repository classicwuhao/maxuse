260  private double computeMessageArrowAngle(double x, double y) {
261  List<WayPoint> wayPoints = fEdge.getWayPoints();
262  WayPoint wayPointOne = wayPoints.get(0);
263  WayPoint wayPointTwo = wayPointOne.getNextWayPoint();
264 
265  // Search for begin point and end point of a segment of communication
266  // edge.
267  // Distance from drawing point to this segment must be minimum.
268  if (wayPoints.size() > 2) {
269  double nearestDistance = Double.MAX_VALUE;
270 
271  for (int i = 0; i < wayPoints.size() - 1; i++) {
272  WayPoint first = wayPoints.get(i); // Point A
273  WayPoint second = first.getNextWayPoint(); // Point B
274  // Point C (x, y) is the argument of this method
275  // Check if angle CAB > 90 degree
276  double dotCB = (x - first.getX()) * (second.getX() - first.getX()) + (y - first.getY()) * (second.getY() - first.getY());
277  if (dotCB < 0) {
278  double distanceCA = Math.sqrt(Math.pow(x - first.getX(), 2) + Math.pow(y - first.getY(), 2));
279  if (distanceCA < nearestDistance) {
280  nearestDistance = distanceCA;
281  wayPointOne = first;
282  wayPointTwo = second;
283  continue;
284  }
285  }
286  // Check if angle CBA > 90 degree
287  double dotCA = (x - second.getX()) * (first.getX() - second.getX()) + (y - second.getY()) * (first.getY() - second.getY());
288  if (dotCA < 0) {
289  double distanceCB = Math.sqrt(Math.pow(x - second.getX(), 2) + Math.pow(y - second.getY(), 2));
290  if (distanceCB < nearestDistance) {
291  nearestDistance = distanceCB;
292  wayPointOne = first;
293  wayPointTwo = second;
294  continue;
295  }
296  }
297  // Otherwise calculate distance from C to segment AB
298  if (dotCB > 0 && dotCA > 0) {
299  double numerator = Math.abs((first.getX() - second.getX()) * (second.getY() - y) - (second.getX() - x) * (first.getY() - second.getY()));
300  double denominator = Math.sqrt(Math.pow((first.getX() - second.getX()), 2) + Math.pow((first.getY() - second.getY()), 2));
301  double distance = numerator / denominator;
302  if (distance < nearestDistance) {
303  nearestDistance = distance;
304  wayPointOne = first;
305  wayPointTwo = second;
306  }
307  }
308  }
309  }
310  // Compute the angle between target node and source node
311  // The arrow of a message is always parallel to message edge
312  double dx = wayPointTwo.getX() - wayPointOne.getX(), dy = wayPointTwo.getY() - wayPointOne.getY();
313  double angle = Math.atan2(dy, dx);
314  return angle;
315  }
