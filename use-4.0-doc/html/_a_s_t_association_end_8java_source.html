238  public MAssociationEnd gen(Context ctx, int kind) throws SemanticException {
239  // lookup class at association end in current model
240  MClass cls = ctx.model().getClass(fName.getText());
241 
242  if (cls == null )
243  // this also renders the rest of the association useless
244  throw new SemanticException(fName, "Class `" + fName.getText() +
245  "' does not exist in this model.");
246 
247  MMultiplicity mult = fMultiplicity.gen(ctx);
248  if (fOrdered && ! mult.isCollection() ) {
249  ctx.reportWarning(fName, "Specifying `ordered' for " +
250  "an association end targeting single objects has no effect.");
251  fOrdered = false;
252  }
253 
254  List<VarDecl> generatedQualifiers;
255  if (qualifiers.size() == 0) {
256  generatedQualifiers = Collections.emptyList();
257  } else {
258  generatedQualifiers = new ArrayList<VarDecl>(qualifiers.size());
259 
260  for (ASTVariableDeclaration var : qualifiers ) {
261  generatedQualifiers.add(var.gen(ctx));
262  }
263  }
264 
265  mAend = ctx.modelFactory().createAssociationEnd(cls, getRolename(ctx),
266  mult, kind, fOrdered, generatedQualifiers);
267 
268  mAend.setUnion(this.isUnion);
269  mAend.setDerived(this.derivedExpression != null);
270 
271  this.genAnnotations(mAend);
272 
273  return mAend;
274  }

281  public void genDerived(Context ctx) throws SemanticException {
282  if (!this.isDerived()) return;
283 
284  VarDeclList parameter = new VarDeclList(false);
285  boolean exprContextChanged = false;
286  Symtable vars = ctx.varTable();
287  vars.enterScope();
288 
289 
290  try {
291  if (this.deriveParameter == null || this.deriveParameter.isEmpty()) {
292  // Short notation using self
293  if (this.mAend.association().associationEnds().size() == 2) {
294  MClass ot = mAend.getAllOtherAssociationEnds().get(0).cls();
295  parameter.add(new VarDecl("self", ot));
296  ctx.exprContext().push("self", ot);
297  exprContextChanged = true;
298  } else {
299  throw new SemanticException(fName, "Derived n-ary associations must define parameter for the derive expression.");
300  }
301  } else {
302  if (this.deriveParameter.getVarNames().size() != mAend.association().associationEnds().size() - 1) {
303  throw new SemanticException(fName, "Invalid number of parameter for derive expression!");
304  }
305 
306  int parIndex = 0;
307  for (int index = 0; index < mAend.association().associationEnds().size(); ++index) {
308  if (mAend.association().associationEnds().get(index) != mAend ) {
309  // Use association end type. Can be more generic in declaration
310  Type varType = mAend.association().associationEnds().get(index).cls();
311 
312  ASTType astType = deriveParameter.getVarTypes().get(parIndex);
313  if (astType != null) {
314  Type declaredType = astType.gen(ctx);
315  if (!varType.conformsTo(declaredType)) {
316  throw new SemanticException(astType.getStartToken(), "The derive parameter must be of type " + StringUtil.inQuotes(varType.toString()) + " or one of its supertypes.");
317  }
318  varType = declaredType;
319  }
320 
321  parameter.add(new VarDecl(deriveParameter.getVarTokens().get(parIndex), varType));
322  parIndex++;
323  }
324  }
325  }
326 
327  parameter.addVariablesToSymtable(vars);
328 
329  Expression exp = derivedExpression.gen(ctx);
330 
331  // We can ignore redefinition here
332  if (!exp.type().conformsTo(mAend.getType())) {
333  throw new SemanticException(derivedExpression.getStartToken(),
334  "The type " +
335  StringUtil.inQuotes(exp.type().toString()) + " of the derive expression at association end " +
336  StringUtil.inQuotes(mAend.association().toString() + "::" + getRolename(ctx)) + " does not conform to the end type " + StringUtil.inQuotes(mAend.getType()) + ".");
337  }
338 
339  mAend.setDeriveExpression(parameter, exp);
340  } finally {
341  ctx.varTable().exitScope();
342  if (exprContextChanged)
343  ctx.exprContext().pop();
344  }
345  }
