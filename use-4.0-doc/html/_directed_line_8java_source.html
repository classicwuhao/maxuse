296  private I_DirectedLine rotateInRightAngleMultiples(final I_DirectedLine auxiliaryLine,
297  final double remainingAngle) {
298  final double height = auxiliaryLine.getTargetY() - auxiliaryLine.getSourceY();
299  final double width = auxiliaryLine.getTargetX() - auxiliaryLine.getSourceX();
300  if (FloatUtil.equals(remainingAngle, 270.0d)) {
301  return doCreateDirectedLine(sourceX, sourceY, sourceX + height, sourceY - width);
302  }
303  if (FloatUtil.equals(remainingAngle, 180.0d)) {
304  return doCreateDirectedLine(sourceX, sourceY, sourceX - width, sourceY - height);
305  }
306  if (FloatUtil.equals(remainingAngle, 90.0d)) {
307  return doCreateDirectedLine(sourceX, sourceY, sourceX - height, sourceY + width);
308  }
309  return auxiliaryLine;
