144  private void evalClosureAux(Set<Value> resValues, Value elem, EvalContext ctx) {
145 
146  Collection<Value> rangeVal;
147  CollectionType resultType = (CollectionType)type();
148 
149  if (elem.isCollection() && !elem.type().conformsTo(resultType.elemType())) {
150  rangeVal = ((CollectionValue)elem).collection();
151  } else {
152  rangeVal = new ArrayList<Value>(1);
153  rangeVal.add(elem);
154  }
155 
156  // loop over range elements
157  for (Value elemVal : rangeVal) {
158 
159  // bind element variable to range element, if variable was
160  // declared
161  if (!fElemVarDecls.isEmpty())
162  ctx.pushVarBinding(fElemVarDecls.varDecl(0).name(), elemVal);
163 
164  // evaluate closure expression
165  Value val = fQueryExp.eval(ctx);
166 
167  // add element or elements to result
168  if (val.isCollection() && !val.type().conformsTo(resultType.elemType()) ) {
169  CollectionValue colVal = (CollectionValue)val;
170  for (Value elem2 : colVal.collection()) {
171  if (!resValues.contains(elem2)) {
172  resValues.add(elem2);
173  evalClosureAux(resValues, elem2, ctx);
174  }
175  }
176  } else {
177  if (!resValues.contains(val)) {
178  resValues.add(val);
179  evalClosureAux(resValues, val, ctx);
180  }
181  }
182 
183  if (!fElemVarDecls.isEmpty())
184  ctx.popVarBinding();
185  }
