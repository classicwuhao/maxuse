74  public VarDeclList gen(Context ctx, Expression range) throws SemanticException
75  {
76  // variable type may be omitted in query expressions
77  Type sourceElementType;
78 
79  if (range.type().isTypeOfVoidType()) {
80  sourceElementType = TypeFactory.mkVoidType();
81  } else {
82  sourceElementType = ((CollectionType) range.type()).elemType();
83  }
84 
85  // build list of VarDecls
86  VarDeclList varDeclList = new VarDeclList(false);
87  Type thisType;
88 
89  for (int index = 0; index < fIdList.size(); ++index) {
90  Token id = fIdList.get(index);
91  ASTType type = fTypeList.get(index);
92 
93  if (varDeclList.containsName(id.getText())) {
94  throw new SemanticException(id, "double declared variable");
95  }
96 
97  if (type != null) {
98  thisType = type.gen(ctx);
99  if (!sourceElementType.conformsTo(thisType))
100  throw new SemanticException(id, "Invalid type "
101  + StringUtil.inQuotes(thisType)
102  + " for source element type "
103  + StringUtil.inQuotes(sourceElementType)
104  + " specified.");
105  } else {
106  thisType = sourceElementType;
107  }
108 
109  VarDecl decl = new VarDecl(id, thisType);
110  varDeclList.add(decl);
111  }
112  return varDeclList;
113  }
