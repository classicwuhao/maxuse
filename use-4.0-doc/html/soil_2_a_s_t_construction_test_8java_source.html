103  // empty
104  stats.add("");
105  // variable assignment (expression)
106  stats.add("v := 42");
107  // variable assignment (new object)
108  stats.add("v := new C");
109  // variable assignment (new object with mandatory name)
110  stats.add("v := new C('o1')");
111  // variable assignment (new link object)
112  stats.add("v := new AC between (o1, o2)");
113  // variable assignment (new link object with mandatory name)

115  // attribute assignment
116  stats.add("o.att := 42");
117  // link object creation
118  stats.add("new AC between (o1, o2)");
119  // link object creation with mandatory name
120  stats.add("new AC('lo1') between (o1, o2)");
121  // object creation
122  stats.add("new C");
123  // object creation with mandatory name
124  stats.add("new C('o1')");
125  // object destruction
126  stats.add("destroy o");
127  // multiple object destruction
128  stats.add("destroy o1, o2");
129  // link insertion
130  stats.add("insert (o1, o2) into A");
131  // link deletion
132  stats.add("delete (o1, o2) from A");
133  // conditional execution
134  stats.add("if expr then v := 42 else v := 24 end");
135  // iteration
136  stats.add("for iter in setExpr do v := v + iter end");
137  // operation call
138  stats.add("o1.opName(argExpr1, argExpr2)");
139 
140  for (String stat1 : stats) {
141  for (String stat2 : stats) {
142  setUp();
143  String combinedStat = stat1 + "; " + stat2;
144  fResult = constructAST(combinedStat);
145  assertNotNull(fResult);
146  // while we're at it: sequence of sequences
147  combinedStat = combinedStat + "; " + combinedStat;

151  }
