174  protected Expression genNavigation(Context ctx, Token rolenameToken,
175  MClassifier srcClass,
176  Expression srcExpr,
177  MNavigableElement dst,
178  List<ASTExpression> explicitRolenameOrQualifiers,
179  List<ASTExpression> qualifiers )
180  throws SemanticException
181  {
182  Expression res = null;
183 
184  // find source end
185  MNavigableElement src = null;
186 
187  if (srcClass.equals(dst.association())) {
188  return new ExpNavigationClassifierSource(dst.cls(), srcExpr, dst);
189  }
190 
191  if (navigationNeedsExplicitRolename(srcClass, dst)) {
192  if (explicitRolenameOrQualifiers.size() == 0) {
193  // an explicit rolename is needed, but not provided
194  throw new SemanticException(
195  rolenameToken,
196  "The navigation path from "
197  + StringUtil.inQuotes(srcClass.name())
198  + " to "
199  + StringUtil.inQuotes(dst.nameAsRolename())
200  + " is ambiguous, "
201  + "but no qualification of the source association was given.");
202  }
203 
204  if (explicitRolenameOrQualifiers.size() > 1) {
205  // an explicit rolename is needed, but more than one was
206  // provided
207  throw new SemanticException(
208  rolenameToken,
209  "More then one qualification for the ambiguous navigation path from "
210  + StringUtil.inQuotes(srcClass.name())
211  + " to "
212  + StringUtil.inQuotes(dst.nameAsRolename())
213  + " was given. May be you interchanged it with qualifier values?");
214  }
215 
216  ASTExpression explicitRolenameExp = explicitRolenameOrQualifiers.get(0);
217 
218  if (!(explicitRolenameExp instanceof ASTOperationExpression)) {
219  // Must be a OperationExpression which encapsulates an IDENT
220  throw new SemanticException(rolenameToken,
221  "Invalid qualification given");
222  }
223 
224  ASTOperationExpression explicitRolenameOpExp = (ASTOperationExpression)explicitRolenameExp;
225  Token explicitRolenameToken = explicitRolenameOpExp.getOpToken();
226 
227  src = dst.association().getSourceEnd(srcClass, dst, explicitRolenameToken.getText());
228 
229  if ( src == null ) {
230  throw new SemanticException( explicitRolenameToken,
231  "Identifier `" + explicitRolenameToken.getText() + "' is not a correct"
232  + " rolename which is associated with `" + srcClass.name() + "'" );
233  }
234  } else {
235  if (!explicitRolenameOrQualifiers.isEmpty()) {
236  if (!qualifiers.isEmpty()) {
237  throw new SemanticException(rolenameToken,
238  "No qualification required for navigation path from "
239  + StringUtil.inQuotes(srcClass.name()) + " to "
240  + StringUtil.inQuotes(dst.nameAsRolename())
241  + "required!");
242  }
243 
244  // Qualifiers are provided
245  qualifiers = explicitRolenameOrQualifiers;
246  }
247 
248  src = dst.association().getSourceEnd(srcClass, dst, null);
249  }
250 
251  List<Expression> qualifierExpressions;
252  if (qualifiers.isEmpty()) {
253  qualifierExpressions = Collections.emptyList();
254  } else {
255  qualifierExpressions = new ArrayList<Expression>();
256  for (ASTExpression qualifierExp : qualifiers) {
257  qualifierExpressions.add(qualifierExp.gen(ctx));
258  }
259  }
260 
261  if ( src == null )
262  throw new SemanticException(rolenameToken,
263  "Identifier `" + rolenameToken.getText() +
264  "' is not a role name.");
265  try {
266  res = new ExpNavigation(srcExpr, src, dst, qualifierExpressions);
267  } catch (ExpInvalidException ex) {
268  throw new SemanticException(rolenameToken, ex);
269  }
270  return res;
271  }

283  protected boolean navigationNeedsExplicitRolename(MClassifier srcClass, MNavigableElement dst) {
284  // Only associations with ends > 2 can have ambiguous navigation expressions.
285  if (dst.association().reachableEnds().size() == 2) return false;
286 
287  int possibleSourceEnds = 0;
288 
289  for (MNavigableElement nav : dst.association().reachableEnds()) {
290  if (! nav.equals( dst ) ) {
291  if (srcClass.isSubClassOf(nav.cls()) ) {
292  possibleSourceEnds++;
293  }
294  }
295  }
296 
297  return possibleSourceEnds > 1;
298  }
