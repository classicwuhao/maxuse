304  public static GGeneratorArguments parseCallstring(String str) {
305  GGeneratorArguments args = new GGeneratorArguments();
306  StringTokenizer st = new StringTokenizer(str);
307 
308  boolean allOptionsFound = false;
309  boolean error = false;
310  boolean limitOptionFound = false;
311  boolean outputOptionFound = false;
312  boolean randomOptionFound = false;
313 
314  String message = null;
315 
316  try {
317  while (!allOptionsFound && !error) {
318  String optionOrFilename = st.nextToken();
319 
320  if (optionOrFilename.equals("-s")) {
321  if (!args.checkStructure())
322  error = true;
323  else
324  args.setCheckStructure(false);
325  } else if (optionOrFilename.equals("-r")) {
326  if (randomOptionFound)
327  error = true;
328  else {
329  try {
330  args.setRandomNr(new Long(st.nextToken()));
331  } catch (NumberFormatException e) {
332  error = true;
333  }
334  error = error || (args.getRandomNr().longValue() <= 0);
335  if (error)
336  message = "the parameter of the -r"
337  + " option must be a positive number"
338  + " (< 2^63).";
339  }
340  randomOptionFound = true;
341  } else if (optionOrFilename.equals("-l")) {
342  if (limitOptionFound)
343  error = true;
344  else {
345  try {
346  args.setLimit(new Long(st.nextToken()));
347  } catch (NumberFormatException e) {
348  error = true;
349  }
350  error = error || (args.getLimit().longValue() <= 0);
351  if (error)
352  message = "the parameter of the -l"
353  + " option must be a positive number"
354  + " (< 2^63).";
355  }
356  limitOptionFound = true;
357  } else if (optionOrFilename.equals("-si")) {
358  try {
359  args.setChecksBeforeSortInvariants(Integer.valueOf(st.nextToken()));
360  } catch (NumberFormatException e) {
361  error = true;
362  }
363 
364  if (error)
365  message = "the parameter of the -si"
366  + " option must be a positive number.";
367  } else if (optionOrFilename.equals("-so")) {
368  String sortOrder = st.nextToken();
369 
370  if (sortOrder.equals("c") || sortOrder.equals("combined"))
371  args.setInvariantSortOrder(InvariantSortOrder.COMBINED);
372  else if (sortOrder.equals("f") || sortOrder.equals("fastest"))
373  args.setInvariantSortOrder(InvariantSortOrder.FASTEST);
374  else if (sortOrder.equals("m") || sortOrder.equals("mostFailed"))
375  args.setInvariantSortOrder(InvariantSortOrder.MOST_FAILED);
376  else
377  error = true;
378 
379  if (error)
380  message = "Invalid sort order " + sortOrder;
381 
382  } else if (optionOrFilename.equals("-b")
383  || optionOrFilename.equals("-d")
384  || optionOrFilename.equals("-bf")
385  || optionOrFilename.equals("-df")
386  || optionOrFilename.equals("-t")
387  || optionOrFilename.equals("-dc")
388  || optionOrFilename.equals("-ac")
389  || optionOrFilename.equals("-nb")) {
390  // an output option
391  if (outputOptionFound)
392  error = true;
393  else if (optionOrFilename.equals("-b"))
394  args.setDoPrintBasics(true);
395  else if (optionOrFilename.equals("-d"))
396  args.setDoPrintDetails(true);
397  else if (optionOrFilename.equals("-bf")) {
398  args.setDoPrintBasics(true);
399  args.setPrintFilename(st.nextToken());
400  } else if (optionOrFilename.equals("-df")) {
401  args.setDoPrintDetails(true);
402  args.setPrintFilename(st.nextToken());
403  } else if (optionOrFilename.equals("-t")) {
404  args.setPrintTimeRelatedData(false);
405  } else if (optionOrFilename.equals("-dc")) {
406  args.setUseTryCuts(false);
407  } else if (optionOrFilename.equals("-ac")) {
408  args.setUseMinCombinations(false);
409  } else if (optionOrFilename.equals("-nb")) {
410  args.setCalculateBarriers(false);
411  } else
412  outputOptionFound = true;
413  } else {
414  // optionOrFilename must be a filename
415  if (optionOrFilename.startsWith("-"))
416  error = true;
417  else {
418  allOptionsFound = true;
419  if (optionOrFilename.startsWith("\"")) {
420  // Quoted filename
421  while (st.hasMoreElements()) {
422  optionOrFilename += " " + st.nextToken();
423  if (optionOrFilename.endsWith("\""))
424  break;
425  }
426  }
427 
428  args.setFilename(optionOrFilename);
429  args.setCallString(st.nextToken(""));
430  }
431  }
432  }
433  } catch (NoSuchElementException e) {
434  error = true;
435  }
436 
437  if (error) {
438  if (message != null)
439  Log.error(message);
440  else {
441  Log.error("syntax is `start [-l <num>][-r <num>][-si <num>][-so [c|f|m]][-sif<num>]"
442  + "[-b|-d|-bf <FILE>|-df <FILE>|-t|-c|-ac|-dc] "
443  + "FILE PROCNAME([paramlist])'");
444  }
445  return null;
446  }
447 
448  return args;
449  }
