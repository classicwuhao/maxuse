351  void updateInvariants() {
352  if (!fSystem.state().checkStructure(
353  new PrintWriter(new NullWriter()))) {
354  // cannot evaluate on ill-formed state
355  for (int i = 0; i < fClassInvariants.length; i++) {
356  fInvBadObjects.put(fClassInvariants[i], null);
357  }
358  }
359  for (int i = 0; i < fClassInvariants.length; i++) {
360  if(!fClassInvariants[i].isActive()){
361  continue;
362  }
363  try {
364  Set<MObject> badObjects = new HashSet<MObject>();
365  Evaluator evaluator = new Evaluator();
366  Expression expr = fClassInvariants[i]
367  .getExpressionForViolatingInstances();
368  Value v = evaluator.eval(expr, fSystem.state(),
369  new VarBindings());
370 
371  Iterator<Value> valIter = ((SetValue) v).collection().iterator();
372  while (valIter.hasNext()) {
373  ObjectValue oVal = (ObjectValue) valIter.next();
374  badObjects.add(oVal.value());
375  }
376  fInvBadObjects.put(fClassInvariants[i], badObjects);
377  } catch (Exception e) {
378  // No need to handle e.
379  }
380  }
