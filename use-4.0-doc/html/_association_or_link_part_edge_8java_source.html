166  @Override
167  public PlaceableNode findNode(double x, double y) {
168  if (!isInitialized()) return null;
169 
170  PlaceableNode res = null;
171 
172  if (fOpt.isShowRolenames() && getTargetRolename() != null
173  && getTargetRolename().isVisible()
174  && getTargetRolename().occupies(x, y)) {
175  // Do not break here. We search in the same order
176  // which is used for drawing. There may be another
177  // node drawn on top of this node. That node should be
178  // picked.
179  res = getTargetRolename();
180  } else if ( fOpt.isShowMutliplicities()
181  && getTargetMultiplicity() != null
182  && getTargetMultiplicity().isVisible()
183  && getTargetMultiplicity().occupies( x, y ) ) {
184  res = getTargetMultiplicity();
185  } else {
186  res = getNonSpecialWayPoint(x, y);
187  }
188 
189  return res;
190  }

207  void drawEdge( Graphics2D g ) {
208  EdgeProperty n1 = null;
209  Point2D p1 = null;
210  WayPoint n2 = null;
211  Point2D p2 = null;
212 
213  boolean targetNavigable = false;
214 
215  // draw all line segments
216 
217  if ( !fWayPoints.isEmpty() ) {
218  Iterator<WayPoint> it = fWayPoints.iterator();
219 
220  if ( it.hasNext() ) {
221  n1 = it.next();
222  n1.draw( g );
223  }
224 
225  while( it.hasNext() ) {
226  n2 = it.next();
227  p1 = n1.getCenter();
228  p2 = n2.getCenter();
229 
230  // draw way points
231  n2.draw( g );
232 
233  try {
234  if ( targetNavigable && n2.getSpecialID() == WayPointType.TARGET ) {
235  drawAssociationKind( g, p1, p2, DIRECTED_EDGE );
236  } else {
237  DirectedEdgeFactory.drawAssociation( g, (int) p1.getX(),
238  (int) p1.getY(),
239  (int) p2.getX(),
240  (int) p2.getY(),
241  isDashed);
242  }
243  n1 = n2;
244  } catch ( Exception ex ) {
245  ex.printStackTrace();
246  }
247  }
248  }
249  }

255  protected void drawAssociationKind( Graphics g, Point2D p2d1, Point2D p2d2, int kind ) {
256 
257  Point p1 = new Point((int)Math.round(p2d1.getX()), (int)Math.round(p2d1.getY()));
258  Point p2 = new Point((int)Math.round(p2d2.getX()), (int)Math.round(p2d2.getY()));
259 
260  try {
261  // draw association
262  switch ( kind ) {
263  case MAggregationKind.NONE:
264  DirectedEdgeFactory.drawAssociation( g, p1.x, p1.y, p2.x, p2.y, isDashed );
265  break;
266  case MAggregationKind.AGGREGATION:
267  DirectedEdgeFactory.drawAggregation( g, p1.x, p1.y, p2.x, p2.y, isDashed );
268  break;
269  case MAggregationKind.AGGREGATION + DIRECTED_EDGE:
270  DirectedEdgeFactory.drawDirectedAggregation( g, p1.x, p1.y, p2.x, p2.y, isDashed );
271  break;
272  case MAggregationKind.COMPOSITION:
273  DirectedEdgeFactory.drawComposition( g, p1.x, p1.y, p2.x, p2.y, isDashed );
274  break;
275  case MAggregationKind.COMPOSITION + DIRECTED_EDGE:
276  DirectedEdgeFactory.drawDirectedComposition( g, p1.x, p1.y, p2.x, p2.y, isDashed );
277  break;
278  case DIRECTED_EDGE:
279  DirectedEdgeFactory.drawDirectedEdge( g, p1.x, p1.y, p2.x, p2.y, isDashed );
280  break;
281  default:
282  DirectedEdgeFactory.drawAssociation( g, p1.x, p1.y, p2.x, p2.y, isDashed );
283  break;
284  }
285  } catch ( Exception ex ) {
286  ex.printStackTrace();
287  }
288  }
