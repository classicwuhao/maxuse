55  String res;
56  if (isInfixOrPrefix()) {
57  if (args.length == 1) {
58  // e.g. `not true', -2, +3
59  // insert blank between operator and expression to
60  // avoid `--' which would be interpreted as comment

62  } else
63  // e.g. `3 + 4'
