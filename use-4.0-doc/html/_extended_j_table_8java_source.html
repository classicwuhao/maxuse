100  public boolean editCellAt(int row, int column, EventObject e){
101  boolean result = super.editCellAt(row, column, e);
102  if(!result){
103  return false;
104  }
105  final Component editor = getEditorComponent();
106  if (editor == null || !(editor instanceof JTextComponent)) {
107  return result;
108  }
109  if (e instanceof KeyEvent) {
110  ((JTextComponent) editor).selectAll();
111  }
112  else if(e instanceof MouseEvent){
113  // must be run later; otherwise the click to edit the field immediately deselects the text
114  SwingUtilities.invokeLater(new Runnable() {
115  @Override
116  public void run() {
117  ((JTextComponent) editor).selectAll();
118  }
119  });
120  }
121  return result;
122  }
