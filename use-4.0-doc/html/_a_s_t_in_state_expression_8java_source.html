59  public Expression gen(Context ctx) throws SemanticException {
60  Expression res = null;
61  Expression expr;
62 
63  // check for empty source: do we have a context expression that
64  // is implicitly assumed to be the source expression?
65  if (fSourceExpr != null ) {
66  expr = fSourceExpr.gen(ctx);
67  } else {
68  ExprContext ec = ctx.exprContext();
69  if (! ec.isEmpty() ) {
70  // construct source expression
71  ExprContext.Entry e = ec.peek();
72  expr = new ExpVariable(e.fName, e.fType);
73  } else
74  throw new SemanticException(fOpToken, "Need an object to apply `" +
75  fOpToken.getText() + "'.");
76  }
77 
78  if (!expr.type().isKindOfCollection(VoidHandling.EXCLUDE_VOID) && fFollowsArrow) {
79  ctx.reportWarning(fOpToken, "application of `" + fOpToken.getText() +
80  "' to a single value should be done with `.' " +
81  "instead of `->'.");
82  }
83 
84  // If we find a source expression of type Collection(T),
85  // this might be a shorthand for the collect operation,
86  // e.g. `c.oclIsKindOf(Employee)'
87  // is a valid shorthand for `c->collect(e |
88  // e.oclIsKindOf(Employee))'
89  if (expr.type().isKindOfCollection(VoidHandling.EXCLUDE_VOID) && !fFollowsArrow) {
90  if (Options.disableCollectShorthand )
91  throw new SemanticException(fOpToken, MSG_DISABLE_COLLECT_SHORTHAND);
92 
93  CollectionType cType = (CollectionType ) expr.type();
94  Type elemType = cType.elemType();
95 
96  // transform c.oclIsKindOf(t) into c->collect($e | $e.oclIsKindOf(t))
97  ExpVariable eVar = new ExpVariable("$e", elemType);
98  Expression collectBody = genExpr(eVar);
99  res = genImplicitCollect(expr, collectBody, elemType);
100  } else {
101  res = genExpr(expr);
102  }
103 
104  return res;
105  }

107  private Expression genExpr(Expression sourceExpr) throws SemanticException
108  {
109  if (!sourceExpr.type().isTypeOfClass()) {
110  throw new SemanticException(fOpToken, "Need an object to apply `oclInState(" + this.fStateIdentifier.getText() + ")'.");
111  }
112 
113  MClass srcClass = (MClass)sourceExpr.type();
114  String stateName = fStateIdentifier.getText();
115 
116  Set<MProtocolStateMachine> psms = srcClass.getAllOwnedProtocolStateMachines();
117  if (psms.isEmpty()) {
118  throw new SemanticException(fOpToken, "Invalid use of oclInState, because the class " + StringUtil.inQuotes(srcClass) + " has no defined state machines.");
119  }
120 
121  MState state = null;
122 
123  for (MProtocolStateMachine sm : psms) {
124  state = sm.getState(stateName);
125  if (state != null) {
126  break;
127  }
128  }
129 
130  if (state == null) {
131  throw new SemanticException(
132  fOpToken,
133  "Invalid use of oclInState, because the class "
134  + StringUtil.inQuotes(srcClass)
135  + " has no state machine containing a state with the given name "
136  + StringUtil.inQuotes(stateName) + ".");
137  }
138 
139  return new ExpOclInState(sourceExpr, state);
140  }
