95  public Set<MClass> getSelectedPathClasses() {
96  Set<MClass> classes = new HashSet<MClass>();
97  ClassPathTableModel model = (ClassPathTableModel)this.fTableModel;
98 
99  for (Row<MClass> row : model.getRows()) {
100  Map<MClass, Integer> allPathes = getAllPathClasses(row.item);
101  int maxDepth = row.value;
102 
103  for (Entry<MClass, Integer> entry : allPathes.entrySet()) {
104  if (entry.getValue() <= maxDepth) {
105  classes.add(entry.getKey());
106  }
107  }
108  }
109 
110  return classes;
111  }

116  public static Map<MClass, Integer> getAllPathClasses(MClass mc) {
117 
118  Map<MClass, Integer> result = new HashMap<MClass, Integer>();
119 
120  // For FIFO handling of remaining classes to process
121  LinkedList<MClass> buffer = new LinkedList<MClass>();
122 
123  buffer.addLast(mc);
124  result.put(mc, Integer.valueOf(0));
125 
126  MClass currentClass;
127  Set<? extends MClassifier> relatedClasses;
128  int depth;
129 
130  while (buffer.size() > 0) {
131  currentClass = buffer.poll();
132  depth = result.get(currentClass).intValue() + 1;
133 
134  // Check generalizations
135  relatedClasses = currentClass.parents();
136  addNewClassesWithDepth(buffer, result, relatedClasses, depth);
137 
138  // Check the subclasses
139  relatedClasses = currentClass.children();
140  addNewClassesWithDepth(buffer, result, relatedClasses, depth);
141 
142  // Check classes related from associations
143  for (MAssociation association : currentClass.associations()) {
144  relatedClasses = association.associatedClasses();
145  addNewClassesWithDepth(buffer, result, relatedClasses, depth);
146  }
147  }
148 
149  return result;
150  }

158  private static void addNewClassesWithDepth(LinkedList<MClass> buffer, Map<MClass, Integer> result,
159  Set<? extends MClassifier> relatedClasses, int depth) {
160  if (relatedClasses != null) {
161  for (MClassifier cls : relatedClasses) {
162  if (!(cls instanceof MClass)) continue;
163 
164  if (!result.containsKey(cls)) {
165  result.put((MClass)cls, Integer.valueOf(depth));
166  buffer.add((MClass)cls);
167  }
168  }
169  }
