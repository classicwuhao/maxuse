362  /*
363  * Notify all listeners that have registered interest for
364  * notification on this event type. The event instance
365  * is lazily created using the parameters passed into
366  * the fire method.
367  */
368  public void fireStateChanged() {
369  // Guaranteed to return a non-null array
370  Object[] listeners = fListenerList.getListenerList();
371  SortChangeEvent e = null;
372  // Process the listeners last to first, notifying
373  // those that are interested in this event
374  for (int i = listeners.length-2; i >= 0; i -= 2) {
375  if (listeners[i] == SortChangeListener.class ) {
376  // Lazily create the event:
377  if (e == null) {
378  e = new SortChangeEvent(this);
379  }
380  ((SortChangeListener) listeners[i+1]).stateChanged(e);
381  }
382  }
383  }
