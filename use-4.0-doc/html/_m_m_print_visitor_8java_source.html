142  @Override
143  public void visitAssociationClass( MAssociationClass e ) {
144  indent();
145  visitAnnotations(e);
146  if ( e.isAbstract() ) {
147  print( keyword( "abstract" ) );
148  print( ws() );
149  }
150 
151  print( keyword( "associationclass" ) );
152  print( ws() );
153  print( id( e.name() ) );
154 
155  Set<? extends MClass> parents = e.parents();
156  if ( !parents.isEmpty() ) {
157  fOut.print( ws() + other( "<" ) + ws() +
158  other( StringUtil.fmtSeq( parents.iterator(), "," ) ) );
159  }
160 
161  // visit aggregation kind
162  if ( e.aggregationKind() == MAggregationKind.NONE ) {
163  // normale associationclass
164  indent();
165  println( ws() + keyword( "between" ) );
166  } else {
167  // aggregations or composition
168  indent();
169  println( ws() + keyword( MAggregationKind.name( e.aggregationKind() ) ) + ws() +
170  keyword( "between" ) );
171  }
172  incIndent();
173 
174  visitAssociationEnds( e );
175  visitAttributesAndOperations( e );
176 
177  indent();
178  println( keyword( "end" ) );
179  }

181  @Override
182  public void visitAssociationEnd(MAssociationEnd e) {
183  indent();
184  visitAnnotations(e);
185  StringBuilder result = new StringBuilder();
186 
187  result.append(id(e.cls().name()));
188  result.append(other("[" + e.multiplicity() + "]"));
189  result.append(ws());
190  result.append(keyword("role"));
191  result.append(ws());
192  result.append(id(e.name()));
193 
194  if (e.hasQualifiers()) {
195  result.append(" ");
196  result.append(keyword("qualifier"));
197  result.append(" (");
198 
199  boolean first = true;
200  for (VarDecl q : e.getQualifiers()) {
201  if (!first)
202  result.append(", ");
203  result.append(q.toString());
204  first = false;
205  }
206  result.append(")");
207  }
208 
209  if (e.getSubsettedEnds().size() > 0) {
210  for (MAssociationEnd end : e.getSubsettedEnds()) {
211  result.append(ws());
212  result.append(keyword("subsets"));
213  result.append(ws());
214  result.append(end.nameAsRolename());
215  }
216  }
217 
218  if (e.getRedefinedEnds().size() > 0) {
219  for (MAssociationEnd end : e.getRedefinedEnds()) {
220  result.append(ws());
221  result.append(keyword("redefines"));
222  result.append(ws());
223  result.append(end.nameAsRolename());
224  }
225  }
226 
227  if (e.isUnion()) {
228  result.append(ws());
229  result.append(keyword("union"));
230  }
231 
232  if (e.isOrdered()) {
233  result.append(ws());
234  result.append(keyword("ordered"));
235  }
236 
237  if (e.isDerived()) {
238  result.append(ws());
239  result.append(keyword("derived"));
240  result.append(ws());
241  result.append(other("="));
242  result.append(ws());
243  e.getDeriveExpression().toString(result);
244  }
245 
246  println(result.toString());
247  }
