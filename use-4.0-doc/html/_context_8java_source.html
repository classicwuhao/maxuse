110  public void buildVarTable(Map<String, Type> symTable)
111  {
112  Symtable newSymtable = new Symtable();
113  try {
114  for (Entry<String, Type> entry : symTable.entrySet()) {
115  newSymtable.add(entry.getKey(), entry.getValue(), null);
116  }
117  } catch (SemanticException e) {
118  // since the exception gets thrown if we add something with
119  // the same name as an existing entry, and we're adding stuff
120  // from a map (in which keys are unique), we can safely assume
121  // that we won't end up here unless someone changes the behavior
122  // of the .add method
123  System.err.println("please check org.tzi.use.parser.Context:buildVarTable()");
124  }
125 
126  fVarTable = newSymtable;
127  }
