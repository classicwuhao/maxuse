83  private void buildPartitions() {
84  // Each partition contains only combinations from one element of the unique list
85  // e.g.: {a,b} {c,d} results in:
86  // 0 = {(a,c), (a,d)}, 1 = {(b,c), (b,d)}
87  partitionedCombinations = new ArrayList<List<Pair<T>>>();
88 
89  List<T> l1 = (unique == UniqueList.FIRST_IS_UNIQUE ? secondList : firstList);
90  List<T> l2 = (unique == UniqueList.FIRST_IS_UNIQUE ? firstList : secondList);
91 
92  for (int index = 0; index < l1.size(); index++) {
93  List<Pair<T>> partition = new ArrayList<Pair<T>>();
94 
95  for (int index2 = 0; index2 < l2.size(); index2++) {
96  Pair<T> entry = new Pair<T>();
97  if (unique == UniqueList.FIRST_IS_UNIQUE) {
98  entry.first = l2.get(index2);
99  entry.second = l1.get(index);
100  } else {
101  entry.first = l1.get(index);
102  entry.second = l2.get(index2);
103  }
104  partition.add(entry);
105  }
106 
107  if (partition.size() > 0)
108  partitionedCombinations.add(partition);
109  }
110  }

112  private void initState() {
113  toDo = new Stack<MinCombinationsIterator<T>.State>();
114 
115  // Loops over all partitions of the combinations and calls combine.
116  // The tails is reduced by one after every call, because the previous call includes already the combinations
117  for (int index = 0; index < partitionedCombinations.size(); ++index) {
118  toDo.push(new State(Collections.<Pair<T>>emptyList(), partitionedCombinations.subList(index, partitionedCombinations.size())));
119  }
120 
121  // Empty LinkSet is first
122  this.nextLinkSet = Collections.emptyList();
123  }

125  private void buildNextCombination() {
126  if (toDo.isEmpty()) {
127  nextLinkSet = null;
128  return;
129  }
130 
131  State s = toDo.peek();
132 
133  List<Pair<T>> myCombinations = s.tail.get(0);
134 
135  if (s.index + 1 == myCombinations.size()) {
136  toDo.pop();
137  }
138 
139  this.nextLinkSet = new ArrayList<Pair<T>>(s.head);
140  nextLinkSet.add(myCombinations.get(s.index));
141 
142  for (int i = 1; i < s.tail.size(); ++i) {
143  List<List<Pair<T>>> newTail = s.tail.subList(i, s.tail.size());
144  toDo.push(new State(new LinkedList<Pair<T>>(nextLinkSet), newTail));
145  }
146 
147  s.index++;
148 
149  }
