84  private void printDetailedHelpByKey(String key) {

88  String detail = resource.getString(key+".detail");
89  detail = detail.replaceAll("\\n",Options.LINE_SEPARATOR + " ");

125  private void printHeader(String header) {
126  int length = 72;
127  StringBuffer buf = new StringBuffer();
128  for (int i=0; i<(length-header.length())/2; i++) {
129  buf.append("=");
130  }
131  buf.append(header);
132  for (int i=0; i<(length-header.length())/2; i++) {
133  buf.append("=");
134  }
135  if (buf.length() != length)
136  buf.append("=");
137  System.out.println(buf);
