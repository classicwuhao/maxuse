312  private void mergeTransitions(Collection<TransitionEdge> transitionsToMerge) {
313  List<TransitionEdge> toDo = new ArrayList<TransitionEdge>(transitionsToMerge);
314  // More natural for a caller, that the first element is used as a merge target
315  Collections.reverse(toDo);
316 
317  while (!toDo.isEmpty()) {
318  TransitionEdge mergeTarget = toDo.get(toDo.size() - 1);
319  toDo.remove(toDo.size() - 1);
320 
321  for (int i = toDo.size() - 1; i >= 0; --i) {
322  TransitionEdge toMerge = toDo.get(i);
323 
324  if (mergeTarget.source() == toMerge.source() &&
325  mergeTarget.target() == toMerge.target()) {
326  // Can be merged. Copy all handled transitions to mergeTarget
327  for (MTransition t : toMerge.getTransitions()) {
328  transitionMappings.put(t, mergeTarget);
329  }
330  mergeTarget.mergeTransitionEdge(toMerge);
331  fGraph.removeEdge(toMerge);
332  fEdgeSelection.remove(toMerge);
333  toDo.remove(i);
334 
335  if (mergeTarget.isReflexive()) {
336  VertexNode vNode = (VertexNode)mergeTarget.source();
337  Integer count = reflexiveTransitionCount.get(vNode.getVertex());
338  reflexiveTransitionCount.put(vNode.getVertex(), count - 1);
339  }
340  }
341  }
342  }

391  @Override
392  public void restorePlacementInfos(PersistHelper helper, int version) {
393 
394  // Restore vertex nodes
395  helper.getNav().push();
396 
397  AutoPilot ap = new AutoPilot(helper.getNav());
398 
399  try {
400  ap.selectXPath("./node");
401 
402  try {
403  while (ap.evalXPath() != -1) {
404  String name = helper.getElementStringValue("name");
405  MVertex v = parentView.stateMachine.getDefaultRegion().getSubvertex(name);
406 
407  // Could be deleted
408  if (v != null) {
409  VertexNode node = vertexMappings.get(v);
410  node.restorePlacementInfo(helper, version);
411  }
412  }
413  } catch (XPathEvalException ex) {
414  fLog.append(ex.getMessage());
415  } catch (NavException ex) {
416  fLog.append(ex.getMessage());
417  }
418  } catch (XPathParseException ex) {
419  fLog.append(ex.getMessage());
420  }
421 
422  ap.resetXPath();
423  helper.getNav().pop();
424 
425  // Restore edges
426  Set<MTransition> unhandledTransitions = new HashSet<MTransition>(this.transitionMappings.keySet());
427 
428  try {
429  ap.selectXPath("./edge");
430 
431  try {
432  while (ap.evalXPath() != -1) {
433  // Remember xpath position.
434  helper.getNav().push();
435 
436  TransitionEdge firstEdge = null;
437 
438  String source = helper.getElementStringValue("source");
439  String target = helper.getElementStringValue("target");
440 
441  helper.toFirstChild("Transitions");
442  helper.toFirstChild("Transition");
443 
444  do {
445  String trigger = helper.getElementStringValue("trigger");
446  String guard = helper.getElementStringValue("guard");
447  String post = helper.getElementStringValue("post");
448 
449  MTransition t = findTransition(unhandledTransitions, source, target, trigger, guard, post);
450 
451  if (t != null) {
452  TransitionEdge edge = transitionMappings.get(t);
453 
454  if (firstEdge == null) {
455  // Restore transition way points
456  helper.getNav().push();
457  helper.toParent();
458  helper.toParent();
459  edge.restorePlacementInfo(helper, version);
460  helper.getNav().pop();
461  firstEdge = edge;
462  } else {
463  // Merged to another edge or stand alone edge
464  if (edge != firstEdge) {
465  if (edge.getTransitions().size() == 1) {
466  fGraph.removeEdge(edge);
467  } else {
468  edge.removeTransition(t);
469  }
470 
471  firstEdge.addTransition(t);
472  transitionMappings.put(t, firstEdge);
473  }
474  }
475 
476  // We handled t
477  unhandledTransitions.remove(t);
478 
479  if (helper.toFirstChild("edgeproperty")) {
480  if (helper.getAttributeValue("type").equals("TransitionLabel")) {
481  firstEdge.restoreLabel(t, helper, version);
482  }
483  helper.toParent();
484  }
485  }
486  } while (helper.toNextSibling("Transition"));
487 
488  // Move pointer to xpath position.
489  helper.getNav().pop();
490  }
491  } catch (XPathEvalException ex) {
492  fLog.append(ex.getMessage());
493  } catch (NavException ex) {
494  fLog.append(ex.getMessage());
495  }
496  } catch (XPathParseException ex) {
497  fLog.append(ex.getMessage());
498  }
499 
500  ElementFactory f = new ElementFactory();
501 
502  // Add all unhandled transitions as new transitions
503  for (MTransition t : unhandledTransitions) {
504  TransitionEdge e = this.transitionMappings.get(t);
505  if (e.getTransitions().size() == 1) {
506  fGraph.removeEdge(e);
507  } else {
508  e.removeTransition(t);
509  }
510  this.transitionMappings.remove(t);
511  f.createTransitionEdge(t, false);
512  }
513  }

515  @Nullable
516  private MTransition findTransition(Set<MTransition> collection, String source, String target, String trigger, String guard, String post) {
517  MTransition found = null;
518  // Compare all values
519  for (MTransition t : collection) {
520  String tGuard = t.getGuard() == null ? "" : t.getGuard().toString();
521  String tPost = (t instanceof MProtocolTransition && ((MProtocolTransition) t)
522  .getPostCondition() != null) ? ((MProtocolTransition) t)
523  .getPostCondition().toString() : "";
524 
525  if (source.equals(t.getSource().name()) &&
526  target.equals(t.getTarget().name()) &&
527  trigger.equals((t.getTrigger() == null ? "" : t.getTrigger().toString())) &&
528  guard.equals(tGuard) &&
529  post.equals(tPost)) {
530 
531  if (found != null)
532  return null;
533  found = t;
534  }
535  }
536 
537  // Single transition found
538  if (found != null) return found;
539 
540  // Compare, ignore post
541  for (MTransition t : collection) {
542  if (source.equals(t.getSource().name()) &&
543  target.equals(t.getTarget().name()) &&
544  trigger.equals(t.getTrigger().toString()) &&
545  guard.equals(t.getGuard() == null ? "" : t.getGuard().toString())) {
546  if (found != null)
547  return null;
548  found = t;
549  }
550  }
551  // Single transition found
552  if (found != null) return found;
553 
554  // Compare, ignore post and guard
555  for (MTransition t : collection) {
556  if (source.equals(t.getSource().name()) &&
557  target.equals(t.getTarget().name()) &&
558  trigger.equals(t.getTrigger().toString())) {
559  if (found != null)
560  return null;
561  found = t;
562  }
563  }
564  // Single transition found
565  if (found != null) return found;
566 
567  // Compare, ignore post, guard and trigger
568  for (MTransition t : collection) {
569  if (source.equals(t.getSource().name()) &&
570  target.equals(t.getTarget().name())) {
571  if (found != null)
572  return null;
573  found = t;
574  }
575  }
576 
577  // Single transition found or null
578  return found;
579  }

718  public TransitionEdge createTransitionEdge(MTransition t, boolean mergeIfPossible) {
719  TransitionEdge result = null;
720 
721  if (mergeIfPossible) {
722  Set<EdgeBase> edges = fGraph.edgesBetween(vertexMappings.get(t.getSource()), vertexMappings.get(t.getTarget()), !t.isReflexive());
723  if (!edges.isEmpty()) {
724  result = (TransitionEdge)edges.iterator().next();
725  result.addTransition(t);
726  }
727  }
728 
729  if (result == null) {
730  TransitionEdgeCreator c = edgeCreators.get(t.getClass());
731  int reflexiveCount = reflexiveTransitionCount.get(t.getSource());
732 
733  result = c.create(t, reflexiveCount);
734 
735  if (t.isReflexive()) {
