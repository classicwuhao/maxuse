593  public WayPoint addWayPoint( double x, double y ) {
594  boolean occupies = false;
595  Line2D line = new Line2D.Double();
596  Rectangle2D testArea = new Rectangle2D.Double(x - 2, y - 2, 4, 4);
597 
598  WayPoint w1 = null;

601  // checking every line segment of this edge.
602  Iterator<WayPoint> it = fWayPoints.iterator();
603  if ( it.hasNext() ) {

605  }
606 
607  while ( it.hasNext() ) {
608  w2 = it.next();
609  line = new Line2D.Double( w1.getCenter(), w2.getCenter() );
610  occupies = line.intersects( testArea );
