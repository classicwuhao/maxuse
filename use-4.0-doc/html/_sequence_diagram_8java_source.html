759  private synchronized void createLifelines() {
760  // executed commands
761  List<Event> events = fSystem.getAllEvents();
762  // first lifeline has column 1
763  int column = 1;
764  // antecessor lifeline
765  Lifeline antecessor = null;
766  // List for saving deleted links
767  ArrayList<AssLifeline> deletedLls = new ArrayList<AssLifeline>();
768 
769  // view all commands
770  for (Event event : events) {
771  // i create command
772  if (event instanceof ObjectCreatedEvent) {
773  MObject obj = ((ObjectCreatedEvent) event).getCreatedObject();
774 
775  ObjLifeline ll = (ObjLifeline) fLifelines.get(obj);
776  // need new lifeline?
777  if (ll == null) {
778  // create new lifeline
779  ll = new ObjLifeline(column, obj, antecessor, this);
780  // set new lifeline as successor of last created
781  // lifeline
782  if (antecessor != null) {
783  antecessor.setSuccessor(ll);
784  }
785  // put new lifeline in fLifelines
786  fLifelines.put(obj, ll);
787  // lifeline already exists
788  } else {
789  // delete all saved activations for this lifeline
790  ll.activationsList = new ArrayList<Activation>();
791  }
792  // set ll as antecessor
793  antecessor = ll;
794  // next new lifeline is positioned next to this lifeline
