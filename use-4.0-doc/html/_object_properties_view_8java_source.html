132  private void update() {
133  // initialize table model
134  if ( haveObject() ) {
135  MObjectState objState = fObject.state(fSystem.state());
136  fAttributeValueMap = objState.attributeValueMap();
137 
138  Collection<MAttribute> attributes = ModelBrowserSorting.getInstance().sortAttributes( fAttributeValueMap.keySet() );
139 
140  attributes = Collections2.filter(attributes, new Predicate<MAttribute>() {
141  @Override
142  public boolean apply(MAttribute input) {
143  return !input.isDerived();
144  }
145  });
146 
147  fAttributes = Lists.newArrayList(attributes);
148  fValues = new String[fAttributes.size()];
149  for (int i = 0; i < fValues.length; i++) {
150  fValues[i] = fAttributeValueMap.get(fAttributes.get(i)).toString();
151  }
152  } else {
153  fAttributes = Collections.emptyList();
154  fValues = new String[0];
155  }
156  fireTableDataChanged();
157  }

235  private void applyChanges() {
236 
237  if (fTable.getCellEditor() != null) {
238  fTable.getCellEditor().stopCellEditing();
239  }
240 
241  if (!haveObject()) {
242  return;
243  }
244 
245  // Don't refresh after first change...
246  fSystem.getEventBus().unregister(this);
247  boolean error = false;
248 
249  for (int i = 0; i < fValues.length; ++i) {
250  MAttribute attribute = fAttributes.get(i);
251  String newValue = fValues[i];
252  String oldValue = fAttributeValueMap.get(attribute).toString();
253 
254  if (!newValue.equals(oldValue)) {
255 
256  StringWriter errorOutput = new StringWriter();
257  Expression valueAsExpression =
258  OCLCompiler.compileExpression(
259  fSystem.model(),
260  fSystem.state(),
261  newValue,
262  "<input>",
263  new PrintWriter(errorOutput, true),
264  fSystem.varBindings());
265 
266  if (valueAsExpression == null) {
267  JOptionPane.showMessageDialog(
268  fMainWindow,
269  errorOutput,
270  "Error",
271  JOptionPane.ERROR_MESSAGE);
272  error = true;
273  continue;
274  }
275 
276  try {
277  fSystem.execute(
278  new MAttributeAssignmentStatement(
279  fObject,
280  attribute,
281  valueAsExpression));
282 
283  } catch (MSystemException e) {
284  JOptionPane.showMessageDialog(
285  fMainWindow,
286  e.getMessage(),
287  "Error",
288  JOptionPane.ERROR_MESSAGE);
289  error = true;
290  }
291  }
292  }
293 
294  if (!error) update();
295 
296  fSystem.getEventBus().register(this);
297  }

306  private void updateGUIState() {
307  // temporarily turn off action listener, since setting the
308  // model triggers a select action which cannot be
309  // distinguished from a user initiated selection
310  fObjectComboBox.removeActionListener(fObjectComboBoxActionListener);
311 
312  // build list of names of currently existing objects
313  MSystemState state = fSystem.state();
314  Set<MObject> allObjects = state.allObjects();
315  ArrayList<String> livingObjects = new ArrayList<String>();
316 
317  for (MObject obj : allObjects) {
318  if (obj.exists(state) )
319  livingObjects.add(obj.name());
320  }
321 
322  if (livingObjects.isEmpty() ) {
323  livingObjects.add(NO_OBJECTS_AVAILABLE);
324  fObjectComboBox.setEnabled(false);
325  fObject = null;
326  } else
327  fObjectComboBox.setEnabled(true);
328 
329  String[] objNames = livingObjects.toArray(new String[]{});
330  Arrays.sort(objNames);
331 
332  // create combo box with available objects
333  fObjectComboBox.setModel(new DefaultComboBoxModel<String>(objNames));
334 
335  // try to keep selection
336  if (haveObject() )
337  fObjectComboBox.setSelectedItem(fObject.name());
338  fObjectComboBox.addActionListener(fObjectComboBoxActionListener);
339  }
