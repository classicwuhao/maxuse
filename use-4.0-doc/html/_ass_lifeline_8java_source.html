123  protected void drawLifeline(Graphics2D g, FontMetrics fm) {
124  // the values for the start- and end-point and the height of the
125  // area on which the diagram should be drawn -> depends on the value
126  // of fOnlyView
127  int SB_yValue;
128  if (sequenceDiagram.isOnlyView()) {
129  // set values of the visible view
130  SB_yValue = (int) sequenceDiagram.getView().getY();
131  } else {
132  // diagram starts and ends at 0
133  SB_yValue = 0;
134  }
135 
136  // y-position where lifeline starts
137  int y = fProperties.yScroll() - 20;
138  // positions where objectbox starts and lifeline ends
139  int y_start = 0, y_end = -1;
140 
141  // List of activation frames for this lifeline
142  fFrames = new ArrayList<Frame>();
143  stateNodesList = new ArrayList<StateNode>();
144 
145  // regard each actiovation of the lifeline
146  Iterator<Activation> activationIter = activationsList.iterator();
147 
148  synchronized (this) {
149  while (activationIter.hasNext()) {
150  y_start = objectBox.getYPosOfBoxStart();
151  Activation a = (Activation) activationIter.next();
152  // y-value of the activation message
153  y = a.getYOfActivationMessArrow();
154 
155  // if insert- or create-command-> calculate correct position
156  // for the object box
157  if (a.getEvent() instanceof LinkInsertedEvent) {
158  // if object box was not scrolled to the top so far
159  if (y - objectBox.getHeight() / 2 > y_start)
160  y_start = y + objectBox.getHeight() / 2;
161 
162  if (hasStatesMachine) {
163  StateNode sn = new StateNode(initState);
164  int yPosStateNode;
165  if (fProperties.getActManDist() == -1) {
166  yPosStateNode = a.getYEndOfActivation() + fProperties.actStep();
167  } else {
168  yPosStateNode = a.getYEndOfActivation() + fProperties.getActManDist();
169  }
170  setupState(sn, yPosStateNode, g.getFont());
171  stateNodesList.add(sn);
172  }
173  }
174 
175  // if position of object box is on top of the front of the
176  // diagram
177  if (y_start + objectBox.getHeight() < fProperties.yScroll() + SB_yValue) {
178  // set y-Start at the front of the diagram
179  y_start = SB_yValue + fProperties.yScroll() - objectBox.getHeight();
180  }
181  // is delete- or destroy-command
182  if (a.getEvent() instanceof LinkDeletedEvent) {
183  // lifeline ends at the position of the corresponding
184  // message
185  y_end = y;
186  // otherwise if not create-command
187  } else {
188  // position of return arrow of the current activation
189  int fEnd = a.getEnd();
190  // if frame still active
191  if (fEnd == 0)
192  fEnd = sequenceDiagram.getNumSteps();
193 
194  // calculate position subject to the user settings for
195  // the message distances
196  if (fProperties.getActManDist() == -1) {
197  fEnd = (fEnd - a.getStart()) * fProperties.actStep();
198  } else {
199  fEnd = (fEnd - a.getStart()) * fProperties.getActManDist();
200  }
201  // add new Frame to fFrames; in case of insert consider
202  // the 'inserted as' messages
203  if (a.getEvent() instanceof LinkInsertedEvent)
204  fFrames.add(new Frame(xPos, y + (objectBox.getHeight() + 1) / 2, fEnd - 10, a.getNesting()));
205  // in case of openter-command -> draw only frame if
206  // operation call was successful
207  else if (!(a.getEvent() instanceof OperationEnteredEvent && !((OperationEnteredEvent) a.getEvent()).getOperationCall()
208  .enteredSuccessfully())) {
209  fFrames.add(new Frame(xPos, y, fEnd, a.getNesting()));
210 
211  if (hasStatesMachine) {
212  if (a.getEvent() instanceof OperationEnteredEvent
213  && ((OperationEnteredEvent) a.getEvent()).getOperationCall().hasExecutedTransitions()
214  && ((OperationEnteredEvent) a.getEvent()).getOperationCall().enteredSuccessfully()) {
215 
216  Collection<Set<MTransition>> transitions = ((OperationEnteredEvent) a.getEvent()).getOperationCall().getExecutedTransitions()
217  .values();
218 
219  MVertex newState = null;
220 
221  for (Set<MTransition> transitionSet : transitions) {
222  for (MTransition trans : transitionSet) {
223  if (trans.getSource().name().equals(currentStateName)) {
224  newState = trans.getTarget();
225  currentStateName = newState.name();
226  break;
227  }
228  }
229  }
230 
231  if (newState != null) {
232  StateNode sn = new StateNode(new MState(newState.toString()));
233  int yPosStateNode;
234  if (fProperties.getActManDist() == -1) {
235  yPosStateNode = y + fEnd + fProperties.actStep();
236  } else {
237  yPosStateNode = y + fEnd + fProperties.getActManDist();
238  }
239  setupState(sn, yPosStateNode, g.getFont());
240  stateNodesList.add(sn);
241  }
242  }
243  }
244  }
245  }
246  }
247  }// end of while
248 
249  // if only inserted as-Message is send to this Lifeline -> position
250  // the object box on the front of the diagram
251  if (y_start == 0)
252  y_start = SB_yValue + fProperties.yScroll() - objectBox.getHeight();
253  // if only visible view should be drawn and the user has scrolled in
254  // vertical direction
255  if (sequenceDiagram.isOnlyView() && sequenceDiagram.getView().getY() > 0) {
256  drawZigZag(g);
257  }
258 
259  // if there was no destroy- or delete- command for this lifeline
260  if (y_end == -1) {
261  y_end = sequenceDiagram.getLastYPos() + fProperties.yStart();
262  }
263 
264  yEndPos = y_end;
265 
266  if (sequenceDiagram.getChoosedLinelines().isSelected(this)) {
267  drawSelectedBound(g, y_start, y_end);
268  }
269 
270  // draw objectBox in the right position (->y_start)
271  objectBox.drawBox(g, fm, y_start, sequenceDiagram.getChoosedLinelines().isSelected(this));
272 
273  // if only the visible view should be drawn
274  if (sequenceDiagram.isOnlyView()) {
275  // set drawing area below the zigzag pattern
276  g.clipRect((int) sequenceDiagram.getView().getX(),

278  .getWidth(), (int) sequenceDiagram.getView().getHeight() - fProperties.yScroll() - sequenceDiagram.getScrollCounter() * 4);
279  }
280 
281  // draw dashedLine in the right position and with the right height
282  drawDashedLine(xPos, y_start + objectBox.getHeight(), y_end, g);
283 
284  // draw frames for this lifeline
285  drawFrames(g);
286 
287  // regard each activation of this lifeline
288  activationIter = activationsList.iterator();
289  synchronized (this) {

291  Activation a = (Activation) activationIter.next();
292  y = a.getYOfActivationMessArrow();
293 
294  // if Object destroyed or Link deleted -> draw cross

296  y_end = y;

299  }
300  }
301 
302  if (fProperties.isStatesShown() && hasStatesMachine) {
303  // draw states for this lifeline
304  drawStateNodes(g);
