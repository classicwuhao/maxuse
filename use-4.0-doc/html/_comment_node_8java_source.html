124  public void drawCommentNodeString(Graphics2D g2d, String text, int x, int y) {
125  FontMetrics fm = g2d.getFontMetrics();
126  String[] lines = text.split("\n");
127  int totalLines = commentNodeHeight / fm.getHeight();
128  int actualLines;
129 
130  int textMaxWidth = commentNodeWidth - (2*COMMENTNODE_STRING_DISTANCE);
131 
132  if(lines.length <= totalLines) {
133  actualLines = lines.length;
134  } else {
135  actualLines = totalLines - 1;
136  }
137 
138  for (int i = 0; i < actualLines; i++) {
139  String currentLine = lines[i];
140  int currentLinePixelLength = SwingUtilities.computeStringWidth(fm, currentLine);
141  if(currentLinePixelLength+10 > textMaxWidth){
142  String ellipsis = "...";
143  int currentStringPixelLength = SwingUtilities.computeStringWidth(fm, ellipsis);
144  int currentStringCharCount = 0;
145  int totalStringCharCount = currentLine.length();
146  while(currentStringCharCount < totalStringCharCount && (currentStringPixelLength += fm.charWidth(currentLine.charAt(currentStringCharCount))) < textMaxWidth){
147  currentStringCharCount++;
148  }
149  currentLine = currentLine.substring(0, currentStringCharCount)+ ellipsis;
150  }
151  g2d.drawString(currentLine, x, y += g2d.getFontMetrics().getHeight());
152  }
153 
154  if(actualLines < lines.length) {
155  g2d.drawString("...", x, y += fm.getHeight());
156  }
157  }
