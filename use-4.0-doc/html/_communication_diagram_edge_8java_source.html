203  private void drawEdge(Graphics2D g) {
204  EdgeProperty n1 = null;
205  Point2D p1 = null;
206  WayPoint n2 = null;
207  Point2D p2 = null;
208 
209  // draw all line segments
210 
211  if (!fWayPoints.isEmpty()) {
212  Iterator<WayPoint> it = fWayPoints.iterator();
213 
214  if (it.hasNext()) {
215  n1 = it.next();
216  n1.draw(g);
217  }
218 
219  while (it.hasNext()) {
220  n2 = it.next();
221  p1 = n1.getCenter();
222  p2 = n2.getCenter();
223 
224  // draw way points
225  n2.draw(g);
226 
227  try {
228  DirectedEdgeFactory.drawAssociation(g, (int) p1.getX(), (int) p1.getY(), (int) p2.getX(), (int) p2.getY(), isDashed);
229  n1 = n2;
230  } catch (Exception ex) {
231  ex.printStackTrace();
232  }
233  }
234  }
