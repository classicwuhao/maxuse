268  int calculateMessLength() {
269  // message which will be sent by the message of this activation
270  String message = createMessage();
271  // the FontMetrics object used for drawing the diagram
272  FontMetrics fm = sequenceDiagram.getFontMetrics(fProperties.getFont());
273  // width of Activation-Message + 8 as distance to the lifelines
274  int messLength = fm.stringWidth(message) + 8;
275  // owner-lifeline of the activation
276 
277  // distance from goal-lifeline to the source-lifeline
278  // (-> difference of the column-numbers)
279  int srcDistance = ownerLifeline.column();
280  if (sourceActivation != null && sourceActivation.owner() != null) {
281  srcDistance = ownerLifeline.calculateDistance(sourceActivation.owner());
282  }
283  // if recursive activation
284  if (srcDistance == 0) {
285  messLength = fm.stringWidth(message) + 15 + 14;
286  // otherwise, if soruce-lifeline on the left side of
287  // this lifeline
288  } else {
289  // if command of activation = create
290  // TODO: This is messy
291  if (message.equals("create") || message.startsWith("create link")) {
292  // the create-arrow ends at the beginning of the
293  // ObjectBox
294  messLength += ownerLifeline.getObjectBox().getWidth() / 2;
295  // if command of activation = insert
296  }
297  // if there is a return arrow
298  if (endRowOfActivationAnswer > 0) {
299  // if openter-command
300  if (eventOfActivation instanceof OperationEnteredEvent) {
301  // calculate length of the result message
302  MOperationCall opCall =
303  ((OperationEnteredEvent)eventOfActivation).getOperationCall();
304 
305  if (fProperties.showValues()) {
306  Value result = opCall.getResultValue();
307  if (result != null) {
308  String resultLabel = result.toString();
309  if (fm.stringWidth(resultLabel) + 14 > messLength)
310  messLength = fm.stringWidth(resultLabel) + 14;
311  }
312  }
313  }
314  }
315  // consider the nestings of the corresponding lifelines
316  if (nestingOfOwnerLifeline > 0) {
317  messLength += fProperties.frWidth() / 2;
318  }
319  if (nestingOfOwnerLifeline > 1)
320  messLength += (sourceActivation.nestingOfOwnerLifeline - 1) * fProperties.frOffset();
321 
322  if (sourceActivation != null) {
323  if (sourceActivation.owner() != null && sourceActivation.nestingOfOwnerLifeline > 1)
324  messLength += (sourceActivation.nestingOfOwnerLifeline - 1)
325  * fProperties.frOffset();
326  if (sourceActivation.owner() == null || sourceActivation.nestingOfOwnerLifeline > 0)
327  messLength += fProperties.frWidth() / 2;
328  } else {
329  messLength += fProperties.frWidth() / 2;
330  }
331 
332  if (!(eventOfActivation instanceof ObjectDestroyedEvent)
333  && !(eventOfActivation instanceof LinkDeletedEvent)) {
334  if (nestingOfOwnerLifeline > 0) {
335  messLength += fProperties.frWidth() / 2;
336  }
337  }
338  }
339  // no recursive message
340  if (srcDistance != 0) {
341  // calculate distance between two lifelines subject to the
342  // message length
343  // and the distance between source- and goal-lifeline
344  messageLength = messLength / srcDistance;
345  return messLength / srcDistance;
346  // recursive message
347  } else {
348  // message is on the right side of the lifeline -> affect
349  // position
350  // of successor-lifelines -> return negative value
351  messageLength = -messLength;
352  return -messLength;
353  }
