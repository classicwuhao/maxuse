100  public void startProcedure(String callstr, GGeneratorArguments args) {
101  fLastResult = null;
102  fConfig = args;
103 
104  boolean didShowWarnigs = Log.isShowWarnings();
105  Log.setShowWarnings(false);
106 
107  GProcedureCall call = null;
108  PrintWriter pw = null;
109  PrintWriter resultPw = null;
110 
111  long startTime = System.currentTimeMillis();
112 
113  try {
114  Log.verbose("Compiling procedures from " + fConfig.getFilename() + ".");
115  fProcedures = ASSLCompiler.compileProcedures(
116  fSystem.model(),
117  new FileInputStream(fConfig.getFilename()),
118  fConfig.getFilename(),
119  new PrintWriter(System.err) );
120  if (fProcedures != null) {
121  Log.verbose("Compiling `" + callstr + "'.");
122  call = ASSLCompiler.compileProcedureCall(fSystem.model(),
123  fSystem.state(),
124  fProcedures,
125  callstr,
126  "<input>",
127  new PrintWriter(System.err)
128  );
129  }
130 
131  if (call == null) {
132  Log.error( "No procedure found for call " + callstr + " in " + fConfig.getFilename() );
133  } else {
134  resultPw = new PrintWriter(System.out);
135  if (fConfig.getPrintFilename() == null)
136  pw = resultPw;
137  else
138  pw = new PrintWriter(
139  new BufferedWriter(new FileWriter(fConfig.getPrintFilename())));
140 
141  collector = new GCollectorImpl(fConfig.doPrintBasics(), fConfig.doPrintDetails());
142  collector.setLimit(fConfig.getLimit().longValue());
143  if (fConfig.doBasicPrinting())
144  collector.setBasicPrintWriter(pw);
145  if (fConfig.doPrintDetails())
146  collector.setDetailPrintWriter(pw);
147 
148  if (fConfig.isCalculateBarriers()) {
149  call.getProcedure().calculateBarriers(collector, fModel);
150  }
151 
152  GChecker checker = new GChecker(fModel, fConfig);
153  Log.verbose(call.getProcedure().toString() + " started...");
154 
155  try {
156  GEvalProcedure evalproc = new GEvalProcedure( call.getProcedure() );
157  evalproc.eval(call.evaluateParams(fSystem.state()),
158  fSystem.state(),
159  collector,
160  checker,
161  fConfig);
162 
163  long endTime = System.currentTimeMillis();
164  fLastResult = new GResult( collector,
165  checker,
166  fConfig.getRandomNr().longValue(),
167  endTime - startTime);
168 
169  if (collector.existsInvalidMessage()) {
170  pw.print("There were errors.");
171  if (!fConfig.doBasicPrinting()) {
172  pw.print(" Use the -b or -d option to get further information.");
173  } else {
174  pw.print(" See output ");
175  if (fConfig.getPrintFilename() != null) {
176  pw.print("(" + fConfig.getPrintFilename() + ")");
177  }
178  pw.println("for details.");
179  }
180  }
181  try {
182  if (Log.isVerbose())
183  printResult(resultPw);
184  } catch (GNoResultException e) {
185  throw new RuntimeException(
186  "Although the generator computed a result, it"
187  + "is not available for printing." );
188  }
189  } catch (GEvaluationException e) {
190  internalError(e, fConfig.getRandomNr().longValue());
191  Log.error("The system state may be changed in use.");
192  } catch (StackOverflowError ex) {
193  Log.error("Evaluation aborted because of a stack " +
194  "overflow error. Maybe there were too many "+
195  "elements in a sequence of a for-loop.");
196  Log.error("The system state may be changed in use.");
197  }
198  }
199  } catch (FileNotFoundException e) {
200  Log.error( e.getMessage() );
201  } catch (IOException e) {
202  Log.error( e.getMessage() );
203  } finally {
204  if (pw != null ) {
205  pw.flush();
206  if (fConfig.getPrintFilename() != null )
207  pw.close();
208  }
209  if (resultPw != null )
210  resultPw.flush();
211 
212  Log.setShowWarnings(didShowWarnigs);
213  }
214  }

244  public void printResult(PrintWriter pw) throws GNoResultException {
245  pw.println(String.format("Random number generator was initialized with %d.", lastResult().randomNr()));
246 
247  long numSnapshots = lastResult().collector().numberOfCheckedStates();
248 
249  if (fConfig.printTimeRelatedData()) {
250  double duration = lastResult().getDuration() / 1000D;
251  double snapShotsPerSecond = Double.NaN;
252 
253  if (duration > 0) {
254  snapShotsPerSecond = (numSnapshots / duration);
255  }
256 
257  pw.println(String.format("Checked %,d snapshots in %,.3fs (%,.0f snapshots/s).", numSnapshots, duration, snapShotsPerSecond));
258  } else {
259  pw.println(String.format("Checked %,d snapshots.", numSnapshots));
260  }
261 
262  if (!lastResult().collector().getBarriers().isEmpty()) {
263  long numBarrierChecks = lastResult().collector().getBarriersHit();
264  long overallChecks = numSnapshots + numBarrierChecks;
265 
266  double duration = lastResult().getDuration() / 1000D;
267  double checksPerSecond = Double.NaN;
268 
269  if (duration > 0) {
270  checksPerSecond = (overallChecks / duration);
271  }
272 
273  if (fConfig.printTimeRelatedData())
274  pw.println(String.format("Checks including barriers: %,d (%,.0f checks/s).", overallChecks, checksPerSecond));
275  else
276  pw.println(String.format("Checks including barriers: %,d.", overallChecks));
277  }
278 
279  if (fConfig.useTryCuts())
280  pw.println(String.format("Made %,d try cuts.", lastResult().collector().getCuts()));
281 
282  if (fConfig.useMinCombinations())
283  pw.println(String.format("Ignored at least %,d useless link combinations.", lastResult().collector().getIgnoredStates()));
284 
285  if (fConfig.isCalculateBarriers())
286  pw.println(String.format("Added %,d barriers.", lastResult().collector().getNumCalculatedBarriers()));
287 
288  pw.println(String.format("Barriers blocked %,d times.", lastResult().collector().getBarriersHit()));
289 
290  if (lastResult().collector().limit() != Long.MAX_VALUE)
291  pw.println(String.format("Limit was set to %,d.", lastResult().collector().limit()));
292  if (!lastResult().collector().validStateFound() )
293  pw.println("Result: No valid state found.");
294  else {
295  pw.println("Result: Valid state found.");
296  pw.println("Commands to produce the valid state:");
297  for (MStatement statement : lastResult().collector().statements()) {
298  pw.println(statement.getShellCommand());
299  }
300  if (lastResult().collector().statements().isEmpty())
301  pw.println("(none)");
302  }
