156  public void testExpression() throws IOException {
157  MModel model = new ModelFactory().createModel("Test");
158  // read expressions and expected results from file
159  BufferedReader in = new BufferedReader(new FileReader(TEST_EXPR_FILE));
160  int lineNr = 0;
161 
162  while (true) {
163  String line = in.readLine();
164  lineNr++;
165 
166  if (line == null) {
167  break;
168  }
169  if (line.startsWith("##")) {
170  if (VERBOSE) {
171  System.out.println("testing " + line.substring(2).trim());
172  }
173  continue;
174  }
175  if (line.length() == 0 || line.startsWith("#")) {
176  continue;
177  }
178 
179  String expStr = line;
180  while (true) {
181  line = in.readLine();
182  lineNr++;
183 
184  if (line == null) {
185  in.close();
186  throw new RuntimeException("missing result line");
187  }
188  if (line.startsWith("-> ")) {
189  break;
190  }
191  expStr += " " + line.trim();
192  }
193  String resultStr = line.substring(3);
194 
195  if (VERBOSE) {
196  System.out.println("expression: " + expStr);
197  }
198 
199  InputStream stream = new ByteArrayInputStream(expStr.getBytes());
200 
201  Expression expr =
202  OCLCompiler.compileExpression(
203  model,
204  stream,
205  TEST_EXPR_FILE,
206  new PrintWriter(System.err),
207  new VarBindings());
208  assertNotNull(expr + " compiles", expr);
209 
210  MSystemState systemState = new MSystem(model).state();
211 
212  Value val = new Evaluator().eval(expr, systemState);
213  assertEquals(TEST_EXPR_FILE + ":" + lineNr + " evaluate: " + expStr, resultStr, val.toStringWithType());
214  }
215 
216  in.close();
217  }
