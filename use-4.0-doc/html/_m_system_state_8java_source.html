756  throws MSystemException {

760  PrintWriter out = new PrintWriter(sw);
761

763  throw new MSystemException("Cannot create link for association with derived end.");
764  }
765 
766  validateLinkQualifiers(assoc, qualifierValues);
767 
768  if (!validateRedefinesForLink(assoc, objects, out)) {
769  throw new MSystemException(sw.toString());
770  }
771 
772  // checks if assoc is an associationclass
773  if (assoc instanceof MAssociationClass) {
774  // specifies if there is already an existing linkobject of this
775  // association class between the objects
776  if (hasLinkBetweenObjects(assoc, objects, qualifierValues)) {
777  throw new MSystemException(
778  "Cannot insert two linkobjects of the same type"
779  + " between one set of objects!");
780  }
781

783 
784  // creates a linkobject with a generated name
785  link = createLinkObject((MAssociationClass) assoc, name, objects, qualifierValues);
786 
787  } else if ((assoc.aggregationKind() == MAggregationKind.AGGREGATION)
788  || (assoc.aggregationKind() == MAggregationKind.COMPOSITION)) {
789 
790  MLinkSet linkSet = fLinkSets.get(assoc);
791  link = new MLinkImpl(assoc, objects, qualifierValues);
792  if (linkSet.contains(link))
793  throw new MSystemException("Link `" + assoc.name()
794  + "' between ("
795  + StringUtil.fmtSeq(objects.iterator(), ",")
796  + ") already exist.");
797 
798  // The graph to store the information of the whole/part hierachy
799  MWholePartLink wholePartLink = new MWholePartLinkImpl(link);
800  MObject source = wholePartLink.source();
801  MObject target = wholePartLink.target();
802 
803  fWholePartLinkGraph.add(source);
804  fWholePartLinkGraph.add(target);
805 
806  // the link is irreflexive
807  if (wholePartLink.isReflexive())
808  Log.warn("Warning: Object `" + source.name()
809  + "' cannot be a part of itself.");
810

812  if (assoc.aggregationKind() == MAggregationKind.COMPOSITION) {
813  // finding all edges (links) whose target is the target object
814  // of the new link
815  Iterator<MObject> sourceEdgeIter = fWholePartLinkGraph.sourceNodeSet(target).iterator();
816

819  sourceEdgeIter.next()).iterator();

828  + "'. Object `"
829  + target.name()
830  + "' is already component of another object.");
831  }
832  }
833  }
834  }

836  // whole/part link
837  if (fWholePartLinkGraph.existsPath(target, source))
838  Log.warn("Warning: Insert has resulted in a cycle in the part-whole hierarchy. Object `"

840  + "' is a direct or indirect part of `"
841  + target.name() + "'.");

852  + "' between ("

854  + ") already exist.");

856  }
