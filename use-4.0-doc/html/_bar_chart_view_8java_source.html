71  private Dimension calcPreferredSize() {
72  FontMetrics fm = getFontMetrics(fFont);
73  fMaxNameWidth = fm.stringWidth(fHeader1);
74  for (int i = 0; i < fNames.length; i++)
75  fMaxNameWidth = Math.max(fMaxNameWidth,
76  fm.stringWidth(fNames[i].toString()));

93  public synchronized void setValues(int[] values) {
94  if (values.length != fNames.length )
95  throw new IllegalArgumentException("values.length != fNames.length");
96  fMaxValue = Integer.MIN_VALUE;
97  fValues = values;
98  for (int i = 0; i < fValues.length; i++)
99  if (fValues[i] > fMaxValue )
100  fMaxValue = fValues[i];
101  if (fMaxValue == 0 )
102  fMaxValue = 1;
103  repaint();
104  }
