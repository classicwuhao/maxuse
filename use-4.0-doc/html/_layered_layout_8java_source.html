119  private void layerWalk(LayoutNode node, int layer) {
120  Set<LayoutNode> predecessors = fOutGraph.sourceNodeSet(node);

180  private void placeNodesOnLayers() {
181  System.out.println("placeNodesOnLayers: " + fOutGraph);
182  //System.out.println("fHeight = " + fHeight);
183 
184  // determine size of each layer and set a relative (arbitrary)
185  // order for each node on its layer
186  int[] layerSize = new int[fHeight];
187  Iterator<LayoutNode> nodeIter = fOutGraph.iterator();
188  while (nodeIter.hasNext() ) {
189  LayoutNode node = nodeIter.next();
190  node.fX = node.fLayerX = layerSize[node.fLayer]++;
191  }
192 
193  // create layer array
194  int width = 0;
195  LayoutNode[][] layers = new LayoutNode[fHeight][];
196  for (int i = 0; i < fHeight; i++) {
197  int layerWidth = layerSize[i];
198  if (layerWidth > width)
199  width = layerWidth;
200  layers[i] = new LayoutNode[layerWidth];
201  }
202 
203  // place nodes into layer array
204  nodeIter = fOutGraph.iterator();
205  while (nodeIter.hasNext() ) {
206  LayoutNode node = (LayoutNode) nodeIter.next();
207  layers[node.fLayer][node.fLayerX] = node;
208  }
209 
210  // place sinks on bottom layer
211  int numSinks = layers[0].length;
212  if (numSinks == 1 ) {
213  layers[0][0].fX = width / 2;
214  } else {
215  numSinks--;
216  for (int j = 0; j <= numSinks; j++)
217  layers[0][j].fX = j * (width - 1) / numSinks;
218  }
219 
220  // layout layers bottom-up
221  for (int i = 1; i < fHeight; i++) {
222  BitSet occupiedPositions = new BitSet(width);
223  for (int j = 0; j < layers[i].length; j++) {
224  LayoutNode u = layers[i][j];
225  // calculate barycenter of neighbors
226  int sum = 0;
227  Set<LayoutNode> neighbors = fOutGraph.targetNodeSet(u);
228  nodeIter = neighbors.iterator();
229  while (nodeIter.hasNext() ) {
230  LayoutNode v = (LayoutNode) nodeIter.next();
231  sum += v.fX;
232  }
233  int x = sum / neighbors.size();
234  int xd = 0;
235  int xn;
236  while (true ) {
237  xn = x + xd;
238  if (xn < width && ! occupiedPositions.get(xn) )
239  break;
240  xn = x - xd;
241  if (xn >= 0 && ! occupiedPositions.get(xn) )
242  break;
243  xd++;
244  }
245  occupiedPositions.set(xn);
246  u.fX = xn;
247  }
248  }
