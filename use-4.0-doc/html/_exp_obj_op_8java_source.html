77  public Value eval(EvalContext ctx) {
78  if (isPre()) {
79  ctx = new EvalContext(ctx.preState(), ctx.preState(), ctx.varBindings(), ctx);
80  }
81 
82  ctx.enter(this);
83 
84  Value result = UndefinedValue.instance;
85 
86  Value selfVal = fArgs[0].eval(ctx);
87 
88  if (selfVal.isUndefined() ||
89  !(selfVal instanceof ObjectValue)) {
90 
91  ctx.exit(this, result);
92  return result;
93  }
94 
95  MObject self = ((ObjectValue)selfVal).value();
96 
97  if ((isPre() && (self.state(ctx.preState()) == null)) ||
98  (!isPre() && (self.state(ctx.postState()) == null))) {
99 
100  ctx.exit(this, result);
101  return result;
102  }
103 
104  MOperation operation = self.cls().operation(fOp.name(), true);
105 
106  if (!operation.isCallableFromOCL()) {
107  throw new RuntimeException("Cannot call operation " + operation);
108  }
109 
110  List<String> parameterNames = operation.paramNames();
111  Value[] arguments = new Value[parameterNames.size()];
112  for (int i = 1; i < fArgs.length; ++i) {
113  arguments[i-1]= fArgs[i].eval(ctx);
114  }
115 
116  // this must be done _after_ all parameters have been evaluated,
117  // since the parameter names could shadow values which are
118  // needed for a later parameter (see test\t005.*)
119  ctx.pushVarBinding("self", selfVal);
120  for(int i = 0; i < parameterNames.size(); ++i) {
121  ctx.pushVarBinding(
122  parameterNames.get(i),
123  arguments[i]);
124  }
125 
126  MOperationCall operationCall =
127  new MOperationCall(this, self, operation, arguments);
128 
129  operationCall.setPreferredPPCHandler(ExpressionPPCHandler.getDefaultOutputHandler());
130 
131  MSystem system = ctx.postState().system();
132 
133  try {
134  system.enterQueryOperation(ctx, operationCall, false);
135  operationCall.setExecutionFailed(true);
136 
137  if (operation.hasExpression()) {
138  result = operation.expression().eval(ctx);
139  }
140 
141  operationCall.setExecutionFailed(false);
142  } catch (MSystemException e) {
143  throw new RuntimeException(e);
144  } finally {
145  try {
146  if (operationCall.enteredSuccessfully()) {
147  system.exitQueryOperation(ctx, result);
148  }
149  } catch (Exception e){ }
150  ctx.popVarBindings(fArgs.length);
151  ctx.exit(this, result);
152  }
153 
154  return result;
155  }
