365  addChildNodes( top, "Invariants", sortedInvariants );

369  addChildNodes( top, "Pre-/Postconditions", sortedConditions );

371  Set<Map.Entry<String, Collection<?>>> modelCollectionEntrySet = this.modelCollections.entrySet();
372

409  public void stateChanged( SortChangeEvent e ) {
410  ArrayList<Integer> pathWereExpanded = new ArrayList<Integer>();
411  int selectedRow = -1;
412 
413  // which nodes are expanded
414  for ( int i=0; i<fTree.getRowCount(); i++ ){
415  if ( fTree.isExpanded( i ) ){
416  pathWereExpanded.add( i );
417  }
418  // which node is selected
419  if ( fTree.isRowSelected( i ) ) {
420  selectedRow = i;
421  }
422  }
423 
424  fTop.removeAllChildren();
425  createNodes( fTop );
426  fTreeModel.reload();
427  fHtmlPane.setText( "" );
428 
429  // expand all nodes that were expanded.
430  for ( int i=0; i<pathWereExpanded.size(); i++ ){
431  fTree.expandRow( pathWereExpanded.get(i).intValue() );
432  }
433  // set selected node again.
434  if ( selectedRow >= 0 ) {
435  fTree.setSelectionRow( selectedRow );
436  }
437  }

465  if (listeners[i] == HighlightChangeListener.class ) {
466  // Lazily create the event:

468  e = new HighlightChangeEvent(this);
469  e.setModelElement( elem );
470  e.setHighlight( highlight );
471  }
472  ((HighlightChangeListener) listeners[i+1]).stateChanged(e);
473  }
474  }
