520  @Override

522  return currentIterator.hasNext() || !toDo.isEmpty();

525  @Override
526  public N next() {
527  if (currentIterator.hasNext()) {
528  return currentIterator.next();
529  } else if (!toDo.isEmpty()) {
530  ++depth;
531  Set<N> targetNodes = getNodeSet(toDo.pop());
532  addToDo(targetNodes);
533  // Only nodes with have connected nodes are added to toDo.
534  // Therefore, we can safely use the iterator next.
535  currentIterator = targetNodes.iterator();
536  return currentIterator.next();
537  } else {
538  throw new NoSuchElementException();
539  }
540  }
