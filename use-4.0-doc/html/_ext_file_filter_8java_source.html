163  public boolean accept(File f) {

182  public String getExtension(File f) {
183  if (f != null) {
184  String filename = f.getName();
185  int i = filename.lastIndexOf('.');
186  if (i>0 && i<filename.length()-1) {
187  return filename.substring(i+1).toLowerCase();
188  }
189  }
190  return null;
191  }

223  public String getDescription() {
224  if (fullDescription == null) {
225  if (description == null || isExtensionListInDescription()) {
226  if (description != null) {
227  fullDescription = description;
228  }
229  fullDescription += " (";
230  // build the description from the extension list
231  Iterator<String> extensions = filters.keySet().iterator();
232  if (extensions != null) {
233  fullDescription += "." + (String) extensions.next();
234  while (extensions.hasNext()) {
235  fullDescription += ", " + (String) extensions.next();
236  }
237  }
238  fullDescription += ")";
239  } else {
240  fullDescription = description;
241  }
242  }
243  return fullDescription;
244  }
