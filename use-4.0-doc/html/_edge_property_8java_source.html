241  if ( isSelected() ) {

245  FontMetrics fm = g2.getFontMetrics();
246  int lineHeight = fm.getAscent() + fm.getDescent();
247 
248  int y = (int)Math.round(getY() + fm.getAscent() + MARGIN_VERTICAL / 2);
249  int x = (int)Math.round(getX() + MARGIN_HORIZONTAL / 2);
250 
251  String[] lines = fName.split("\n");
252  for (String line : lines) {
253  g2.drawString(line, x, y);
254  y += lineHeight;
255  }
256  }
