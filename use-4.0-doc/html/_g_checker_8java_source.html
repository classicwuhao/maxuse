89  public boolean check(final MSystemState state, final IGCollector collector) {
90  // resort the invariants starting every 10.000th check.
91  // invariants, which are often invalid, will be checked first.
92  if (sortCounter == checksBeforeSort) {
93  sortCounter = 0;
94 
95  Arrays.sort(fInvariantStatistics, invariantComparator);
96  ++sortCount;
97 
98  for (int i = 0; i < fInvariantStatistics.length; ++i)
99  fInvariantStatistics[i].localReset();
100 
101  } else {
102  ++sortCounter;
103  }
104 
105  // evaluating invariants
106  boolean result = true;
107  long start;
108  boolean valid;
109  Value value;
110 
111  for (int k = 0; k < fSize && result; k++) {
112  GInvariantStatistic stat = fInvariantStatistics[k];
113  if (stat.getInvariant().isActive() && !stat.isCheckedByBarrier() ) {
114 
115  try {
116  start = System.nanoTime();
117 
118  value = fEvaluator.eval( stat.getInvariant().expandedExpression(), state );
119 
120  valid = value.isDefined() && ((BooleanValue) value).isFalse() == stat.getInvariant().isNegated();
121 
122  stat.registerResult(valid, System.nanoTime() - start);
123 
124  if (!valid) {
125  if (collector.doBasicPrinting())
126  collector.basicPrintWriter().println(stat.getInvariant().toString() + " invalid.");
127 
128  result = false;
129  break;
130  }
131  } catch (MultiplicityViolationException e) {
132  if (collector.doDetailPrinting()) {
133  collector.detailPrintWriter().println("An error occured while checking an invariant:");
134  collector.detailPrintWriter().println(e.getMessage());
135  }
136  stat.registerException();
137  return false;
138  }
139  }
140  }
141 
142  // checking structure
143  if (result && fCheckStructure) {
144  start = System.nanoTime();
145  result = state.checkStructure(NullPrintWriter.getInstance(), false);
146  if (!result && collector.doBasicPrinting()) {
147  collector.basicPrintWriter().println("invalid structure.");
148  }
149  fStructureStatistic.registerResult(result, System.nanoTime() - start);
150  }
151 
152  if (result && collector.doBasicPrinting())
153  collector.basicPrintWriter().println("valid state.");
154 
155  return result;
156  }
