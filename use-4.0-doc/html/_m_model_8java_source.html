219  public void addAssociation(MAssociation assoc)
220  throws MInvalidModelException {
221  if (assoc.associationEnds().size() < 2) {
222  if (!(assoc instanceof MAssociationClass) || ((MAssociationClass)assoc).parents().isEmpty())
223  throw new IllegalArgumentException("Illformed association `"
224  + assoc.name() + "': number of associationEnds == "
225  + assoc.associationEnds().size());
226  }
227 
228  if (fAssociations.containsKey(assoc.name()))
229  throw new MInvalidModelException(
230  "Model already contains an association named `"
231  + assoc.name() + "'.");
232 
233  if (!(assoc instanceof MAssociationClass) && fClasses.containsKey(assoc.name()))
234  throw new MInvalidModelException("Model already contains a class `" + assoc.name() + "'.");
235 
236  // check for role name conflicts: for each class the set of
237  // navigable classes must have unique role names
238  for (MClass cls : assoc.associatedClasses()) {
239  Map<String, ? extends MNavigableElement> aends = cls.navigableEnds();
240  List<String> newRolenames = new ArrayList<String>();
241 
242  for (MNavigableElement elem : assoc.navigableEndsFrom(cls)) {
243  String newRolename = elem.nameAsRolename();
244 
245  newRolenames.add( newRolename );
246 
247  if (aends.containsKey(newRolename)) {
248  // Inherited?
249  boolean inherited = false;
250  MNavigableElement otherEnd = aends.get(newRolename);
251 
252  if (otherEnd.association() instanceof MAssociationClass && assoc instanceof MAssociationClass) {
253  MAssociationClass otherCls = (MAssociationClass)otherEnd.association();
254  MAssociationClass ourCls = (MAssociationClass)assoc;
255 
256  if (ourCls.allParents().contains(otherCls)) {
257  inherited = true;
258  }
259  }
260 
261  if (!inherited) {
262  throw new MInvalidModelException("Association end `"
263  + newRolename
264  + "' navigable from class `"
265  + cls.name()
266  + "' conflicts with same rolename in association `"
267  + ((MNavigableElement) aends.get(newRolename))
268  .association().name() + "'.");
269  }
270  }
271  }
272 
273  // tests if the rolenames are already used in one of the subclasses
274  for (MClass subCls : CollectionUtil.<MClassifier, MClass>downCastUnsafe(cls.allChildren())) {
275  for ( int i = 0; i < newRolenames.size(); i++ ) {
276  String newRolename = newRolenames.get(i);
277  if ( subCls.navigableEnds().containsKey( newRolename ) ) {
278  throw new MInvalidModelException("Association end `"
279  + newRolename
280  + "' navigable from class `"
281  + subCls.name()
282  + "' conflicts with same rolename in association `"
283  + ((MNavigableElement) subCls.navigableEnds()
284  .get( newRolename ))
285  .association().name() + "'.");
286  }
287  }
288  }
289  }
290 
291  // for each class register the association and the
292  // reachable association ends
293  for (MAssociationEnd aend : assoc.associationEnds()) {
294  MClass cls = aend.cls();
295  cls.registerNavigableEnds(assoc.navigableEndsFrom(cls));
296  }
297 
298  assoc.setModel(this);
299  fGenGraph.add(assoc);
300 
301  fAssociations.put(assoc.name(), assoc);
302  }

363  public void addGeneralization(MGeneralization gen)
364  throws MInvalidModelException {
365  // generalization is irreflexive
366  MClassifier child = gen.child();
367  if (gen.isReflexive()) {
368  throw new MInvalidModelException("Class `" + child
369  + "' cannot be a superclass of itself.");
370  }
371 
372  // check for cycles that might be introduced by adding the new
373  // generalization
374  MClassifier parent = gen.parent();
375  if (fGenGraph.existsPath(parent, child)) {
376  throw new MInvalidModelException(
377  "Detected cycle in generalization hierarchy. Class `"
378  + parent.name()
379  + "' is already a subclass of `"
380  + child.name() + "'.");
381  }
382 
383  if(!parent.getClass().isAssignableFrom(child.getClass())) {
384  throw new MInvalidModelException(
385  "Invalid inheritance relation between meta elements "
386  + StringUtil.inQuotes(child.getClass()
387  .getSimpleName()
388  + "::" + child.name())
389  + " < "
390  + StringUtil.inQuotes(parent.getClass()
391  .getSimpleName()
392  + "::" + parent.name()));
393  }

396  final boolean parentIsAssocClass = parent instanceof MAssociationClass;

406  throw new MInvalidModelException("Association classes can only inherit from association classes.");
