75  protected MLinkImpl(MAssociation assoc, List<MObject> objects, List<List<Value>> qualifierValues) throws MSystemException {
76  fAssociation = assoc;
77  if (assoc.associationEnds().size() != objects.size() )
78  throw new IllegalArgumentException("Number of association ends (" +
79  assoc.associationEnds().size() +
80  ") does not match number of passed objects (" +
81  objects.size() + ")");
82 
83  linkEnds = new MLinkEnd[assoc.associationEnds().size()];
84  this.qualifierValues = qualifierValues;
85  Iterator<MAssociationEnd> it1 = assoc.associationEnds().iterator();
86  Iterator<MObject> it2 = objects.iterator();
87  boolean hasQualifiers = (qualifierValues != null && qualifierValues.size() > 0);
88  Iterator<List<Value>> it3 = (hasQualifiers ? qualifierValues.iterator() : null);
89 
90  int newHashCode = fAssociation.hashCode();
91  linkedObjects = new MObject[objects.size()];
92  int i = 0;
93 
94  while (it1.hasNext() && it2.hasNext() ) {
95  MAssociationEnd aend = it1.next();
96  MObject obj = it2.next();
97  List<Value> endQualifierValues;
98 
99  if (hasQualifiers)
100  endQualifierValues = it3.next();
101  else
102  endQualifierValues = null;
103 
104  MLinkEnd lend = new MLinkEnd(aend, obj, endQualifierValues);
105  newHashCode += lend.hashCode();
106  linkEnds[i] = lend;
107  linkedObjects[i] = obj;
108  ++i;
109  }
110 
111  this.hashCode = newHashCode;
112  }
