604  private synchronized int getLongestMessage(FontMetrics fm) {
605  int returnValue = 0;
606  // regard each activation of this lifeline
607  for (int i = 0; i < activationsList.size(); i++) {
608  Activation a = (Activation) activationsList.get(i);
609  // length of the longest message of the regarded activation
610  int messLength = a.getMessLength();
611  // if bigger than fMaxMessLength -> new fMaxMessLength
612  if (messLength > 0 && messLength > maxOfMessLength)
613  maxOfMessLength = messLength;
614  // if messLength < 0 and longest message of successor is smaller
615  // then -messLength
616  // -> set fMaxMessLangth of successor
617  else if (successor != null && messLength < 0 && -messLength > successor.getMaxMessLength())
618  successor.setMaxMessLength(-messLength);
619  // if negative messLength is smaller than returnValue
620  // -> set returnValue = messLength
621  if (messLength < 0 && returnValue > messLength)
622  returnValue = messLength;
623  }
624  return returnValue;
625  }

650  void calculatePosition(FontMetrics fm) {
651  // column of this lifeline in the diagram
652  int column = column();
653  // antecessor of this lifeline -> must not be drawn in the diagram
654  Lifeline ll = antecessor;
655  // calculate drawn antecessor lifeline
656  while (ll != null && (!ll.isDraw() || ll.isHidden())) {
657  ll = ll.getAntecessor();
658  }
659  // calculate longest message for this lifeline
660  int returnValue = getLongestMessage(fm);
661  // beginning of the diagram
662  int xValue = fProperties.getLeftMargin() + fProperties.frWidth();
663  // if there is no lifeline chosen by the user
664  if (sequenceDiagram.getChoosedLinelines().isEmpty()) {
665  // calculate xValue and in case of fDraw=true fXValue
666  // subject to the user settings for the distance
667  // of two lifelines
668  if (!fProperties.llLikeLongMess()) {
669  if (!fProperties.individualLl()) {
670  xValue = fProperties.getLeftMargin() + fProperties.frWidth() + column * fProperties.llStep();
671  } else {
672  xValue = xPos;
673  }
674  if (fDraw) {
675  xPos = xValue;
676  }
677  } else {
678  if (fProperties.individualLl()) {
679  if (ll != null) {
680  xValue = ll.xValue() + maxOfMessLength;
681  } else {
682  xValue += maxOfMessLength;
683  if (maxOfMessLength == 0) {
684  xValue += fProperties.frWidth() / 2 + objectBox.getWidth() / 2;
685  }
686  }
687  } else {
688  xValue += fProperties.maxActMess() * column;
689  }
690  if (fDraw) {
691  xPos = xValue;
692  }
693  if (fProperties.individualLl() && ll != null && (ll.xValue() + ll.getObjectBox().getWidth() / 2 > xValue - objectBox.getWidth() / 2 - 5)) {
694  xPos = ll.xValue() + ll.getObjectBox().getWidth() / 2 + objectBox.getWidth() / 2 + 5;
695  }
696  }
697  }
698  // set x-value of the object box
699  objectBox.setX(xPos);
700  // calculate start- and end-x-value of the object box
701  objectBox.calculateStart(fm);
702  objectBox.calculateEnd(fm);
703 
704  // calculate last x-value of this lifeline
705  int lastX = xPos + objectBox.getWidth() / 2;
706  if (returnValue < 0 && xPos + objectBox.getWidth() / 2 - returnValue > sequenceDiagram.getLastXValue())
707  lastX = xPos + objectBox.getWidth() / 2 - returnValue;
708  if (sequenceDiagram.getLastXValue() < lastX && fDraw) {
709  sequenceDiagram.setLastXValue(lastX);
710  }
711 
712  // calculate width of object box

714  // if there was no wider box so far -> set maximum width in
715  // fObProperties
716  if (fObProperties.maxWidth(labelWidth))
717  fObProperties.setMaxWidth(labelWidth);
718  // calculate height of object box
719  int labelHeight = fProperties.getFont().getSize();
720  // if there was no higher box so far -> set maximum height in
721  // fObProperties
722  if (fObProperties.maxHeight(labelHeight))
723  fObProperties.setMaxHeight(labelHeight);
