104  // generate the condition expression and check if it's boolean
105  Expression condition = generateExpression(fCondition);
106  if (!condition.type().isTypeOfBoolean()) {
107  throw new CompilationFailedException(this, "Expression "
108  + StringUtil.inQuotes(fCondition.getStringRep())

112  }
113 
114  // generate the then and else statements
115  MStatement[] compiledStatements = new MStatement[2];
116  ASTStatement[] statements = { fThenStatement, fElseStatement };
117 
118  for (int i = 0; i < 2; ++i) {
119  fSymtable.storeState();
120  compiledStatements[i] = generateStatement(statements[i]);
121  fSymtable.restoreState(this);
122 
123  // assigned(conditional) = assigned(then) U assigned(else)
124  fAssignedSet.add(statements[i].fAssignedSet);
125  }
126 
127  return new MConditionalExecutionStatement(
128  condition,
129  compiledStatements[0],
130  compiledStatements[1]);
