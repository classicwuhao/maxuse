86  public void layout() {
87  final int N = fNodes.size();
88  final double k1 = 1;
89  final double k2 = 10000;
90 
91  double xc = 0.0;
92  double yc = 0.0;
93  for (int i = 0; i < N; i++) {
94  N v = fNodes.get(i);
95 
96  double xv = v.getCenter().getX();
97  double yv = v.getCenter().getY();
98 
99  // spring force
100  Iterator<N> uIter = fGraph.sourceNodeSet(v).iterator();
101  double sumfx1 = 0.0;
102  double sumfy1 = 0.0;
103  while (uIter.hasNext() ) {
104  N u = uIter.next();
105 
106  double xu = u.getCenter().getX();
107  double yu = u.getCenter().getY();
108  double dx = xv - xu;
109  double dy = yv - yu;
110  double d = Math.sqrt(dx * dx + dy * dy);
111  d = (d == 0) ? .0001 : d;
112  double c = k1 * (d - fEdgeLen) / d;
113  sumfx1 += c * dx;
114  sumfy1 += c * dy;
115  }
116 
117 
118  // electrical force
119 
120  uIter = fGraph.iterator();
121  double sumfx2 = 0.0;
122  double sumfy2 = 0.0;
123  while (uIter.hasNext() ) {
124  N u = uIter.next();
125  if (u == v )
126  continue;
127 
128  double xu = u.getCenter().getX();
129  double yu = u.getCenter().getY();
130  double dx = xv - xu;
131  double dy = yv - yu;
132  double d = dx * dx + dy * dy;
133  if (d > 0 ) {
134  double c = k2 / (d * Math.sqrt(d));
135  sumfx2 += c * dx;
136  sumfy2 += c * dy;
137  }
138  }
139 
140  // store new positions
141  fXn[i] = xv - Math.max(-5, Math.min(5, sumfx1 - sumfx2));
142  fYn[i] = yv - Math.max(-5, Math.min(5, sumfy1 - sumfy2));
143 
144  // for determining the center of the graph
145  xc += fXn[i];
146  yc += fYn[i];
147  }
148 
149  // offset from center of graph to center of drawing area
150  double dx = fWidth / 2 - xc / N;
151  double dy = fHeight / 2 - yc / N;
152 
153  // use only small steps for smooth animation
154  dx = Math.max(-2.5, Math.min(2.5, dx));
155  dy = Math.max(-2.5, Math.min(2.5, dy));
156 
157  // set new positions
158  for (int i = 0; i < N; i++) {
159  N v = fNodes.get(i);
160  double halfWidth = v.getWidth() / 2;
161  double halfHeight = v.getHeight() / 2;
162 
163  // move each node towards center of drawing area and keep
164  // it within bounds
165  double x = Math.max(fMarginX + halfWidth, Math.min(fWidth - fMarginX - halfWidth, fXn[i] + dx));
166  double y = Math.max(fMarginY + halfHeight, Math.min(fHeight - fMarginY - halfHeight, fYn[i] + dy));
167  v.setCenter(x, y);
168  }
169  }
