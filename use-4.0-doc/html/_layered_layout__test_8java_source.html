39  public static <N, E extends DirectedEdge<N>> Object[] removeCycles (final DirectedGraph<N, E> g) {
40  LinkedList<N> sl = new LinkedList<N>();
41  LinkedList<N> sr = new LinkedList<N>();
42  ArrayList<N> rest = new ArrayList<N>();
43 
44  Iterator<N> nodes = g.iterator();
45  while (nodes.hasNext() ) {
46  N node = nodes.next();
47  int ind = g.numIncomingEdges(node);
48  int outd = g.numOutgoingEdges(node);
49  if (outd == 0 ) {
50  // prepend sinks to sr
51  sr.addFirst(node);
52  } else if (ind == 0 ) {
53  // append source to sl
54  sl.addLast(node);
55  } else {
56  rest.add(node);
57  }
58  }
