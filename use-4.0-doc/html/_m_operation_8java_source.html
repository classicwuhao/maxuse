196  public void setExpression(Expression expr) throws MInvalidModelException {
197  // If no result type is set, use type of the expression
198  if (fResultType == null) {
199  fResultType = expr.type();
200  } else if (expr.type() == null) {
201  throw new MInvalidModelException("The operation " + StringUtil.inQuotes(this.cls().name() + "." + this.name()) +
202  " does not have a declared result type and the result type of the defined expression could not be calcuated. This" +
203  " can happen when an operation without a declared result type is calling itself recursively.");
204  } else if (! expr.type().conformsTo(fResultType) ) {
205  throw new MInvalidModelException("Expression type `" +
206  expr.type() +
207  "' does not match declared result type `" + fResultType + "'.");
208  }
209  fExpr = expr;

294  public boolean isValidOverrideOf(MOperation op) {
295  if (!this.name().equals(op.name())) return false;
296 
297  if (this.fResultType == null) {
298  if (op.fResultType != null) return false;
299  } else {
300  if (!this.fResultType.conformsTo(op.fResultType))
301  return false;
302  }
303 
304  if (this.fVarDeclList.size() != op.fVarDeclList.size())
305  return false;
306 
307  int index = 0;
308  for (VarDecl opVar : op.fVarDeclList) {
309  if (!opVar.type().conformsTo(this.fVarDeclList.varDecl(index).type()))
310  return false;
311  ++index;
312  }
313 
314  return true;
315  }
