176  public void calcLinks(){
177  // Associations and connected Objects
178  fAssoc = new Vector<String>();
179  fConnectedObject = new TreeMap<String, Set<MObject>>();
180  MClass objClass = fObject.cls();
181 
182  // get associations this object might be participating in
183  Set<MAssociation> assocSet = objClass.allAssociations();
184 
185  for (MAssociation assoc : assocSet) {
186  MLinkSet linkSet = fSystem.state().linksOfAssociation(assoc) ;
187  // check all association ends the objects' class is
188  // connected to
189  Set<MObject> connectedObjects = new HashSet<MObject>();
190 
191  for (MLink link : linkSet.links()) {
192  if(link.linkedObjects().contains(fObject)) {
193  //add association if it's still not in the list
194  if(!fAssoc.contains(assoc.toString()))
195  fAssoc.add(assoc.toString());
196 
197  Set<MObject> convertToSet = new HashSet<MObject>(link.linkedObjects());
198 
199  // remove fObject from dummy if it is not connected reflexive
200  MObject[] objSet = {fObject};
201  if(!linkSet.hasLinkBetweenObjects(objSet))
202  convertToSet.remove(fObject);
203 
204  connectedObjects.addAll(convertToSet);
205  }
206  }
207 
208  fConnectedObject.put(assoc.toString(), connectedObjects);
209 
210  }
