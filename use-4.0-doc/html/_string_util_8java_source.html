547  switch ( c ) {

550  case '\r' : return "\\r";
551  case '\\' : return "\\\\";

553  case '"' : return ( quoteToEscape == '"' ) ? "\"" : "\\\"";

558  }
559  else {
560  return "\\" + Integer.toString(c,8);
561  }
562  }
563  else {
564  return String.valueOf(c);
565  }
566  }
