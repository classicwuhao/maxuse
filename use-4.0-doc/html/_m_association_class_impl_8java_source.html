87  MAssociation assoc = src.association();
88  if (assoc.associationEnds().size() > 2)

93  otherEnd = (MAssociationEnd) assoc.associationEnds().get(1);
94  else
95  otherEnd = (MAssociationEnd) assoc.associationEnds().get(0);
96 
97  if (src.hasQualifiers()) {
98  if (qualifiedAccess) {
99  if (otherEnd.multiplicity().isCollection()) {
100  if ( otherEnd.isOrdered() )

102  else
103  return TypeFactory.mkSet( this );
104  }
105  } else {
106  if ( otherEnd.isOrdered() )
107  return TypeFactory.mkSequence( this );
108  else

113  if (otherEnd.isOrdered())

118  }
