117  private void initState() {
118 
119  // read Classes
120  Collection<MClass> allClasses = fSystem.model().classes();
121  for (MClass cls : allClasses) {
122  fClassDiagram.addClass( cls );
123  }
124 
125  // read Enumerations
126  Collection<EnumType> allEnums = fSystem.model().enumTypes();
127  for (EnumType enumeration : allEnums) {
128  fClassDiagram.addEnum( enumeration );
129  }
130 
131  // read signals
132  for (MSignal s : fSystem.model().getSignals()) {
133  fClassDiagram.addSignal( s );
134  }
135 
136  // read generalizations
137  DirectedGraph<MClassifier, MGeneralization> genGraph = fSystem.model().generalizationGraph();
138  Iterator<MGeneralization> edgeIter = genGraph.edgeIterator();
139  while ( edgeIter.hasNext() ) {
140  MGeneralization gen = edgeIter.next();
141  fClassDiagram.addGeneralization( gen );
142  }
143 
144  // read Associations
145  Collection<MAssociation> allAssociations = fSystem.model().associations();
146  for (MAssociation assoc : allAssociations) {
147  fClassDiagram.addAssociation( assoc );
148  }
149 
150  fClassDiagram.initialize();
151  }
