95  public MModel gen(Context ctx) {
96  MModel model = ctx.modelFactory().createModel(fName.getText());
97  model.setFilename(ctx.filename());
98  ctx.setModel(model);
99 
100  this.genAnnotations(model);
101 
102  // (1a) add user-defined types to model
103  for (ASTEnumTypeDefinition e : fEnumTypeDefs) {
104  EnumType enm;
105  try {
106  enm = e.gen(ctx);
107  model.addEnumType(enm);
108  } catch (SemanticException ex) {
109  ctx.reportError(ex);
110  } catch (MInvalidModelException ex) {
111  ctx.reportError(fName, ex);
112  }
113  }
114 
115  // (1b) add empty classes to model
116  Iterator<ASTClass> cIt = fClasses.iterator();
117  while(cIt.hasNext()) {
118  ASTClass c = cIt.next();
119 
120  try {
121  MClass cls = c.genEmptyClass(ctx);
122  model.addClass(cls);
123  } catch (SemanticException ex) {
124  ctx.reportError(ex);
125  cIt.remove();
126  } catch (MInvalidModelException ex) {
127  ctx.reportError(fName, ex);
128  cIt.remove();
129  }
130  }
131 
132  // (1c) add empty association classes to model
133  Iterator<ASTAssociationClass> acIt = fAssociationClasses.iterator();
134  while ( acIt.hasNext() ) {
135  ASTAssociationClass ac = acIt.next();
136  try {
137  // The association class can just be added as a class so far,
138  // because to keep the order of generating a model.
139  // The association class will be added as an association in step 3b.
140  MAssociationClass assocCls = ac.genEmptyAssocClass( ctx );
141  model.addClass( assocCls );
142  } catch ( SemanticException ex ) {
143  ctx.reportError( ex );
144  acIt.remove();
145  } catch ( MInvalidModelException ex ) {
146  ctx.reportError( fName, ex );
147  acIt.remove();
148  }
149  }
150 
151  // (1c) add empty signals to model
152  {
153  Iterator<ASTSignal> iter = this.signals.iterator();
154  while (iter.hasNext()) {
155  ASTSignal s = iter.next();
156 
157  try {
158  MSignal signal = s.genEmptySignal(ctx);
159  model.addSignal(signal);
160  } catch (SemanticException ex) {
161  ctx.reportError( ex );
162  iter.remove();
163  } catch (MInvalidModelException e1) {
164  ctx.reportError( s.getName(), e1 );
165  iter.remove();
166  }
167  }
168  }
169 
170  // (2a) add attributes and set generalization
171  // relationships. The names of all classes are known at this
172  // point
173  for (ASTClass c : fClasses) {
174  c.genAttributesOperationSignaturesAndGenSpec(ctx);
175  }
176 
177  // (2b) add attributes and set generalization
178  // relationships of the association classes.
179  // The names of all classes are known at this point
180  for (ASTAssociationClass ac : fAssociationClasses) {
181  ac.genAttributesOperationSignaturesAndGenSpec( ctx );
182  }
183 
184  // (2c) add attributes and set generalization relationships
185  // of signals
186  for (ASTSignal s : signals) {
187  s.genAttributesAndGenSpec( ctx );
188  }
189 
190  // (3a) add associations. Classes are known and can be
191  // referenced by role names.
192  for (ASTAssociation a : fAssociations) {
193  try {
194  a.gen(ctx, model);
195  } catch (SemanticException ex) {
196  ctx.reportError(ex);
197  }
198  }
199 
200  for (ASTClass c : fClasses) {
201  c.genStateMachinesAndStates(ctx);
202  }
203 
204  for (ASTAssociationClass ac : fAssociationClasses) {
205  ac.genStateMachinesAndStates(ctx);
206  }
207 
208  // (3b) add association classes as associations.
209  // Classes are known and can be referenced by role names.
210  for (ASTAssociationClass ac : fAssociationClasses) {
211  try {
212  // The association class is now added as an association.
213  // It is added here to keep the order of generating a model.
214  // The association class is already added as a class in step 1c.
215  MAssociationClass assocClass = ac.genAssociation( ctx );
216  model.addAssociation( assocClass );
217  } catch ( SemanticException ex ) {
218  ctx.reportError( ex );
219  } catch ( MInvalidModelException ex ) {
220  ctx.reportError( fName, ex );
221  }
222  }
223 
224  // (3c) Generalization of association classes might leave out new
225  // rolenames. Add them from parent.
226  for (ASTAssociationClass ac : fAssociationClasses) {
227  try {
228  ac.genAssociationFinal( ctx );
229  } catch ( MInvalidModelException ex ) {
230  ctx.reportError( fName, ex );
231  }
232  }
233 
234 
235  // (3c) add associationEnd specific constraints, e. g. subsets
236  // Role names are known and can be subset
237  for (ASTAssociation a : fAssociations) {
238  try {
239  a.genEndConstraints(ctx);
240  } catch (SemanticException ex) {
241  ctx.reportError(ex);
242  }
243  }
244 
245  // (3c) add associationEnd specific constraints, e. g. subsets
246  // Role names are known and can be subset
247  for (ASTAssociationClass a : fAssociationClasses) {
248  try {
249  a.genEndConstraints(ctx);
250  } catch (SemanticException ex) {
251  ctx.reportError(ex);
252  }
253  }
254 
255  // (4a) generate bodies of association and non-association classes
256  // All class interfaces are known and association features
257  // are available for expressions.
258  for (ASTClass c : fClasses) {
259  c.genOperationBodiesAndDerivedAttributes(ctx);
260  }
261 
262  for (ASTAssociationClass ac : fAssociationClasses) {
263  ac.genOperationBodiesAndDerivedAttributes(ctx);
264  }
265 
266 
267  // (4b) generate constraints of association and non-association classes
268  // All class interfaces are known and association features
269  // are available for expressions.
270  for (ASTClass c : fClasses) {
271  c.genConstraints(ctx);
272  }
273 
274  for (ASTAssociationClass ac : fAssociationClasses) {
275  ac.genConstraints(ctx);
276  }
277 
278  // (5a) generate global constraints. All class interfaces are
279  // known and association features are available for
280  // expressions.
281  for (ASTConstraintDefinition c : fConstraints) {
282  c.gen(ctx);
283  }
284 
285  // (5b) generate pre-/postconditions.
286  for (ASTPrePost ppc : fPrePosts) {
287  try {
288  ppc.gen(ctx);
289  } catch (SemanticException ex) {
290  ctx.reportError(ex);
291  }
292  }
293 
294  // Gen transitions
295  for (ASTClass c : fClasses) {
296  c.genStateMachineTransitions(ctx);
297  }
298 
299  for (ASTAssociationClass ac : fAssociationClasses) {
300  ac.genStateMachineTransitions(ctx);
301  }
302 
303  return model;
304  }
