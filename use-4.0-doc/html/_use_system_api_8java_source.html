621  public Value evaluate(String expression) throws UseApiException {
622  StringWriter errBuffer = new StringWriter();
623  PrintWriter errorPrinter = new PrintWriter(errBuffer, true);
624 
625  WarningType orgWarningOclAny = Options.checkWarningsOclAnyInCollections();
626  WarningType orgWarningUnrelated = Options.checkWarningsUnrelatedTypes();
627 
628  Options.setCheckWarningsOclAnyInCollections(WarningType.IGNORE);
629  Options.setCheckWarningsUnrelatedTypes(WarningType.IGNORE);
630 
631  Expression expr = OCLCompiler.compileExpression(modelApi.getModel(), expression,
632  "USE Api", errorPrinter, system.varBindings());
633 
634  Options.setCheckWarningsOclAnyInCollections(orgWarningOclAny);
635  Options.setCheckWarningsUnrelatedTypes(orgWarningUnrelated);
636 
637  if (expr == null) {
638  throw new UseApiException("Invalid expression "
639  + StringUtil.inQuotes(expression) + "!\n"
640  + errBuffer.toString());
641  }
642 
643  Evaluator evaluator = new Evaluator(false);
644  Value val;
645 
646  try {
647  val = evaluator.eval(expr, system.state(), system.varBindings(),
648  NullPrintWriter.getInstance());

650  throw new UseApiException("Evaluation failed due to a multiplicity violation!", e);
651  }

664  private Value[][] getQualiferValuesFromExpression(
665  String[][] qualifierValueExpressions, MAssociation mAssoc)
666  throws UseApiException {
667  Value[][] qualifierValues;
668 
669  if (qualifierValueExpressions != null && qualifierValueExpressions.length > 0) {
670  int numEnds = mAssoc.associationEnds().size();
671  qualifierValues = new Value[numEnds][];
672  for (int iEnd = 0; iEnd < numEnds; ++iEnd) {
673  if (qualifierValueExpressions[iEnd] == null || qualifierValueExpressions[iEnd].length == 0) {
674  qualifierValues[iEnd] = new Value[0];
675  } else {
676  int numValues = qualifierValueExpressions[iEnd].length;
677  qualifierValues[iEnd] = new Value[numValues];
678  for (int iValue = 0; iValue < numValues; ++numValues) {
679  qualifierValues[iEnd][iValue] = evaluate(qualifierValueExpressions[iEnd][iValue]);
680  }
681  }
682  }
683  } else {
684  qualifierValues = new Value[0][0];
685  }
686  return qualifierValues;
687  }
