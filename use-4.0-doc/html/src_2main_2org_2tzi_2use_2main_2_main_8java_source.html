184  Log.debug("Starting gui without plugin runtime!");

193  new Class[] { Session.class, IRuntime.class });
194  Log.debug("Invoking method create with ["

197  create
198  .invoke(null,
199  new Object[] { session, pluginRuntime });
200  }
201  } catch (Exception e) {
202  Log.error("FATAL ERROR.", e);
203  System.exit(1);
204  }
205  }
