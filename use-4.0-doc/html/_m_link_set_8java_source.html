328  public MLink linkBetweenObjects(List<MObject> objects, List<List<Value>> qualifierValues) {
329  boolean equal = true;
330 
331  Collection<MLink> links = this.objectsToLinksMap.get(objects);
332 
333  for (MLink link : links) {
334  if (link.linkedObjects().equals(objects)) {
335  if (link.association().hasQualifiedEnds()) {
336  equal = true; // true, because if no qualifiers exists it is the correct link
337  for (int index = 0; index < link.association().associationEnds().size(); ++index) {
338  MAssociationEnd end = link.association().associationEnds().get(index);
339  if (!link.linkEnd(end).qualifierValuesEqual(qualifierValues.get(index)))
340  equal = false;
341  }
342 
343  if (equal)
344  return link;
345  } else {
346  return link;
347  }
348  }
349  }
350 
351  return null;
352  }
