79  ctx.enter(this);

81  Value res = UndefinedValue.instance;
82  Value v = fSourceExpr.eval(ctx);
83  if (v.isObject() ) {
84  ObjectValue ov = (ObjectValue) v;
85  MObject obj = ov.value();
86  // Note: an undefined value can still be casted to a
87  // subtype! See initialization of res above
88  if (obj.exists(ctx.postState()) && obj.cls().conformsTo(targetType) )
89  res = new ObjectValue((MClass) targetType, obj);
90  } else {
91  // value is fine if its type is equal or a subtype of the
92  // expected type
93  if (v.type().conformsTo(targetType) ) {
94  res = v;
95  }
96  }
97  ctx.exit(this, res);
