181  public void printOn(PrintWriter out) {
182  // determine column widths
183  int[] widths = new int[fColumns];
184  Iterator<Object> rowIter = fRows.iterator();
185  while (rowIter.hasNext() ) {
186  Object r = rowIter.next();
