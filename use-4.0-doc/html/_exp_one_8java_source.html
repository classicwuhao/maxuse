98  private int evalAux(int nesting, CollectionValue rangeVal, EvalContext ctx) {
99  int found = 0;
100 
101  for (Value elemVal : rangeVal) {
102 
103  // bind element variable to range element, if variable was
104  // declared
105  if (!fElemVarDecls.isEmpty())
106  ctx.pushVarBinding(fElemVarDecls.varDecl(nesting).name(), elemVal);
107 
108  if (!fElemVarDecls.isEmpty() && nesting < fElemVarDecls.size() - 1) {
109  // call recursively to iterate over range while
110  // assigning each value to each element variable
111  // eventually
112  found += evalAux(nesting + 1, rangeVal, ctx);
113 
114  if (found > 1) {
115  if (ctx.isEnableEvalTree()) {
116  // don't change the result value and continue iteration
117  evalAux(nesting + 1, rangeVal, ctx);
118  } else {
119  if (!fElemVarDecls.isEmpty())
120  ctx.popVarBinding();
121  break;
122  }
123  }
124  } else {
125  // evaluate predicate expression
126  Value queryVal = fQueryExp.eval(ctx);
127 
128  // undefined query values default to false
129  if (queryVal.isUndefined())
130  queryVal = BooleanValue.FALSE;
131 
132  // don't change the result value when expression is true
133  found += ((BooleanValue) queryVal).value() ? 1 : 0;
134 
135  if (found > 1 && !ctx.isEnableEvalTree()) {
136  if (!fElemVarDecls.isEmpty())
137  ctx.popVarBinding();
138  break;
139  }
140  }
141 
142  if (!fElemVarDecls.isEmpty())
143  ctx.popVarBinding();
144  }
145 
146  return found;
147  }
