112  public MTransition gen(Context ctx, MStateMachine sm) throws SemanticException {
113 
114  boolean hasErrors = false;
115  MTransition result;
116 
117  MVertex sourceVertex = sm.getDefaultRegion().getSubvertex(source.getText());
118 
119  if (sourceVertex == null) {
120  ctx.reportError(source, "Unknown source vertex " + StringUtil.inQuotes(source.getText()));
121  hasErrors = true;
122  }
123 
124  MVertex targetVertex = sm.getDefaultRegion().getSubvertex(target.getText());
125  if (targetVertex == null) {
126  ctx.reportError(source, "Unknown target vertex " + StringUtil.inQuotes(target.getText()));
127  hasErrors = true;
128  }
129 
130  if (hasErrors) return null;
131 
132  if (sm instanceof MProtocolStateMachine) {
133  result = new MProtocolTransition(sm.getDefaultRegion(), sourceVertex, targetVertex);
134  } else {
135  result = new MTransition(sm.getDefaultRegion(), sourceVertex, targetVertex);
136  }
137 
138  if (event != null) {
139  /*
140  * UML Spec: [6] An initial transition at the topmost level (region
141  * of a statemachine) either has no trigger or it has a trigger with
142  * the stereotype "create".
143  * self.source.oclIsKindOf(Pseudostate)
144  * implies (self.source.oclAsType(Pseudostate).kind = #initial)
145  * implies (self.source.container = self.stateMachine.top) implies
146  * ((self.trigger->isEmpty) or (self.trigger.stereotype.name = 'create'))
147  */
148  if (sourceVertex instanceof MPseudoState
149  && ((MPseudoState) sourceVertex).getKind() == MPseudoStateKind.initial) {
150  if (!event.getText().equals("create")) {
151  throw new SemanticException(event, "Initial transition must be unnamed or named " + StringUtil.inQuotes("create"));
152  }
153 
154  result.setTrigger(MTrigger.create(event.getText(), ctx.currentClass()));
155  }
156  } else if (operationName != null ){
157  // Operation call trigger?
158  MOperation op = ctx.currentClass().operation(operationName.getText(), true);
159 
160  if (op == null) {
161  ctx.reportError(source, "Unknown operation " + StringUtil.inQuotes(operationName.getText()));
162  hasErrors = true;
163  } else if (op.hasExpression()) {
164  ctx.reportError(source, "The query operation " + StringUtil.inQuotes(op.toString()) + " cannot be used as a transition event, because query operations are not allowed to change the system state.");
165  hasErrors = true;
166  }
167 
168  if (op != null) {
169  result.setTrigger(MTrigger.create(op));
170  }
171  }
172 
173  Expression prePostConditionExp = null;
174  if (preCondition != null) {
175  prePostConditionExp = genPrePost(result, ctx, sm, true, this.preCondition);
176  result.setGuard(prePostConditionExp);
177  }
178 
179  if (result instanceof MProtocolTransition && postCondition != null) {
180  prePostConditionExp = genPrePost(result, ctx, sm, false, this.postCondition);
181  ((MProtocolTransition)result).setPostCondition(prePostConditionExp);
182  }
183 
184  return result;
185  }

187  private Expression genPrePost(MTransition t, Context ctx, MStateMachine sm, boolean isPre, ASTExpression expr) throws SemanticException {
188  MClass cls = sm.getContext();
189  Expression conditionExp = null;
190 
191  // enter context variable into scope of invariant
192  Symtable vars = ctx.varTable();
193  vars.enterScope();
194 
195  try {
196  // create pseudo-variable "self"
197  vars.add("self", cls, null);
198  ctx.exprContext().push("self", cls);
199 
200  t.getTrigger().buildEnvironment(vars, ctx.exprContext(), isPre);
201 
202  ctx.setInsidePostCondition(!isPre);
203  conditionExp = expr.gen(ctx);
204  ctx.setInsidePostCondition(false);
205 
206  if (!conditionExp.type().isTypeOfBoolean()) {
207  throw new SemanticException(expr.getStartToken(), "A "
208  + (isPre ? "guard" : "post condition")
209  + " must be a boolean expression.");
210  }
211  } catch (NullPointerException ex) {
212  // Can be raised by MModel if the owning operation
213  // was not successfully generated.
214  return null;
215  }
216 
217  vars.exitScope();
218  ctx.exprContext().pop();
219 
220  return conditionExp;
221  }
