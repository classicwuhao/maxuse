164  HashSet<MObject> objects = new HashSet<MObject>();
165  Iterator<EdgeBase> it = this.diagram.getGraph().edgeIterator();
166  String name = node.name();
167

170

172  BinaryAssociationOrLinkEdge edge = (BinaryAssociationOrLinkEdge) o;
173 
174  if (edge.getAssociation() != null && edge.getAssociation().equals(node)) {
175  MObject mo = ((ObjectNodeActivity) (edge.source())).object();
176  if (!selectedObjectsOfAssociation.contains(mo)) {
177  objects.add(mo);
178  }
179  mo = ((ObjectNodeActivity) (edge.target())).object();
180  if (!selectedObjectsOfAssociation.contains(mo) && !objects.contains(mo)) {
181  objects.add(mo);
182  }
183  return objects;
184  }
185  }
186  }
