73  public Expression gen(Context ctx) throws SemanticException {
74  String opname = fOp.getText();
75  Expression res = null;
76  Expression range, expr;
77 
78  // check for empty range: do we have a context expression that
79  // is implicitly assumed to be the source expression?
80  if (fRange != null ) {
81  range = fRange.gen(ctx);
82  } else {
83  ExprContext ec = ctx.exprContext();
84  if (! ec.isEmpty() ) {
85  // construct source expression
86  ExprContext.Entry e = ec.peek();
87  range = new ExpVariable(e.fName, e.fType);
88  } else
89  throw new SemanticException(fOp, "Need a collection to apply `" +
90  opname + "'.");
91  }
92 
93  if (!range.type().isKindOfCollection(VoidHandling.INCLUDE_VOID))
94  throw new SemanticException(fOp,
95  "Source of `" + opname + "' expression must be a collection, " +
96  "found source expression of type `" + range.type() + "'.");
97 
98  VarDeclList declList = new VarDeclList(true);
99 
100  if (fDeclList.isEmpty() ) {
101  // when there are no explicit var decls, we declare an
102  // internal variable with the element type
103  ExprContext ec = ctx.exprContext();
104  CollectionType ct = (CollectionType) range.type();
105  String var = ec.push(ct.elemType());
106  expr = fExpr.gen(ctx);
107  ec.pop();
108 
109  // use the generated element variable
110  VarDecl decl = new VarDecl(var, ct.elemType());
111  declList.add(decl);
112  } else {
113  declList = fDeclList.gen(ctx, range);
114 
115  // enter declared variable into scope before evaluating
116  // the argument expression
117  Symtable vars = ctx.varTable();
118  vars.enterScope();
119  declList.addVariablesToSymtable(vars);
120  expr = fExpr.gen(ctx);
121  vars.exitScope();
122  }
123 
124  try {
125  Integer id = ParserHelper.queryIdentMap.get(opname);
126  if (id == null )
127  throw new SemanticException(fOp,
128  "Internal error: unknown query operation `" +
129  opname + "'.");
130 
131  int idval = id.intValue();
132  switch ( idval ) {
133  case ParserHelper.Q_SELECT_ID:
134  case ParserHelper.Q_COLLECT_ID:
135  case ParserHelper.Q_COLLECTNESTED_ID:
136  case ParserHelper.Q_REJECT_ID:
137  case ParserHelper.Q_ISUNIQUE_ID:
138  case ParserHelper.Q_SORTEDBY_ID:
139  case ParserHelper.Q_ANY_ID:
140  case ParserHelper.Q_CLOSURE_ID:
141  VarDecl decl;
142  if (declList.isEmpty() )
143  decl = null;
144  else if (declList.size() == 1 )
145  decl = declList.varDecl(0);
146  else
147  throw new SemanticException(fOp, "Only one element variable in " +
148  opname + " expression allowed.");
149  switch ( idval ) {
150  case ParserHelper.Q_SELECT_ID:
151  res = new ExpSelect(decl, range, expr);
152  break;
153  case ParserHelper.Q_COLLECTNESTED_ID:
154  res = new ExpCollectNested(decl, range, expr);
155  break;
156  case ParserHelper.Q_COLLECT_ID:
157  res = new ExpCollect(decl, range, expr);
158  break;
159  case ParserHelper.Q_REJECT_ID:
160  res = new ExpReject(decl, range, expr);
161  break;
162  case ParserHelper.Q_ISUNIQUE_ID:
163  res = new ExpIsUnique(decl, range, expr);
164  break;
165  case ParserHelper.Q_SORTEDBY_ID:
166  res = new ExpSortedBy(decl, range, expr);
167  break;
168  case ParserHelper.Q_ANY_ID:
169  res = new ExpAny(decl, range, expr);
170  break;
171  case ParserHelper.Q_CLOSURE_ID:
172  res = new ExpClosure(decl, range, expr);
173  break;
174  default:
175  // TODO: ignore or error on default?
176  }
177  break;
178  case ParserHelper.Q_ONE_ID:
179  res = new ExpOne(declList, range, expr);
180  break;
181  case ParserHelper.Q_EXISTS_ID:
182  res = new ExpExists(declList, range, expr);
183  break;
184  case ParserHelper.Q_FORALL_ID:
185  res = new ExpForAll(declList, range, expr);
186  break;
187  default:
188  // internal error
189  throw new SemanticException(fOp,
190  "Internal error: unknown query operation `" +
191  opname + "'.");
192  }
193  } catch (ExpInvalidException ex) {
194  throw new SemanticException(fOp, ex);
195  }
196  return res;
197  }
