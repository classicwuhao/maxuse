207  public Expression getExpressionForViolatingInstances() {
208 
209  try {
210  Expression allInstances = new ExpAllInstances(fClass);
211  Expression current = negated ? ExpStdOp.create("not", new Expression[]{ fBody }) : fBody ;
212  // For invariants with more than one iteration variable
213  // a simple Reject does not work, because it allows only one
214  // iteration variable. Therefore, we introduce the
215  // additional variables by using allInstances()->forAll(v|...)
216  // inside of the reject expression.
217  // For example
218  // context a1,a2,a3:A inv:
219  // true
220  // will be expanded to
221  // A.allInstances()->reject(a1|
222  // A.allInstances()->forAll(a2|
223  // A.allInstances()->forAll(a3| true)))
224  for (int i = fVars.size() - 1; i > 0; --i) {
225  current = new ExpForAll(fVars.varDecl(i), allInstances, current);
226  }
227  return new ExpReject(fVars.varDecl(0), allInstances, current);
228 
229  } catch (ExpInvalidException ex) {
230  throw new RuntimeException("getExpressionForViolatingInstances failed", ex);
231  }
232  }

241  public Expression getExpressionForSatisfyingInstances() {
242  try {
243  Expression allInstances = new ExpAllInstances(fClass);
244  Expression current = negated ? ExpStdOp.create("not", new Expression[]{ fBody }) : fBody ;
245 
246  // For invariants with more than one iteration variable
247  // a simple select does not work, because it allows only one
248  // iteration variable. Therefore, we introduce the
249  // additional variables by using allInstances()->forAll(v|...)
250  // inside of the select expression.
251  // For example
252  // context a1,a2,a3:A inv:
253  // true
254  // will be expanded to
255  // A.allInstances()->select(a1|
256  // A.allInstances()->forAll(a2|
257  // A.allInstances()->forAll(a3| true)))
258  for (int i = fVars.size() - 1; i > 0; --i) {
