99  public void genStateInvariant(Context ctx, MStateMachine sm) throws SemanticException {
100  if (stateInvariant == null || !(genVertex instanceof MState)) return;
101 
102  MState genState = (MState)genVertex;
103  MClass cls = sm.getContext();
104  Expression conditionExp = null;
105 
106  // enter context variable into scope of invariant
107  Symtable vars = ctx.varTable();
108  vars.enterScope();
109 
110  try {
111  // create pseudo-variable "self"
112  vars.add("self", cls, null);
113  ctx.exprContext().push("self", cls);
114 
115  conditionExp = stateInvariant.gen(ctx);
116 
117  if (!conditionExp.type().isTypeOfBoolean()) {
118  throw new SemanticException(stateInvariant.getStartToken(),
119  "A state invariant must be a boolean expression.");
120  }
121 
122  genState.setStateInvariant(conditionExp);
123  } catch (NullPointerException ex) {
124  // Can be raised by MModel if the owning operation
125  // was not successfully generated.
126  }
127 
128  vars.exitScope();
129  ctx.exprContext().pop();
130  }
