172  protected void calculateBounds() {
173  double width = nameRect.width;
174  double height = nameRect.height;
175 
176  if (fOpt.isShowAttributes()) {
177  width = Math.max(width, attributesRect.width);
178  height += attributesRect.height;

180

183  height += statesRect.height;

198  public void updateContent() {
199  String value;
200  MObjectState objState = fObject.state( fParent.system().state() );
201 
202  if (objState == null) return;
203 
204  for ( int i = 0; i < fAttributes.size(); i++ ) {
205  MAttribute attr = fAttributes.get( i );
206  Value val = objState.attributeValue( attr );
207 
208  if (val instanceof EnumValue) {
209  value = "#" + ((EnumValue)val).value();
210  } else {
211  value = val.toString();
212  }
213 
214  fValues[i] = (attr.isDerived() ? "/" : "") + attr.name() + "=" + value;
215  }
216 
217  for ( int i = 0; i < fStateMachines.size(); i++ ) {
218  MStateMachine sm = fStateMachines.get( i );
219  MState val = objState.getProtocolStateMachineInstance(sm).getCurrentState(sm.getDefaultRegion());
220 
221  fStateValues[i] = sm.name() + "=" + val.name();
222  }
223  }
