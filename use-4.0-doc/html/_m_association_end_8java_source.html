274  private Type getType(Type t, boolean sourceHasQualifiers, boolean qualifiedAccess) {
275  if (sourceHasQualifiers) {
276  if (qualifiedAccess) {
277  if (multiplicity().isCollection()) {
278  if ( isOrdered() )
279  t = TypeFactory.mkOrderedSet( t );
280  else
281  t = TypeFactory.mkSet( t );
282  }
283  } else {
284  if ( isOrdered() )
285  t = TypeFactory.mkSequence( t );
286  else
287  t = TypeFactory.mkBag( t );
288  }
289  } else {
290  if ( multiplicity().isCollection() ) {
291  if (this.hasQualifiers() && !this.isOrdered()) {
292  t = TypeFactory.mkBag( t );
293  } else if (this.hasQualifiers() && this.isOrdered()) {
294  t = TypeFactory.mkSequence( t );
295  } else if ( !this.hasQualifiers() && this.isOrdered() ) {
296  t = TypeFactory.mkOrderedSet( t );
297  } else {
298  t = TypeFactory.mkSet( t );
299  }
300  } else if ( association().associationEnds().size() > 2 ) {
301  t = TypeFactory.mkSet(t);
302  }
303  }
304 
305  return t;
306  }

317  private Type getRedefinedType(MClass sourceObjectType) {
318  Type resultType = null;
319 
320  // If another association end redefines this end with
321  // source from the type of sourceObjectExp or a subtype of sourceObjectExp
322  boolean foundDirectEnd = false;
323 
324  // No redefinition possible, because source type is the opposite of this end
325  if (this.getAllOtherAssociationEnds().get(0).cls().equals(sourceObjectType) )
326  return this.cls();
327 
328  for (MAssociationEnd redefiningEnd : this.getRedefiningEnds()) {
329  // TODO: n-ary
330  MAssociationEnd redefiningEndSrc = redefiningEnd.getAllOtherAssociationEnds().get(0);
331  Type redefiningEndSrcType = redefiningEndSrc.cls();
332 
333  if (redefiningEndSrcType.equals(sourceObjectType)) {
334  resultType = redefiningEnd.cls();
335  foundDirectEnd = true;
336  break;
337  }
338  }
339 
340  // No redefinitions with equal type found.
341  // We need to check inheritance
342  if (!foundDirectEnd) {
343  for (MClass parent : sourceObjectType.parents()) {
344  if (parent instanceof MClass) {
345  resultType = getRedefinedType(parent);
346 
347  if (resultType != null) {
348  break;
349  }
350  }
351  }
352  }
353 
354  return resultType;
355  }
