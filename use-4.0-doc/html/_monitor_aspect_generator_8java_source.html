77  while (true) {
78  String line = in.readLine();

92  private void replaceLine(String line) {
93  if (line.equals(FILE_HEADER) )
94  fOut.println("// Automatically generated from template file. DO NOT EDIT!");
95  else if (line.equals(INTRODUCTION_PARENTS) )
96  writeClassList();
97  else if (line.equals(INTRODUCTION_USEID_FIELD) )
98  writeClassList();
99  else if (line.equals(INTRODUCTION_USEID_SETTER) )
100  writeClassList();
101  else if (line.equals(INTRODUCTION_USEID_GETTER) )
102  writeClassList();
103  else if (line.equals(POINTCUT_CLASS) )
104  writePointcutClass();
105  else if (line.equals(POINTCUT_METHOD) )
106  writePointcutMethod();
107  else if (line.equals(POINTCUT_SETTER) )
108  writePointcutSetter();
109  else
110  throw new RuntimeException("template file contains unknown " +
111  "replacement token: `" + line + "'.");
112  }

195  Iterator<?> it = coll.iterator();
196  boolean isFirst = true;
