43 @SuppressWarnings("serial")
44 public class ObjectCountView extends BarChartView implements View {
45  private MSystem fSystem;
46  private MClass[] fClasses;

48 
49  public ObjectCountView(MSystem system) {
50  super("Class", "# Objects", Color.blue);
51  fSystem = system;
52  Collection<MClass> classes = fSystem.model().classes();
53  fClasses = classes.toArray(new MClass[0]);
54  Arrays.sort(fClasses);
55  setNames(fClasses);
56  fValues = new int[fClasses.length];
57  fSystem.getEventBus().register(this);
58  update();
59  }
60 
61  private void update() {
62  MSystemState systemState = fSystem.state();
63  for (int i = 0; i < fClasses.length; i++) {
64  fValues[i] = systemState.objectsOfClass(fClasses[i]).size();
65  }
66  setValues(fValues);
67  }
68
