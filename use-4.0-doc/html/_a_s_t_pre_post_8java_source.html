67  void gen(Context ctx) throws SemanticException {
68  // find class
69  MClass cls = ctx.model().getClass(fClassName.getText());
70  if (cls == null )
71  throw new SemanticException(fClassName,
72  "Undefined class `" + fClassName.getText() +
73  "'.");
74  // find operation in class
75  MOperation op = cls.operation(fOpName.getText(), false);
76  if (op == null )
77  throw new SemanticException(fOpName,
78  "Class `" + fClassName.getText() +
79  "' has no operation `" + fOpName.getText() +
80  "'.");
81 
82 
83  // map params to VarDeclList
84  VarDeclList varDeclList = new VarDeclList(false);
85 
86  for (ASTVariableDeclaration astDecl : fParamList) {
87  VarDecl decl = astDecl.gen(ctx);
88  try {
89  varDeclList.add(decl);
90  } catch (IllegalArgumentException ex) {
91  throw new SemanticException(astDecl.name(), "Redefinition of `" +
92  decl.name() + "'.");
93  }
94  }
95 
96  // check for identical signature
97  if (! op.paramList().equals(varDeclList) )
98  throw new SemanticException(fOpName, "This signature of operation `" +
99  fOpName.getText() +
100  "' does not match its previous declaration in class `" +
101  fClassName.getText() + "'.");
102 
103  Type resultType = null;
104  if (fResultType != null ) {
105  resultType = fResultType.gen(ctx);
106  if (! op.hasResultType() )
107  throw new SemanticException(fResultType.getStartToken(),
108  "Operation `" +
109  fOpName.getText() +
110  "' has no result type in its previous declaration in class `" +
111  fClassName.getText() + "'.");
112 
113  if (! resultType.equals(op.resultType()) )
114  throw new SemanticException(fResultType.getStartToken(),
115  "Expected result type `" + op.resultType() +
116  "', found `" + resultType + "'.");
117  } else {
118  if (op.hasResultType() )
119  throw new SemanticException(fOpName,
120  "Expected result type `" + op.resultType() + "'.");
121  }
122 
123  // enter parameters into scope of expression
124  Symtable vars = ctx.varTable();
125  vars.enterScope();
126 
127  for (ASTVariableDeclaration astDecl : fParamList) {
128  VarDecl decl = astDecl.gen(ctx);
129  vars.add(astDecl.name(), decl.type());
130  }
131 
132  for (ASTPrePostClause ppc : fPrePostClauses) {
133  ppc.gen(ctx, cls, op);
134  }
135 
136  vars.exitScope();
137  }
