152  public static Point2D intersectionPoint( Line2D line1, Line2D line2) {
153  // getting the intersection coordinate by vector arithmetic
154  // example for the top line:
155  // sx over sy + r * (tx - sx) over (ty - sy)
156  // = line2.getX1() over line2.getY1() + v * (line2.getX2() - line2.getX1()) over (line2.getY2() - line2.getY1())
157 
158  double numerator = 1.0;
159  double denominator = 1.0;
160 
161  numerator = ( line1.getX2() - line1.getX1()) * (line1.getY1() - line2.getY1()) + (line1.getY2() - line1.getY1()) * (line2.getX1() - line1.getX1());
162  denominator = (line2.getY2() - line2.getY1()) * (line1.getX2() - line1.getX1())
163  - (line2.getX2() - line2.getX1()) * (line1.getY2() - line1.getY1());
164 
165  double v = numerator / denominator;
166 
167  double intersection_X = line2.getX1() + v * (line2.getX2() - line2.getX1());
168  double intersection_Y = line2.getY1() + v * (line2.getY2() - line2.getY1());
169 
170  return new Point2D.Double( intersection_X, intersection_Y );
171  }

203  public static Point2D intersectWithCircle(Ellipse2D circle, Line2D intLine, Direction intersectionArea)
204  {
205  // The original code was published here:
206  // http://gravisto.fim.uni-passau.de/doc/guide/plugins/kinds/shape.html
207 
208  if (!FloatUtil.equals(circle.getWidth(), circle.getHeight()))
209  {
210  throw new IllegalArgumentException(
211  "First parameter must be a circle, i.e. height and width " +
212  "must be equal. Were: width=" + circle.getWidth() +
213  " height=" + circle.getHeight());
214  }
215 
216  double cx = circle.getCenterX();
217  double cy = circle.getCenterY();
218  double radius = circle.getWidth() / 2d;
219  double sx = intLine.getX1();
220  double sy = intLine.getY1();
221  double tx = intLine.getX2();
222  double ty = intLine.getY2();
223 
224  double a = ((tx - sx) * (tx - sx)) + ((ty - sy) * (ty - sy));
225  double b = 2d * (((tx - sx) * (sx - cx)) + ((ty - sy) * (sy - cy)));
226  double c = ((cx * cx) + (cy * cy) + (sx * sx) + (sy * sy)) -
227  (2d * ((cx * sx) + (cy * sy))) - (radius * radius);
228  double discr = (b * b) - (4d * a * c);
229 
230  if (discr < 0d)
231  {
232  // line does not intersect
233  return null;
234  }
235  else if (discr <= Double.MIN_VALUE) // epsilon test
236  {
237  // line is tangent
238  double u = (-b) / (2 * a);
239 
240  Point2D res = calculatePointOnLine(u, intLine.getP1(),
241  intLine.getP2());
242 
243  if (((intersectionArea == Direction.NORTH_WEST) && (res.getX() <= cx) &&
244  (res.getY() <= cy)) ||
245  ((intersectionArea == Direction.NORTH_EAST) && (res.getX() >= cx) &&
246  (res.getY() <= cy)) ||
247  ((intersectionArea == Direction.SOUTH_EAST) && (res.getX() >= cx) &&
248  (res.getY() >= cy)) ||
249  ((intersectionArea == Direction.SOUTH_WEST) && (res.getX() <= cx) &&
250  (res.getY() >= cy)))
251  {
252  return res;
253  }
254  }
255  else
256  {
257  double discrsqr = Math.sqrt(discr);
258  double u1 = (-b + discrsqr) / (2d * a); // first result
259  double u2 = (-b - discrsqr) / (2d * a); // second result
260 
261  // there should be only one intersection point ...
262  if ((0d <= u1) && (u1 <= 1.0001d))
263  {
264  Point2D res = calculatePointOnLine(u1, intLine.getP1(),
265  intLine.getP2());
266 
267  if (((intersectionArea == Direction.NORTH_WEST) && (res.getX() <= cx) &&
268  (res.getY() <= cy)) ||
269  ((intersectionArea == Direction.NORTH_EAST) && (res.getX() >= cx) &&
270  (res.getY() <= cy)) ||
271  ((intersectionArea == Direction.SOUTH_EAST) && (res.getX() >= cx) &&
272  (res.getY() >= cy)) ||
273  ((intersectionArea == Direction.SOUTH_WEST) && (res.getX() <= cx) &&
274  (res.getY() >= cy)))
275  {
276  return res;
277  }
278  }
279 
280  if ((0d <= u2) && (u2 <= 1d))
281  {
282  Point2D res = calculatePointOnLine(u2, intLine.getP1(),
283  intLine.getP2());
284 
285  if (((intersectionArea == Direction.NORTH_WEST) && (res.getX() <= cx) &&
286  (res.getY() <= cy)) ||
287  ((intersectionArea == Direction.NORTH_EAST) && (res.getX() >= cx) &&
288  (res.getY() <= cy)) ||
289  ((intersectionArea == Direction.SOUTH_EAST) && (res.getX() >= cx) &&
290  (res.getY() >= cy)) ||
291  ((intersectionArea == Direction.SOUTH_WEST) && (res.getX() <= cx) &&
292  (res.getY() >= cy)))
293  {
294  return res;
295  }
296  }
297  }
298 
299  return null;
300  }

316  public static Point2D intersectionPoint(Ellipse2D circle, Point2D point) {
317  if (!FloatUtil.equals(circle.getWidth(), circle.getHeight()))
318  {
319  throw new IllegalArgumentException(
320  "First parameter must be a circle, i.e. height and width " +
321  "must be equal. Were: width=" + circle.getWidth() +
322  " height=" + circle.getHeight());
323  }
324 
325  double dX = point.getX() - circle.getCenterX();
326  double dY = point.getY() - circle.getCenterY();
327  double tTan = dY / dX;
328  double arcT = Math.atan(tTan);
329  double tSin = (Double.isInfinite(tTan) ? 1 : Math.sin(arcT));
330  double tCos = (Double.isInfinite(tTan) ? 0 : Math.cos(arcT));
331 
332  tCos = Math.copySign(tCos, dX);
333  tSin =Math.copySign(tSin, dY);
334 
335  double R = circle.getHeight() / 2;
336  Point2D.Double result = new Point2D.Double();
337  result.x = circle.getCenterX() + R * tCos;
338  result.y = circle.getCenterY() + R * tSin;
339 
340  return result;
341  }

362  public static Point2D intersectionPoint( Shape s, Point2D source, Point2D target, boolean enlargeLine ) {
363  Line2D line = new Line2D.Double(source, target);
364 
365  if (enlargeLine) {
366  Rectangle2D b = s.getBounds2D();
367  double dist = Math.max(b.getWidth(), b.getHeight()) * 2;
368  if (FloatUtil.equals(dist, 0.0d))
369  return source;
370  line = enlargeLine(line, dist);
371  }
372 
373  PathIterator i = s.getPathIterator(null);
374  Point2D.Double p1 = new Point2D.Double();
375  Point2D.Double p2 = new Point2D.Double();
376 
377  double[] coordinates = new double[6];
378  Line2D line2 = new Line2D.Double();
379  double startX = 0;
380  double startY = 0;
381 
382  while (!i.isDone()) {
383  int type = i.currentSegment(coordinates);
384  switch (type) {
385  case PathIterator.SEG_MOVETO:
386  p2.x = startX = coordinates[0];
387  p2.y = startY = coordinates[1];
388  break;
389  case PathIterator.SEG_LINETO:
390  case PathIterator.SEG_QUADTO:
391  case PathIterator.SEG_CUBICTO:
392  p1.x = p2.x;
393  p1.y = p2.y;
394  p2.x = coordinates[0];
395  p2.y = coordinates[1];
396 
397  line2.setLine(p1, p2);
398  if ( line.intersectsLine( line2 ) ) {
399  return Util.intersectionPoint( line, line2 );
400  }
401  break;
402  case PathIterator.SEG_CLOSE:
403  p1.x = p2.x;
404  p1.y = p2.y;
405  p2.x = startX;
406  p2.y = startY;
407 
408  line2.setLine(p1, p2);
409  if ( line.intersectsLine( line2 ) ) {
410  return Util.intersectionPoint( line, line2 );
411  }
412  }
413 
414  i.next();
415  }
416 
417  return source;
418  }
