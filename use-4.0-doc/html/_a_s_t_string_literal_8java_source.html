72  private void convertString() {
73  char[] strArr = fValue.toCharArray();
74  boolean escape = false;
75  StringBuffer buf = new StringBuffer();
76  for (int i = 0; i < strArr.length; ++i) {
77  if (escape) {
78  if (strArr[i] == 'b') {
79  buf.append('\b');
80  } else if (strArr[i] == 't') {
81  buf.append('\t');
82  } else if (strArr[i] == 'n') {
83  buf.append('\n');
84  } else if (strArr[i] == 'r') {
85  buf.append('\r');
86  } else if (strArr[i] == 'f') {
87  buf.append('\f');
88  } else if (strArr[i] == 'u') {
89  // Unicode escape
90  int utf = Integer.parseInt(fValue.substring(i + 1, i + 5), 16);
91  buf.append((char)utf);
92  i += 4;
93  } else if (Character.isDigit(strArr[i])) {
94  // Octal escape
95  int j = 0;
96  for (j = 1; (j < 3) && (i + j < strArr.length); ++j) {
97  if (!Character.isDigit(strArr[i+j]))
98  break;
99  }
100  int octal = Integer.parseInt(fValue.substring(i, i + j), 8);
101  buf.append((char)octal);
102  i += j-1;
103  } else {
104  buf.append(strArr[i]);
105  }
106  escape = false;
107  } else if (strArr[i] == '\\') {
108  escape = true;
109  } else {
110  buf.append(strArr[i]);
111  }
112  }
113  fValue = buf.toString();
114  }
