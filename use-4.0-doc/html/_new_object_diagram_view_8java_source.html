270  void deleteObjects(Set<MObject> objects) {
271  MSequenceStatement sequence = new MSequenceStatement();
272 
273  for (MObject obj : objects) {
274  sequence.appendStatement(
275  new MObjectDestructionStatement(obj.value()));
276  }
277 
278  if (sequence.isEmpty()) {
279  return;
280  }
281  try {
282  fSystem.execute(sequence.simplify());
283  } catch (MSystemException e) {
284  JOptionPane.showMessageDialog(
285  fMainWindow,
286  e.getMessage(),
287  "Error",
288  JOptionPane.ERROR_MESSAGE);
