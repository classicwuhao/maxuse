45  public GInstruction createIfMatches( List<Object> param, MModel model ) {
46  // param is a list over Strings or GValueInstruction.

48 
49  if (param.isEmpty() )
50  return null;

52  Object first = param.get(0);

54 
55  if (!matches(first, rest, model))
56  return null;

59

68  private boolean matches( Object first, List<Object> rest, MModel model ) {
69  if (!(first instanceof String))

73  return false;

76
