169  private void apply() {
170 
171  for (JTable table : this.tables) {
172  if (table != null && table.isEditing())
173  table.getCellEditor().stopCellEditing();
174  }
175 
176  List<List<Value>> qualifier = new ArrayList<>();
177  for (int i = 0; i < this.tableModels.length; ++i) {
178  TableModel model = this.tableModels[i];
179  if (model == null) {
180  qualifier.add(Collections.<Value>emptyList());
181  } else {
182  List<Value> endValues = getEndValues(model);
183 
184  if (endValues == null) return;
185 
186  qualifier.add(endValues);
187  }
188  }
189 
190  try {
191  system.execute(
192  new MLinkInsertionStatement(association, participants, qualifier));
193 
194  } catch (MSystemException e) {
195  JOptionPane.showMessageDialog(
196  getParent(),
197  e.getMessage(),
198  "Error",
199  JOptionPane.ERROR_MESSAGE);
200  }
201 
202  this.dispose();
203  }

208  protected List<Value> getEndValues(TableModel model) {
209  List<Value> endValues = new ArrayList<Value>();
210  for (int i2 = 0; i2 < model.getValues().length; ++i2) {
211  String value = model.getValues()[i2];
212 
213  if (value == null) {
214  JOptionPane.showMessageDialog(
215  this,
216  "Please provide a value for the qualifier " + model.getQualifierNames()[i2],
217  "Error",
218  JOptionPane.ERROR_MESSAGE);
219  return null;
220  }
221 
222  StringWriter errorOutput = new StringWriter();
223  Expression valueAsExpression =
224  OCLCompiler.compileExpression(
225  system.model(),
226  system.state(),
227  value,
228  "<input>",
229  new PrintWriter(errorOutput, true),
230  system.varBindings());
231 
232  if (valueAsExpression == null) {
233  JOptionPane.showMessageDialog(
234  this,
235  errorOutput,
236  "Error",
237  JOptionPane.ERROR_MESSAGE);
238  return null;
239  }
240 
241  try {
242  // evaluate it with current system state
243  Evaluator evaluator = new Evaluator(true);
244  Value val = evaluator.eval(valueAsExpression, system.state(), system.varBindings());
245  endValues.add(val);
246  } catch (MultiplicityViolationException e) {
247  JOptionPane.showMessageDialog(
248  this,
249  "Could not evaluate. " + e.getMessage(),
250  "Error",
251  JOptionPane.ERROR_MESSAGE);
252  return null;
253  }
254  }
255 
256  return endValues;
257  }
