103  public void addAttribute(MAttribute attr) throws MInvalidModelException {
104  // check if there already exists an attribute with the
105  // specified name in this class
106  String attrName = attr.name();
107  if (fAttributes.containsKey(attrName) )
108  throw new MInvalidModelException("Class `" + name() +
109  "' already contains an attribute named `" + attrName + "'.");
110 
111  // check if attribute of same name is already defined in a
112  // superclass
113  if (attribute(attrName, true) != null )
114  throw new MInvalidModelException("Attribute with name `" +
115  attrName +
116  "' already defined in a superclass.");
117 
118  // check if attribute of same name is defined in a subclass
119  for (MClass cls : allChildren() ) {
120  if (cls.attribute(attrName, false) != null )
121  throw new MInvalidModelException("Attribute `" +
122  attrName + "' conflicts with existing attribute " +
123  "in class `" + cls.name() + "'.");
124  }
125 
126  // add attribute
127  fAttributes.put(attr.name(), attr);
128  attr.setOwner(this);
129  }

187  public void addOperation(MOperation op) throws MInvalidModelException {
188  // check if there already exists an operation with the same
189  // name in this class
190  String opname = op.name();
191  if (fOperations.containsKey(opname) )
192  throw new MInvalidModelException("Class `" + name() +
193  "' already contains an operation named `" + opname + "'.");
194 
195  // check if operation with same signature is already defined
196  // in a superclass. A redefinition is only allowed with same
197  // number and types of arguments.
198  MOperation superOp = operation(opname, true);
199  if (superOp != null ) {
200  if (!op.isValidOverrideOf(superOp) )
201  throw new MInvalidModelException("Redefinition of operation `" + superOp +
202  "' requires same number and type of arguments.");
203  }
204 
205  // check if operation with same signature is already defined
206  // in a subclass. A redefinition is only allowed with same
207  // number and types of arguments.
208  for (MClass cls : allChildren()) {
209  MOperation subOp = cls.operation(opname, false);
210  if (subOp != null )
211  if (!subOp.isValidOverrideOf(op) )
212  throw new MInvalidModelException("Operation `" + op +
213  "' does not match its redefinition in class `" +
214  cls + "'.");
215  }
216 
217  fOperations.put(op.name(), op);
218  fVTableOperations.put(op.name(), op);
219 
220  op.setClass(this);
221  }

261  public MOperation operation(String name, boolean searchInherited) {
262  MOperation op;
263 
264  if (searchInherited) {
265  op = fVTableOperations.get(name);
266 
267  if (op != null)
268  return op;
269 
270  for (MClass cls : parents()) {
271  op = cls.operation(name, false);
272  if (op != null) {
273  fVTableOperations.put(name, op);
274  return op;
275  }
276  }
277  // FIXME: The compiler has to check a unique binding in case of multiple inheritance
278  for (MClass cls : parents()) {
279  op = cls.operation(name, true);
280  if (op != null) {
281  fVTableOperations.put(name, op);
282  return op;
283  }
284  }
285  } else{
286  op = fOperations.get(name);
287  }
288 
289  return op;
290  }
