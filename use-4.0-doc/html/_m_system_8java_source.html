729  private void doTransition(EvalContext ctx, MOperationCall operationCall) throws MSystemException {
730  if (!operationCall.hasPossibleTransitions())
731  return;
732 
733  MObjectState objState = operationCall.getSelf().state(fCurrentState);
734 
735  if (ctx == null) {
736  VarBindings b = fVariableEnvironment.constructVarBindings();
737  ctx = new EvalContext(operationCall.getPreState(), fCurrentState, b, null, "");
738  }
739 
740  List<TransitionResult> toExecute = new LinkedList<>();
741 
742  for (MProtocolStateMachineInstance psm : objState.getProtocolStateMachinesInstances()) {
743  Map<MRegion, Set<MTransition>> possibleTransitions = operationCall.getPossibleTransitions(psm);
744 
745  // This state machine might not be related to the call
746  if (possibleTransitions == null)
747  continue;
748 
749  Map<MTransition, TransitionResult> results = new HashMap<>();
750 
751  boolean allRegionsValid = true;
752 
753  for (Map.Entry<MRegion, Set<MTransition>> entry : possibleTransitions.entrySet()) {
754  // No transition in this region
755  // FIXME: Check semantics of regions
756  if (entry.getValue().isEmpty())
757  continue;
758 
759  int numValidTransitions = 0;
760  for (MTransition t : entry.getValue()) {
761  TransitionResult r = psm.evaluateTransition(t, ctx, operationCall);
762  if (r.wasSuccessfull()) {
763  numValidTransitions++;
764  toExecute.add(r);
765  }
766  results.put(t, r);
767  }
768 
769  allRegionsValid = allRegionsValid && (numValidTransitions == 1);
770  }
771 
772  if (allRegionsValid) {
773  for (TransitionResult r : toExecute) {
774  psm.doTransition(r.getTransition());
775  operationCall.addExecutedTransition(psm, r.getTransition());
776  fireTransition(operationCall.getSelf(), psm.getProtocolStateMachine(), r.getTransition());
777  }
778  psm.setExecutingTransition(false);
779  } else {
780  throw new MSystemException("No valid transition after operation call " + inQuotes(operationCall) + " for protocol state machine "
781  + inQuotes(psm) + ". Possible transitions:\n" + StringUtil.fmtSeq(results.values(), "\n"));
782  }
783  }
784  }
