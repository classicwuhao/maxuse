108  public void genAttributesOperationSignaturesAndGenSpec(Context ctx) {
109  ctx.setCurrentClass(fClass);
110  if (fSuperClasses != null ) {
111 
112  for(Token id : fSuperClasses) {
113  // lookup parent by name
114  MClass parent = ctx.model().getClass(id.getText());
115 
116  if (parent == null )
117  ctx.reportError(id, "Undefined class `" + id.getText() + "'.");
118  else {
119  try {
120  checkForInheritanceConflicts(parent);
121  MGeneralization gen =
122  ctx.modelFactory().createGeneralization(fClass, parent);
123  ctx.model().addGeneralization(gen);
124  } catch (SemanticException ex) {
125  ctx.reportError(ex);
126  } catch (MInvalidModelException ex) {
127  ctx.reportError(fName, ex);
128  }
129  }
130  }
131  }
132 
133  // add attributes
134  for (ASTAttribute a : fAttributes ) {
135  try {
136  MAttribute attr = a.gen(ctx);
137  fClass.addAttribute(attr);
138  } catch (SemanticException ex) {
139  ctx.reportError(ex);
140  } catch (MInvalidModelException ex) {
141  ctx.reportError(fName, ex);
142  }
143  }
144 
145  // add operation signatures, expressions have to be generated
146  // later when all class interfaces are known
147  for (ASTOperation astOp : fOperations ) {
148  try {
149  MOperation op = astOp.genSignature(ctx);
150  fClass.addOperation(op);
151  } catch (SemanticException ex) {
152  ctx.reportError(ex);
153  astOp.setSignatureGenFailed();
154  } catch (MInvalidModelException ex) {
155  ctx.reportError(fName, ex);
156  astOp.setSignatureGenFailed();
157  }
158  }
159 
160  ctx.setCurrentClass(null);
161  }

164  private void checkForInheritanceConflicts(MClass parent) throws SemanticException {
165  //check for inheritance conflicts
166  for(MClassifier otherParentC : fClass.parents()) {
167  MClass otherParent = (MClass)otherParentC;
168  // check attributes
169  for(MAttribute otherParentAttribute : otherParent.allAttributes()) {
170 
171  for(MAttribute parentAttribute : parent.allAttributes()) {
172  if (parentAttribute.name().equals(otherParentAttribute.name()) && !parentAttribute.type().equals(otherParentAttribute.type())) {
173  throw new SemanticException(fName,"Inheritance conflict: attribute " + parentAttribute.name() +
174  " occurs with different types in the base classes of " +
175  fClass.name());
176  }
177  }
178  }
179 
180  // check operations
181  for(MOperation otherParentOperation : otherParent.allOperations()) {
182  for(MOperation parentOperation : parent.allOperations()) {
183  if (parentOperation.name().equals(otherParentOperation.name()) &&
184  !parentOperation.signature().equals(otherParentOperation.signature()) &&
185  // the operations could be overloaded
186  !(parentOperation.cls().isSubClassOf(otherParentOperation.cls()) || otherParentOperation.cls().isSubClassOf(parentOperation.cls()))) {
187  throw new SemanticException(fName,"Inheritance conflict: operation " + parentOperation.name() +
188  " occurs with different signatures in the base classes of " +
189  fClass.name());
190  }
191  }
192  }
193 
194  }

197  public void genOperationBodiesAndDerivedAttributes(Context ctx) {
198  ctx.setCurrentClass(fClass);
199 
200  // enter pseudo-variable "self" into scope of expressions
201  ctx.exprContext().push("self", fClass);
202  Symtable vars = ctx.varTable();
203  vars.enterScope();
204  try {
205  vars.add("self", fClass, null);
206  } catch (SemanticException ex) {
207  // fatal error?
208  throw new Error(ex);
209  }
210 
211  // generate operation bodies
212  for (ASTOperation astOp : fOperations) {
213  try {
214  astOp.genFinal(ctx);
215  } catch (SemanticException ex) {
216  ctx.reportError(ex);
217  }
218  }
219 
220  // generate derived attributes
221  for (ASTAttribute astAttribute : fAttributes) {
222  try {
223  astAttribute.genDerived(ctx);
224  astAttribute.genInit(ctx);
225  } catch (SemanticException ex) {
226  ctx.reportError(ex);
227  }
228  }

230  vars.exitScope();
231  ctx.exprContext().pop();
232  ctx.setCurrentClass(null);
