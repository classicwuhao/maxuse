112  public void initialize() {
113 
114  // First set possible offsets in deterministic order
115 
116  // Save handled edges, because a single iteration handles
117  // all edges between the same nodes.
118  Set<EdgeBase> handledEdges = new HashSet<EdgeBase>();
119  for (EdgeBase e : getUninitializedEdges()) {
120  if (handledEdges.contains(e)) continue;
121 
122  if (e.isReflexive()) {
123  handledEdges.addAll(setReflexivePositions(e.source()));
124  } else {
125  Set<EdgeBase> edges = this.edgesBetween(e.source(), e.target());
126  setEdgeOffset(edges);
127  handledEdges.addAll(edges);
128  }
129  }

192  protected Set<EdgeBase> setReflexivePositions(PlaceableNode node) {
193  // Check the next free reflexive position

195  Set<EdgeBase> connectedEdges = this.edgesBetween(node, node);

212  protected void setEdgeOffset(Set<EdgeBase> edges) {
213  if (edges.isEmpty()) return;
214 
215  // Keep the order of the edges
216  List<EdgeBase> orderedEdges = new ArrayList<EdgeBase>(edges);
217 
218  double singleOffset = 1.0 / (orderedEdges.size() + 1);
219  Collections.sort(orderedEdges, new Comparator<EdgeBase>() {
220  @Override
221  public int compare(EdgeBase o1, EdgeBase o2) {
222  return Double.compare(o1.getOffset(), o2.getOffset());
223  }
224  });
225 
226  PlaceableNode firstSource = orderedEdges.get(0).source();
227  for (int i = 0; i < orderedEdges.size(); ++i) {
228  EdgeBase edge = orderedEdges.get(i);
229  double offSet = (i + 1) * singleOffset - 0.5;
230  // Align for switched source and targets
231  offSet *= edge.source().equals(firstSource) ? 1 : -1;
232  edge.setOffset( offSet );
233  }
234  }
