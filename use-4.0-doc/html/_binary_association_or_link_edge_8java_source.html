152  protected void initializeWayPoints() {
153  // If only one qualifier, the source needs to be the end with the qualifier
154  if ( isReflexive() && fTargetEnd.hasQualifiers() && !fSourceEnd.hasQualifiers()) {

162  super.initializeWayPoints();
163 
164  if ( isReflexive() && fSourceEnd.hasQualifiers() && fTargetEnd.hasQualifiers() ) {
165  // For two qualifiers we need an additional way point
166 
167  WayPoint p;
168  StrategyRelativeToCorner s;
169  final Direction refPosition2;
170 
171  // Switch from WEST to EAST and vice versa
172  int value = getReflexivePosition().getValue();
173  value = value ^ Direction.WEST.getValue() ^ Direction.EAST.getValue();
174  refPosition2 = Direction.getDirection(value);
175 
176  // Set corner and make relative on x
177  final StrategyRelativeToCorner sRef3 = (StrategyRelativeToCorner)getWayPoint(WayPointType.REFLEXIVE_3).getStrategy();
178  sRef3.setCorner(refPosition2);
179  sRef3.setBasisX(DeltaBasis.RELATIVE);
180  sRef3.setDeltaX(0.6666);
181  sRef3.setMaxDeltaX(MAX_REFLEXIVE_SIZE * 0.6666);
182  sRef3.setMaxDeltaY(MAX_REFLEXIVE_SIZE * 0.6666);
183  Supplier<Double> sRef3Supplier = new Supplier<Double>() {
184  public Double get() {
185  return sRef3.getRelativeNode().getBounds().getWidth() + fTargetWayPoint.getWidth() * 2;
186  }
187  };
188  sRef3.setWidthCalculation(sRef3Supplier);
189 
190  p = new WayPoint(this, WayPointType.REFLEXIVE_4, fAssoc.name(), fOpt);
191  addWayPoint(fWayPoints.size() - 1, p);
192  final StrategyRelativeToCorner sRef4 = new StrategyRelativeToCorner(p, fTarget, refPosition2, 0.6666, DeltaBasis.RELATIVE, -0.3333, DeltaBasis.RELATIVE);
193  sRef4.setMaxDeltaX(MAX_REFLEXIVE_SIZE * 0.6666);
194  sRef4.setMaxDeltaY(MAX_REFLEXIVE_SIZE * 0.3333);
195  Supplier<Double> sRef4Supplier = new Supplier<Double>() {
196  public Double get() {
197  return sRef4.getRelativeNode().getBounds().getWidth() + fTargetWayPoint.getWidth() * 2;
198  }
199  };
200  sRef4.setWidthCalculation(sRef4Supplier);
201  p.setStrategy(sRef4);

203  ResetStrategy resetTarget = new ResetStrategy() {
204  @Override

206  StrategyRelativeToCorner ps = new StrategyRelativeToCorner(fTargetWayPoint, fTarget, refPosition2, 0, DeltaBasis.ABSOLUTE, -0.3333, DeltaBasis.RELATIVE);
207  ps.setMaxDeltaX(MAX_REFLEXIVE_SIZE * 0.3333);
208  return ps;
209  }
210  };
211 
212  // p1 Source stays on x bounds

216  }

234  }

342  @Override
343  void drawEdge( Graphics2D g ) {
344  WayPoint n1 = null;
345  Point2D p1 = null;
346 
347  WayPoint n2 = null;
348  Point2D p2 = null;
349 
350  int source = fSourceEnd.aggregationKind();
351  int target = fTargetEnd.aggregationKind();
352  boolean sourceNavigable = fSourceEnd.isExplicitNavigable();
353  boolean targetNavigable = fTargetEnd.isExplicitNavigable();
354 
355  // draw all line segments
356  if ( !fWayPoints.isEmpty() ) {
357  Iterator<WayPoint> it = fWayPoints.iterator();
358  n1 = it.next();
359 
360  n1.draw(g);
361 
362  while( it.hasNext() ) {
363  n2 = it.next();
364  p1 = n1.getLinePoint();
365  p2 = n2.getLinePoint();
366 
367  // draw nodeOnEdge
368  n2.draw(g);
369 
370  try {
371  if ( source != MAggregationKind.NONE
372  && ( n1.getSpecialID() == WayPointType.SOURCE || n1.getSpecialID() == WayPointType.TARGET )) {
373  if ( targetNavigable
374  && n1.getSpecialID() == WayPointType.SOURCE
375  && n2.getSpecialID() == WayPointType.TARGET ) {
376  drawAssociationKind( g, p2, p1, source + DIRECTED_EDGE );
377  } else {
378  drawAssociationKind( g, p2, p1, source );
379  }
380  } else if ( target != MAggregationKind.NONE
381  && ( n2.getSpecialID() == WayPointType.TARGET || n2.getSpecialID() == WayPointType.SOURCE )) {
382  if ( sourceNavigable
383  && n1.getSpecialID() == WayPointType.SOURCE
384  && n2.getSpecialID() == WayPointType.TARGET ) {
385  drawAssociationKind( g, p1, p2, target + DIRECTED_EDGE );
386  } else {
387  drawAssociationKind( g, p1, p2, target );
388  }
389  } else if ( sourceNavigable && n1.getSpecialID() == WayPointType.SOURCE ) {
390  drawAssociationKind( g, p2, p1, DIRECTED_EDGE );
391  } else if ( targetNavigable && n2.getSpecialID() == WayPointType.TARGET ) {
392  drawAssociationKind( g, p1, p2, DIRECTED_EDGE );
393  } else {
394  DirectedEdgeFactory.drawAssociation( g, (int) Math.round(p1.getX()),
395  (int) Math.round(p1.getY()),
396  (int) Math.round(p2.getX()),
397  (int) Math.round(p2.getY()), isDashed );
398  }
399  n1 = n2;
400  } catch ( Exception ex ) {
401  ex.printStackTrace();
402  }
403  }
404  }
