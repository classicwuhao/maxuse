50  public GInstruction gen(Context ctx) throws SemanticException {
51  // check whether the target variable was declared
52  Type type = ctx.varTable().lookup( fTarget.getText() );
53  if (type == null ) {
54  String err = "Variable "+fTarget.getText()+" must be declared.";
55  throw new SemanticException( fTarget, err );
56  }
57 
58  // the target variable must not be declared within a
59  // loop instruction
60  if (ctx.loopVarNames().contains(fTarget.getText()) ) {
61  String err = "Variable "+fTarget.getText()+" was declared " +
62  "within a loop instruction and can't be the target of an " +
63  "assignment.";
64  throw new SemanticException( fTarget, err );
65  }
66 
67  // fSource must be a GValueInstruction
68  GInstruction source = fSource.gen(ctx);
69  if (! (source instanceof GValueInstruction) ) {
70  String err = "`" + source + "' can't be the source of an " +
71  "assignment, because it has no return value.";
72  throw new SemanticException( fTarget, err );
73  }
74 
75  GValueInstruction valuesource = (GValueInstruction) source;
76  // The type of the source must be the same type
77  // or a subtype of the target variable.
78  if (! valuesource.type().conformsTo(type)) {
79  String err = "Invalid assignment: " +
80  "`"+fTarget.getText()+"' is of type " + type + "." +
81  "`" + valuesource + "' is of type " + valuesource.type() + ".";
82  throw new SemanticException( fTarget, err );
83  }
84 
85  GVariableAssignment assignment =
86  new GVariableAssignment( fTarget.getText(),
87  valuesource );
88 
89  return assignment;
90  }
91 }
