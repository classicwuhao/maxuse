235  @Override
236  public void restoreAdditionalInfo( PersistHelper helper, int version ) {
237  // Restore association name
238  if (fAssocName != null) {
239  helper.toFirstChild(LayoutTags.EDGEPROPERTY);
240  fAssocName.restorePlacementInfo(helper, version);
241  helper.toParent();
242  }
243 
244  if ( fHalfEdges != null ) {
245  for (EdgeBase e : fHalfEdges) {
246  if (e instanceof AssociationOrLinkPartEdge) {
247  AssociationOrLinkPartEdge partEdge = (AssociationOrLinkPartEdge)e;
248  String xpathExpr = "./edge[@type='" + LayoutTags.HALFEDGE + "' and name='" + partEdge.getName() + "']";
249 
250  helper.getNav().push();
251  AutoPilot ap = new AutoPilot(helper.getNav());
252  try {
253  ap.selectXPath(xpathExpr);
254 
255  try {
256  if (ap.evalXPath() != -1)
257  e.restorePlacementInfo(helper, version);
258 
259  } catch (XPathEvalException ex) {
260  helper.getLog().append(ex.getMessage());
261  } catch (NavException ex) {
262  helper.getLog().append(ex.getMessage());
263  }
264  } catch (XPathParseException ex) {
265  helper.getLog().append(ex.getMessage());
266  } finally {
267  helper.getNav().pop();
268  }
269  }
