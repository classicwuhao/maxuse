244  private void genAllSuperTypes(List<Part> selectedSupertypes, List<Part> remainingTypes, Set<Type> result) {
245  if (remainingTypes.isEmpty()) {
246  TupleType tt = new TupleType( selectedSupertypes.toArray(new Part[0]));
247  result.add(tt);
248  return;

251  Part p = remainingTypes.remove(0);

253  //FIXME: Inconsistent handling of OvlVoid. Maybe use parameter (noRecusrionOnVoid) for allSupertypes()
254  if (p.type().isVoidOrElementTypeIsVoid()) {
255  superTypes = new HashSet<Type>();

262  Part p1 = new Part(0, p.name(), t1);
263  selectedSupertypes.add(0, p1);
