95  public Point2D doCalculatePosition(PlaceableNode owner) {
96  if (Double.isNaN(attached.getX()) || Double.isNaN(attached.getY()))
97  return new Point2D.Double();
98 
99  Point2D nextCenter = attached.getAutopositionWayPoint().getCenter();
100 
101  if (Double.isNaN(nextCenter.getX()) || Double.isNaN(nextCenter.getY()))
102  nextCenter = attached.getCenter();
103 
104  PlaceableNode attachedBounds = attached.getCalculationNode();
105  // First build a rectangle were the center of the owner is allowed to stay
106  Rectangle2D ownerMovement = new Rectangle2D.Double(
107  attachedBounds.getX() - owner.getWidth() / 2 - offset_x,
108  attachedBounds.getY() - owner.getHeight() / 2 - offset_y,
109  attachedBounds.getWidth() + owner.getWidth() + offset_x,
110  attachedBounds.getHeight() + owner.getHeight() + offset_y);
111 
112  int y = (placement == Placement.TOP ? -1 : 1);
113  Point2D from = new Point2D.Double(attached.getCenter().getX() - owner.getWidth() / 2 - offset_x, attached.getCenter().getY() + y * (owner.getHeight() / 2 - offset_y));
114  Point2D to = new Point2D.Double(nextCenter.getX() - owner.getWidth() / 2 - offset_x, nextCenter.getY() + y * (owner.getHeight() / 2 - offset_y));
115 
116  Point2D intP = Util.intersectionPoint(ownerMovement, from, to, true);
117 
118  return intP;
119  }
