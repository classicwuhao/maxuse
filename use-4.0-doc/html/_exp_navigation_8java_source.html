93 
94  // if we don't have an object we can't navigate

96  // get the object
97  final ObjectValue objVal = (ObjectValue) val;
98  final MObject obj = objVal.value();
99  final MSystemState state = isPre() ? ctx.preState() : ctx.postState();
100  final Type resultType = type();
101 
102 
103  // get objects at association end
104  List<Value> qualifierValues = new LinkedList<Value>();
105 
106  for (Expression exp : this.qualifierExpressions) {
107  qualifierValues.add(exp.eval(ctx));
108  }
109 
110  List<MObject> objList = obj.getNavigableObjects(state, fSrc, fDst, qualifierValues);
111  if (resultType.isTypeOfClass() ) {
112  if (objList.size() > 1 )
113  throw new MultiplicityViolationException(
114  "expected link set size 1 at " +
115  "association end `" + fDst +
116  "', found: " +
117  objList.size());
118  if (objList.size() == 1 ) {
119  res = new ObjectValue((MClass)type(), objList.get(0));
120  }
121  } else if (resultType.isKindOfCollection(VoidHandling.EXCLUDE_VOID)) {
122  CollectionType ct = (CollectionType)resultType;
123  res = ct.createCollectionValue(oidsToObjectValues(state, objList));
124  } else
125  throw new RuntimeException("Unexpected association end type `" +
126  resultType + "'");
127  }
128 
129  ctx.exit(this, res);
130  return res;
131  }
