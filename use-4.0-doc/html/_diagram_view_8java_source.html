271  public void drawDiagram(Graphics g) {
272  synchronized (fGraph) {
273 
274  Graphics2D g2d = (Graphics2D)g;
275 
276  if ( fOpt.isDoAntiAliasing() ) {
277  Map<?,?> hints = (Map<?,?>)Toolkit.getDefaultToolkit().getDesktopProperty("awt.font.desktophints");
278  if (hints == null) {
279  g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
280  RenderingHints.VALUE_ANTIALIAS_ON);
281  } else {
282  g2d.setRenderingHints(hints);
283  }
284 
285  }
286 
287  Dimension d = getSize();
288  g2d.setColor(getBackground());
289  g2d.fillRect(0, 0, d.width, d.height);
290 
291  if (this.fOpt.showGrid()) {
292  drawGrid(g2d);
293  }
294 
295  fGraph.initialize();
296 
297  for (EdgeBase e : fGraph.getInvalidatedEdges()) {
298  e.calculateSize(g2d);
299  }
300 
301  for (PlaceableNode n : fGraph.getInvalidatedNodes()) {
302  // set rectangle size for nodes
303  // It is important to do this first, otherwise role names will
304  // not be dynamically moved after displaying attributes! And
305  // after hiding an object which is participating in a two link
306  // association, the two links won't be displayed in the right
307  // way.
308  // Set min height wrt. displayed qualifiers
309  int minHeight = 0;
310  Set<EdgeBase> myEdges = fGraph.allEdges(n);
311 
312  for (EdgeBase e : myEdges) {
313  int heightHint = 0;
314  if (e.source().equals(n))
315  heightHint += e.getSourceHeightHint();
316 
317  if (!e.isReflexive() && e.target().equals(n))
318  heightHint += e.getTargetHeightHint();
319 
320  if (heightHint > 0)
321  minHeight += heightHint + 4;
322  }
323  n.setRequiredHeight("BY_EDGES", minHeight + 4);
324  n.calculateSize(g2d);
325  }
326 
327  for (EdgeBase e : fGraph.getInvalidatedEdges()) {
328  e.updatePosition();
329  }
330  fGraph.clearInvalidated();
331 
332  // draw edges
333  // they need to be drawn first otherwise the association will
334  // be drawn above the nodes
335  Iterator<EdgeBase> edgeIterator = fGraph.edgeIterator();
336  while ( edgeIterator.hasNext() ) {
337  EdgeBase e = edgeIterator.next();
338  e.draw(g2d);
339  }
340 
341  double maxX = 0;
342  double maxY = 0;
343 
344  // draw nodes
345  Iterator<PlaceableNode> nodeIterator = fGraph.iterator();
346  while (nodeIterator.hasNext()) {
347  PlaceableNode n = nodeIterator.next();
348  n.draw(g2d);
349  maxX = Math.max(maxX, n.getX() + n.getWidth());
350  maxY = Math.max(maxY, n.getY() + n.getHeight());
351  }
352 
353  edgeIterator = fGraph.edgeIterator();
354  while(edgeIterator.hasNext()) {
355  EdgeBase e = edgeIterator.next();
356  PlaceableNode eastNode = e.getWayPointMostTo(Direction.EAST);
357  maxX = Math.max(maxX, eastNode.getX() + eastNode.getWidth());
358  PlaceableNode southNode = e.getWayPointMostTo(Direction.SOUTH);
359  maxY = Math.max(maxY, southNode.getY() + southNode.getHeight());
360  e.drawProperties(g2d);
361  }
362 
363  Dimension newDimension = new Dimension( (int)maxX + 5, (int)maxY + 5);
364  if (!newDimension.equals(this.getPreferredSize())) {
365  this.setPreferredSize(newDimension);
366  this.revalidate();
367  }
368  }

524  public PlaceableNode findNode(int x, int y) {
525  PlaceableNode res = null;
526 
527  synchronized (fGraph) {
528  // Find possible edge property that occupies the clicked position
529  Iterator<EdgeBase> eIter = fGraph.getVisibleEdgesIterator();
530 
531  while ( eIter.hasNext() ) {
532  EdgeBase e = eIter.next();
533 
534  res = e.findNode(x, y);
535 
536  if (res != null) {
537  return res;
538  }
539  }
540 
541  Iterator<PlaceableNode> nIter = fGraph.getVisibleNodesIterator();
542 
543  while (nIter.hasNext()) {
544  PlaceableNode n = nIter.next();
545  if (!n.isInitialized()) continue;
546 
547  if (n.occupies(x, y)) {
548  // Do not break here. We search in the same order
549  // which is used for drawing. There may be another
550  // node drawn on top of this node. That node should be
551  // picked.
552  res = n;
553  } else if (n.getRelatedNode(x, y) != null) {
554  res = n.getRelatedNode(x, y);
555  }
556  }
557  }
558 
559  return res;
560  }
