70  // This overrides super.owner
71  AttachedWayPoint owner = getOwner();
72 
73  // Initialization phase
74  if (owner.getAutopositionWayPoint() == null)
75  return new Point2D.Double();
76 
77  // Calculate the position using the offset specified in the edge
78  Point2D newCenter;
79 
80  PlaceableNode ourNode = owner.getCalculationNode();
81  // Target can be an arbitrary way point, like a user defined one
82  PlaceableNode otherNode = owner.getAutopositionWayPoint().getCalculationNode();
83 
84  double alpha = Util.calculateAngleBetween(ourNode.getCenter(), otherNode.getCenter());
85  double alphaRad = Math.toRadians(alpha);
86 
87  // This calculates the possible space to use by using a lot on the line to the target
88  double ourLength = ourNode.getCenter().distance(ourNode.getIntersectionCoordinate(alpha - 90)) * 2;
89  double otherLength = otherNode.getCenter().distance(otherNode.getIntersectionCoordinate(alpha - 90)) * 2;
90  // Source and target way points align the offset of the edge,
91  // to be able to calculate the values from both points
92  double offset = Math.min(ourLength, otherLength) * owner.getMyOffset();
93 
94  // Multiply by -1 to keep edges aligned clockwise, i. e., one
95  // edge is always "later" than the other.
96  double offsetX = Math.sin(alphaRad) * offset * -1;
97  double offsetY = Math.cos(alphaRad) * offset;
98 
99  Point2D sourceWithOffset = new Point2D.Double(
100  ourNode.getCenter().getX() + offsetX,
101  ourNode.getCenter().getY() + offsetY);
102 
103  Point2D targetWithOffset = new Point2D.Double(
104  otherNode.getCenter().getX() + offsetX,
105  otherNode.getCenter().getY() + offsetY);
106 
107  newCenter = owner.getAttachedNode().getIntersectionCoordinate(sourceWithOffset, targetWithOffset);
108 
109  return newCenter;
110  }
